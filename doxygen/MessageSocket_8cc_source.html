00237 void
00238 MessageSocket::readable()
00239 {
00240     // Try to read data from the kernel until there is no more left.
00241     while (true) {
00242         if (inbound.bytesRead < sizeof(Header)) {
00243             // Receiving header
00244             ssize_t bytesRead = read(
00245                 reinterpret_cast<char*>(&inbound.header) + inbound.bytesRead,
00246                 sizeof(Header) - inbound.bytesRead);
00247             if (bytesRead == -1) {
00248                 disconnect();
00249                 return;
00250             }
00251             inbound.bytesRead += size_t(bytesRead);
00252             if (inbound.bytesRead < sizeof(Header))
00253                 return;
00254             // Transition to receiving data
00255             inbound.header.fromBigEndian();
00256             if (inbound.header.fixed != 0xdaf4) {
00257                 WARNING("Disconnecting since message doesn't start with magic "
00258                         "0xdaf4 (first two bytes are 0x%02x)",
00259                         inbound.header.fixed);
00260                 disconnect();
00261             }
00262             if (inbound.header.version != 1) {
00263                 WARNING("Disconnecting since message uses version %u, but "
00264                         "this code only understands version 1",
00265                         inbound.header.version);
00266                 disconnect();
00267             }
00268             if (inbound.header.payloadLength > maxMessageLength) {
00269                 WARNING("Disconnecting since message is too long to receive "
00270                         "(message is %u bytes, limit is %u bytes)",
00271                         inbound.header.payloadLength, maxMessageLength);
00272                 disconnect();
00273                 return;
00274             }
00275             inbound.message.setData(new char[inbound.header.payloadLength],
00276                                     inbound.header.payloadLength,
00277                                     Core::Buffer::deleteArrayFn<char>);
00278         }
00279         // Don't use 'else' here; we want to check this branch for two reasons:
00280         // First, if there is a header with a length of 0, the socket won't be
00281         // readable, but we still need to process the message. Second, most of
00282         // the time the header will arrive with at least some data. It makes
00283         // sense to go ahead and try a non-blocking read, rather than going
00284         // back to the event loop.
00285         if (inbound.bytesRead >= sizeof(Header)) {
00286             // Receiving data
00287             size_t payloadBytesRead = inbound.bytesRead - sizeof(Header);
00288             ssize_t bytesRead = read(
00289                 (static_cast<char*>(inbound.message.getData()) +
00290                  payloadBytesRead),
00291                 inbound.header.payloadLength - payloadBytesRead);
00292             if (bytesRead == -1) {
00293                 disconnect();
00294                 return;
00295             }
00296             inbound.bytesRead += size_t(bytesRead);
00297             if (inbound.bytesRead < (sizeof(Header) +
00298                                      inbound.header.payloadLength)) {
00299                 return;
00300             }
00301             handler.handleReceivedMessage(inbound.header.messageId,
00302                                           std::move(inbound.message));
00303             // Transition to receiving header
00304             inbound.bytesRead = 0;
00305         }
00306     }

00310 MessageSocket::read(void* buf, size_t maxBytes)
00311 {
00312     ssize_t actual = recv(receiveSocket.fd, buf, maxBytes, MSG_DONTWAIT);
00313     if (actual > 0)
00314         return actual;
00315     if (actual == 0 || // peer performed orderly shutdown.

00317         return -1;

00320         return 0;

00324 void
00325 MessageSocket::writable()
00326 {
00327     // Each iteration of this loop tries to write one message
00328     // from outboundQueue.
00329     while (true) {
00330 
00331         // Get the next outbound message.
00332         Outbound outbound;
00333         int flags = MSG_DONTWAIT | MSG_NOSIGNAL;
00334         {
00335             std::lock_guard<Core::Mutex> lock(outboundQueueMutex);
00336             if (outboundQueue.empty())
00337                 return;
00338             outbound = std::move(outboundQueue.front());
00339             outboundQueue.pop_front();
00340             if (!outboundQueue.empty())
00341                 flags |= MSG_MORE;
00342         }
00343 
00344         // Use an iovec to send everything in one kernel call: one iov for the
00345         // header, another for the payload.
00346         enum { IOV_LEN = 2 };
00347         struct iovec iov[IOV_LEN];
00348         iov[0].iov_base = &outbound.header;
00349         iov[0].iov_len = sizeof(Header);
00350         iov[1].iov_base = outbound.message.getData();
00351         iov[1].iov_len = outbound.message.getLength();
00352 
00353         { // Skip the parts of the iovec that have already been sent.
00354             size_t bytesSent = outbound.bytesSent;
00355             for (uint32_t i = 0; i < IOV_LEN; ++i) {
00356                 iov[i].iov_base = (static_cast<char*>(iov[i].iov_base) +
00357                                    bytesSent);
00358                 if (bytesSent < iov[i].iov_len) {
00359                     iov[i].iov_len -= bytesSent;
00360                     break;
00361                 } else {
00362                     bytesSent -= iov[i].iov_len;
00363                     iov[i].iov_len = 0;
00364                 }
00365             }
00366         }
00367 
00368         struct msghdr msg;
00369         memset(&msg, 0, sizeof(msg));
00370         msg.msg_iov = iov;
00371         msg.msg_iovlen = IOV_LEN;
00372 
00373         // Do the actual send
00374         ssize_t bytesSent = sendmsg(sendSocket.fd, &msg, flags);
00375         if (bytesSent < 0) {
00376             if (errno == EAGAIN || errno == EWOULDBLOCK || errno == EINTR) {
00377                 // Wasn't able to send, try again later.
00378                 bytesSent = 0;
00379             } else if (errno == ECONNRESET || errno == EPIPE) {
00380                 // Connection closed; disconnect this end.
00381                 // This must be the last line to touch this object, in case
00382                 // handleDisconnect() deletes this object.
00383                 disconnect();
00384                 return;
00385             } else {
00386                 // Unexpected error.
00387                 PANIC("Error while writing to socket %d: %s",
00388                       sendSocket.fd, strerror(errno));
00389             }
00390         }
00391 
00392         // Sent successfully.
00393         outbound.bytesSent += size_t(bytesSent);
00394         if (outbound.bytesSent != (sizeof(Header) +
00395                                    outbound.message.getLength())) {
00396             sendSocketMonitor.setEvents(EPOLLOUT|EPOLLONESHOT);
00397             std::lock_guard<Core::Mutex> lockGuard(outboundQueueMutex);
00398             outboundQueue.emplace_front(std::move(outbound));
00399             return;
00400         }
00401     }

00219 MessageSocket::readable()
00220 {
00221     if (receiveSocket.fd < 0)
00222         return;
00223     // Try to read data from the kernel until there is no more left.
00224     while (true) {
00225         if (inbound.bytesRead < sizeof(Header)) {
00226             // Receiving header
00227             ssize_t bytesRead = read(
00228                 reinterpret_cast<char*>(&inbound.header) + inbound.bytesRead,
00229                 sizeof(Header) - inbound.bytesRead);
00230             if (bytesRead == -1) {
00231                 disconnect();
00232                 return;
00233             }
00234             inbound.bytesRead += bytesRead;
00235             if (inbound.bytesRead < sizeof(Header))
00236                 return;
00237             // Transition to receiving data
00238             inbound.header.fromBigEndian();
00239             if (inbound.header.payloadLength > maxMessageLength) {
00240                 WARNING("Dropping message that is too long to receive "
00241                         "(message is %u bytes, limit is %u bytes)",
00242                         inbound.header.payloadLength, maxMessageLength);
00243                 disconnect();
00244                 return;
00245             }
00246             inbound.message.setData(new char[inbound.header.payloadLength],
00247                                     inbound.header.payloadLength,
00248                                     Buffer::deleteArrayFn<char>);
00249         }
00250         // Don't use 'else' here; we want to check this branch for two reasons:
00251         // First, if there is a header with a length of 0, the socket won't be
00252         // readable, but we still need to process the message. Second, most of
00253         // the time the header will arrive with at least some data. It makes
00254         // sense to go ahead and try a non-blocking read, rather than going
00255         // back to the event loop.
00256         if (inbound.bytesRead >= sizeof(Header)) {
00257             // Receiving data
00258             size_t payloadBytesRead = inbound.bytesRead - sizeof(Header);
00259             ssize_t bytesRead = read(
00260                 (static_cast<char*>(inbound.message.getData()) +
00261                  payloadBytesRead),
00262                 inbound.header.payloadLength - payloadBytesRead);
00263             if (bytesRead == -1) {
00264                 disconnect();
00265                 return;
00266             }
00267             inbound.bytesRead += bytesRead;
00268             if (inbound.bytesRead < (sizeof(Header) +
00269                                      inbound.header.payloadLength)) {
00270                 return;
00271             }
00272             onReceiveMessage(inbound.header.messageId,
00273                              std::move(inbound.message));
00274             // Transition to receiving header
00275             inbound.bytesRead = 0;
00276         }

00295 void
00296 MessageSocket::writable()
00297 {
00298     if (receiveSocket.fd < 0)
00299         return;
00300     // Each iteration of this loop tries to write one message
00301     // from outboundQueue.
00302     while (true) {
00303 
00304         // Get the next outbound message.
00305         Outbound outbound;
00306         int flags = MSG_DONTWAIT | MSG_NOSIGNAL;
00307         {
00308             std::lock_guard<Core::Mutex> lock(outboundQueueMutex);
00309             if (outboundQueue.empty())
00310                 return;
00311             outbound = std::move(outboundQueue.front());
00312             outboundQueue.pop_front();
00313             if (!outboundQueue.empty())
00314                 flags |= MSG_MORE;
00315         }
00316 
00317         // Use an iovec to send everything in one kernel call: one iov for the
00318         // header, another for the payload.
00319         enum { IOV_LEN = 2 };
00320         struct iovec iov[IOV_LEN];
00321         iov[0].iov_base = &outbound.header;
00322         iov[0].iov_len = sizeof(Header);
00323         iov[1].iov_base = outbound.message.getData();
00324         iov[1].iov_len = outbound.message.getLength();
00325 
00326         { // Skip the parts of the iovec that have already been sent.
00327             size_t bytesSent = outbound.bytesSent;
00328             for (uint32_t i = 0; i < IOV_LEN; ++i) {
00329                 iov[i].iov_base = (static_cast<char*>(iov[i].iov_base) +
00330                                    bytesSent);
00331                 if (bytesSent < iov[i].iov_len) {
00332                     iov[i].iov_len -= bytesSent;
00333                     bytesSent = 0;
00334                     break;
00335                 } else {
00336                     bytesSent -= iov[i].iov_len;
00337                     iov[i].iov_len = 0;
00338                 }
00339             }
00340         }
00341 
00342         struct msghdr msg;
00343         memset(&msg, 0, sizeof(msg));
00344         msg.msg_iov = iov;
00345         msg.msg_iovlen = IOV_LEN;
00346 
00347         // Do the actual send
00348         ssize_t bytesSent = sendmsg(sendSocket.fd, &msg, flags);
00349         if (bytesSent < 0) {
00350             if (errno == EAGAIN || errno == EWOULDBLOCK || errno == EINTR) {
00351                 // Wasn't able to send, try again later.
00352                 bytesSent = 0;
00353             } else if (errno == ECONNRESET || errno == EPIPE) {
00354                 // Connection closed; disconnect this end.
00355                 // This must be the last line to touch this object, in case
00356                 // onDisconnect() deletes this object.
00357                 disconnect();
00358                 return;
00359             } else {
00360                 // Unexpected error.
00361                 PANIC("Error while writing to socket %d: %s",
00362                       sendSocket.fd, strerror(errno));
00363             }
00364         }
00365 
00366         // Sent successfully.
00367         outbound.bytesSent += bytesSent;
00368         if (outbound.bytesSent != (sizeof(Header) +
00369                                    outbound.message.getLength())) {
00370             sendSocket.setEvents(EPOLLOUT|EPOLLONESHOT);
00371             std::unique_lock<Core::Mutex> lockGuard(outboundQueueMutex);
00372             outboundQueue.emplace_front(std::move(outbound));
00373             return;
00374         }
00375     }

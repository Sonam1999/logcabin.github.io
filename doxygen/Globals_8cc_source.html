00067 void
00068 Globals::init(uint64_t serverId)
00069 {
00070     if (!raft) {
00071         raft.reset(new RaftConsensus(*this));
00072     }
00073 
00074     if (!raftService) {
00075         raftService.reset(new RaftService(*this));
00076     }
00077 
00078     if (!clientService) {
00079         clientService.reset(new ClientService(*this));
00080     }
00081 
00082     if (!rpcServer) {
00083         rpcServer.reset(new RPC::Server(eventLoop,
00084                                         Protocol::Common::MAX_MESSAGE_LENGTH));
00085 
00086         uint32_t maxThreads = config.read<uint16_t>("maxThreads", 16);
00087         rpcServer->registerService(Protocol::Common::ServiceId::RAFT_SERVICE,
00088                                    raftService,
00089                                    maxThreads);
00090         rpcServer->registerService(Protocol::Common::ServiceId::CLIENT_SERVICE,
00091                                    clientService,
00092                                    maxThreads);
00093 
00094 
00095         std::string configServers = config.read<std::string>("servers", "");
00096         std::vector<std::string> listenAddresses =
00097             Core::StringUtil::split(configServers, ';');
00098         if (listenAddresses.empty()) {
00099             PANIC("No server addresses specified to listen on. "
00100                   "You must set the 'servers' configuration option.");
00101         }
00102         std::string error = "Server ID has no matching address";
00103         for (uint32_t i = 0; i < listenAddresses.size(); ++i) {
00104             if (serverId != 0 && serverId != i + 1)
00105                 continue;
00106             RPC::Address address(listenAddresses[i],
00107                                  Protocol::Common::DEFAULT_PORT);
00108             error = rpcServer->bind(address);
00109             if (error.empty()) {
00110                 NOTICE("Serving on %s", address.toString().c_str());
00111                 raft->serverId = i + 1;
00112                 raft->serverAddress = listenAddresses[i];
00113                 Core::Debug::processName =
00114                     Core::StringUtil::format("%lu", raft->serverId);
00115                 raft->init();
00116                 break;
00117             }
00118         }
00119         if (!error.empty()) {
00120             PANIC("Could not bind to any server address in: %s. "
00121                   "Last error was: %s",
00122                   configServers.c_str(),
00123                   error.c_str());
00124         }
00125     }
00126 
00127     if (!stateMachine) {
00128         stateMachine.reset(new StateMachine(raft, config));
00129     }
00130 
00131 }

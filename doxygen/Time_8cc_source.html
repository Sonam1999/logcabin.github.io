00079 SystemClock::time_point
00080 SteadyTimeConverter::convert(SteadyClock::time_point when)
00081 {
00082     // Note that this relies on signed integer wrapping, so -fwrapv or
00083     // -fno-strict-overflow must be on for correctness under optimizing
00084     // comiplers. The unit tests are pretty good at catching when this isn't
00085     // the case.
00086     std::chrono::nanoseconds diff = when - steadyNow;
00087     SystemClock::time_point then = systemNow + diff;
00088     if (when > steadyNow && then < systemNow) // overflow
00089         return SystemClock::time_point::max();
00090     return then;
00091 }

00108 std::string
00109 padFraction(int64_t fraction, uint64_t digits)
00110 {
00111     if (fraction == 0)
00112         return "";
00113     if (fraction < 0)
00114         fraction *= -1;
00115     while (fraction % 1000 == 0 && digits >= 3) {
00116         digits -= 3;
00117         fraction /= 1000;
00118     }
00119     char ret[digits + 2];
00120     ret[0] = '.';
00121     ret[digits + 1] = '\0';
00122     for (uint64_t i = digits; i > 0; --i) {
00123         ret[i] = char('0' + (fraction % 10));
00124         fraction /= 10;
00125     }
00126     return ret;
00127 }

00134 {

00136     if (nanos / 1000000000L != 0) {

00144     } else if (nanos / 1000L != 0) {
00145         int64_t whole    = nanos / 1000L;
00146         int64_t fraction = nanos % 1000L;
00147         os << whole << padFraction(fraction, 3) << " us";
00148     } else {
00149         os << nanos << " ns";
00150     }

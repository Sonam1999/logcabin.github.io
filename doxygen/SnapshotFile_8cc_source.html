00082 std::string
00083 Reader::readMessage(google::protobuf::Message& message)
00084 {
00085     uint32_t length = 0;
00086     uint64_t r = readRaw(&length, sizeof(length));
00087     if (r < sizeof(length)) {
00088         return format("Could only read %lu bytes of %lu-byte length field in "
00089                       "file %s (at offset %lu of %lu-byte file)",
00090                       r,
00091                       sizeof(length),
00092                       file.path.c_str(),
00093                       bytesRead - r,
00094                       getSizeBytes());
00095     }
00096     length = be32toh(length);
00097     if (getSizeBytes() - bytesRead < length) {
00098         return format("ProtoBuf is %u bytes long but there are only %lu "
00099                       "bytes remaining in file %s (at offset %lu)",
00100                       length,
00101                       getSizeBytes() - bytesRead,
00102                       file.path.c_str(),
00103                       bytesRead);
00104     }
00105     const Core::Buffer buf(const_cast<void*>(contents->get(bytesRead, length)),
00106                            length,
00107                            NULL);
00108     std::string error;
00109     if (!Core::ProtoBuf::parse(buf, message)) {
00110         error = format("Could not parse ProtoBuf at bytes %lu-%lu (inclusive) "
00111                        "in file %s of length %lu",
00112                        bytesRead,
00113                        bytesRead + length -1,
00114                        file.path.c_str(),
00115                        getSizeBytes());
00116     }
00117     bytesRead += length;
00118     if (getSizeBytes() > 1024 && // minimum to keep quiet during unit tests
00119         10 * bytesRead / getSizeBytes() !=
00120         10 * (bytesRead - length) / getSizeBytes()) {
00121         NOTICE("Read %lu%% of snapshot",
00122                100 * bytesRead / getSizeBytes());
00123     }
00124     return error;
00125 }

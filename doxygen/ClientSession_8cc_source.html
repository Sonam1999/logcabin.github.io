00030 /**
00031  * The number of milliseconds to wait until the client gets suspicious about
00032  * the server not responding. After this amount of time elapses, the client
00033  * will send a ping to the server. If no response is received within another
00034  * TIMEOUT_MS milliseconds, the session is closed.
00035  *
00036  * TODO(ongaro): How should this value be chosen?
00037  * Ideally, you probably want this to be set to something like the 99-th
00038  * percentile of your RPC latency.
00039  *

00041  */

00094 void
00095 ClientSession::MessageSocketHandler::handleReceivedMessage(
00096         MessageId messageId,
00097         Buffer message)
00098 {
00099     std::unique_lock<std::mutex> mutexGuard(session.mutex);
00100 
00101     if (messageId == Protocol::Common::PING_MESSAGE_ID) {
00102         if (session.numActiveRPCs > 0 && session.activePing) {
00103             // The server has shown that it is alive for now.
00104             // Let's get suspicious again in another TIMEOUT_MS.
00105             session.activePing = false;
00106             session.timer.schedule(TIMEOUT_MS * 1000 * 1000);
00107         } else {
00108             VERBOSE("Received an unexpected ping response. This can happen "
00109                     "for a number of reasons and is no cause for alarm. For "
00110                     "example, this happens if a ping request was sent out, "
00111                     "then all RPCs completed before the ping response "
00112                     "arrived.");
00113         }
00114         return;
00115     }
00116 
00117     auto it = session.responses.find(messageId);
00118     if (it == session.responses.end()) {
00119         VERBOSE("Received an unexpected response with message ID %lu. "
00120                 "This can happen for a number of reasons and is no cause "
00121                 "for alarm. For example, this happens if the RPC was "
00122                 "cancelled before its response arrived.",
00123                 messageId);
00124         return;
00125     }
00126     Response& response = *it->second;
00127     if (response.status == Response::HAS_REPLY) {
00128         WARNING("Received a second response from the server for "
00129                 "message ID %lu. This indicates that either the client or "
00130                 "server is assigning message IDs incorrectly, or "
00131                 "the server is misbehaving. Dropped this response.",
00132                 messageId);
00133         return;
00134     }
00135 
00136     // Book-keeping for timeouts
00137     --session.numActiveRPCs;
00138     if (session.numActiveRPCs == 0)
00139         session.timer.deschedule();
00140     else
00141         session.timer.schedule(TIMEOUT_MS * 1000 * 1000);
00142 
00143     // Fill in the response
00144     response.status = Response::HAS_REPLY;
00145     response.reply = std::move(message);
00146     response.ready.notify_all();
00147 }

00251     // Be careful not to pass a sockaddr of length 0 to conect(). Although it
00252     // should return -1 EINVAL, on some systems (e.g., RHEL6) it instead
00253     // returns OK but leaves the socket unconnected! See
00254     // https://github.com/logcabin/logcabin/issues/66 for more details.
00255     if (!address.isValid()) {
00256         errorMessage = "Failed to resolve " + address.toString();
00257         return;

00260     // Some TCP connection timeouts appear to be ridiculously long in the wild.
00261     // Limit this to 10 seconds, after which you'd most likely want to retry.
00262     timeout = std::min(timeout, Clock::now() + std::chrono::seconds(10));
00263 
00264     // Setting NONBLOCK here makes connect return right away with EINPROGRESS.
00265     // Then we can monitor the fd until it's writable to know when it's done,
00266     // along with a timeout. See man page for connect under EINPROGRESS.
00267     int fd = socket(AF_INET, SOCK_STREAM|SOCK_NONBLOCK, 0);
00268     if (fd < 0) {
00269         errorMessage = "Failed to create socket";
00270         return;
00271     }
00272 
00273     // According to the spec, connect() could return OK done here, but in
00274     // practice it'll return EINPROGRESS.

00277                       address.getSockAddr(),
00278                       address.getSockAddrLen());

00280         switch (errno) {
00281             case EINPROGRESS:
00282                 waiting = true;
00283                 break;
00284             default:

00286                     "Failed to connect socket to %s: %s",
00287                     address.toString().c_str(),
00288                     strerror(errno));

00290                 return;
00291         }
00292     }
00293 
00294     if (waiting) {
00295         // This is a pretty heavy-weight method of watching a file descriptor
00296         // for a given period of time. On the other hand, it's only a few lines
00297         // of code with the LogCabin::Event classes, so it's easier for now.
00298         Event::Loop loop;
00299         FileNotifier fileNotifier(loop, fd, Event::File::CALLER_CLOSES_FD);
00300         TimerNotifier timerNotifier(loop);
00301         Event::File::Monitor fileMonitor(loop, fileNotifier, EPOLLOUT);
00302         Event::Timer::Monitor timerMonitor(loop, timerNotifier);
00303         timerNotifier.scheduleAbsolute(timeout);

00305             loop.runForever();

00307                 int error = 0;
00308                 socklen_t errorlen = sizeof(error);
00309                 r = getsockopt(fd, SOL_SOCKET, SO_ERROR, &error, &errorlen);
00310                 if (r != 0)

00312                 if (error != 0) {
00313                     errorMessage = Core::StringUtil::format(
00314                         "Failed to connect socket to %s: %s",
00315                         address.toString().c_str(),
00316                         strerror(error));
00317                 }
00318                 break;
00319             }
00320             if (Clock::now() > timeout) {
00321                 errorMessage = Core::StringUtil::format(
00322                     "Failed to connect socket to %s: timeout expired",
00323                     address.toString().c_str());

00330         close(fd);

00405 void
00406 ClientSession::cancel(OpaqueClientRPC& rpc)
00407 {
00408     // The RPC may be holding the last reference to this session. This
00409     // temporary reference makes sure this object isn't destroyed until after
00410     // we return from this method. It must be the first line in this method.
00411     std::shared_ptr<ClientSession> selfGuard(self.lock());
00412 
00413     // There are two ways to cancel an RPC:
00414     // 1. If there's some thread currently blocked in wait(), this method marks
00415     //    the Response's status as CANCELED, and wait() will delete it later.
00416     // 2. If there's no thread currently blocked in wait(), the Response is
00417     //    deleted entirely.
00418     std::unique_lock<std::mutex> mutexGuard(mutex);
00419     auto it = responses.find(rpc.responseToken);
00420     if (it == responses.end())
00421         return;
00422     Response* response = it->second;
00423     if (response->hasWaiter) {
00424         response->status = Response::CANCELED;
00425         response->ready.notify_all();
00426     } else {
00427         delete response;
00428         responses.erase(it);
00429     }
00430 
00431     --numActiveRPCs;
00432     // Even if numActiveRPCs == 0, it's simpler here to just let the timer wake
00433     // up an extra time and clean up. Otherwise, we'd need to grab an
00434     // Event::Loop::Lock prior to the mutex to call deschedule() without
00435     // inducing deadlock.

00472 void
00473 ClientSession::wait(const OpaqueClientRPC& rpc, TimePoint timeout)
00474 {
00475     // The RPC may be holding the last reference to this session. This
00476     // temporary reference makes sure this object isn't destroyed until after
00477     // we return from this method. It must be the first line in this method.
00478     std::shared_ptr<ClientSession> selfGuard(self.lock());
00479 
00480     std::unique_lock<std::mutex> mutexGuard(mutex);
00481     while (true) {
00482         auto it = responses.find(rpc.responseToken);
00483         if (it == responses.end())
00484             return; // RPC was cancelled or already updated
00485         Response* response = it->second;
00486         if (response->status == Response::HAS_REPLY) {
00487             return; // RPC has completed
00488         } else if (response->status == Response::CANCELED) {
00489             // RPC was cancelled, finish cleaning up
00490             delete response;
00491             responses.erase(it);
00492             return;
00493         } else if (!errorMessage.empty()) {
00494             return; // session has error
00495         } else if (timeout < Clock::now()) {
00496             return; // timeout
00497         }
00498         response->hasWaiter = true;
00499         response->ready.wait_until(mutexGuard, timeout);
00500         response->hasWaiter = false;
00501     }

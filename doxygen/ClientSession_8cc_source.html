00024 /**
00025  * The number of milliseconds to wait until the client gets suspicious about
00026  * the server not responding. After this amount of time elapses, the client
00027  * will send a ping to the server. If no response is received within another
00028  * TIMEOUT_MS milliseconds, the session is closed.
00029  *
00030  * TODO(ongaro): How should this value be chosen?
00031  * Ideally, you probably want this to be set to something like the 99-th
00032  * percentile of your RPC latency.
00033  *
00034  * TODO(ongaro): How does this interact with TCP?
00035  */
00036 enum { TIMEOUT_MS = 100 };

00058 void
00059 ClientSession::ClientMessageSocket::onReceiveMessage(MessageId messageId,
00060                                                      Buffer message)
00061 {
00062     std::unique_lock<std::mutex> mutexGuard(session.mutex);
00063 
00064     if (messageId == PING_MESSAGE_ID) {
00065         if (session.numActiveRPCs > 0 && session.activePing) {
00066             // The server has shown that it is alive for now.
00067             // Let's get suspicious again in another TIMEOUT_MS.
00068             session.activePing = false;
00069             session.timer.schedule(TIMEOUT_MS * 1000 * 1000);
00070         } else {
00071             VERBOSE("Received an unexpected ping response. This can happen "
00072                     "for a number of reasons and is no cause for alarm. For "
00073                     "example, this happens if a ping request was sent out, "
00074                     "then all RPCs completed before the ping response "
00075                     "arrived.");
00076         }
00077         return;
00078     }
00079 
00080     auto it = session.responses.find(messageId);
00081     if (it == session.responses.end()) {
00082         VERBOSE("Received an unexpected response with message ID %lu. "
00083                 "This can happen for a number of reasons and is no cause "
00084                 "for alarm. For example, this happens if the RPC was "
00085                 "cancelled before its response arrived.",
00086                 messageId);
00087         return;
00088     }
00089     Response& response = *it->second;
00090     if (response.status == Response::HAS_REPLY) {
00091         WARNING("Received a second response from the server for "
00092                 "message ID %lu. This indicates that either the client or "
00093                 "server is assigning message IDs incorrectly, or "
00094                 "the server is misbehaving. Dropped this response.",
00095                 messageId);
00096         return;
00097     }
00098 
00099     // Book-keeping for timeouts
00100     --session.numActiveRPCs;
00101     if (session.numActiveRPCs == 0)
00102         session.timer.deschedule();
00103     else
00104         session.timer.schedule(TIMEOUT_MS * 1000 * 1000);
00105 
00106     // Fill in the response
00107     response.status = Response::HAS_REPLY;
00108     response.reply = std::move(message);
00109     response.ready.notify_all();
00110 }

00284 ClientSession::cancel(OpaqueClientRPC& rpc)
00285 {
00286     // The RPC may be holding the last reference to this session. This
00287     // temporary reference makes sure this object isn't destroyed until after
00288     // we return from this method. It must be the first line in this method.
00289     std::shared_ptr<ClientSession> selfGuard(self.lock());
00290 
00291     // There are two ways to cancel an RPC:
00292     // 1. If there's some thread currently blocked in wait(), this method marks
00293     //    the Response's status as CANCELED, and wait() will delete it later.
00294     // 2. If there's no thread currently blocked in wait(), the Response is
00295     //    deleted entirely.
00296     std::unique_lock<std::mutex> mutexGuard(mutex);
00297     auto it = responses.find(rpc.responseToken);
00298     if (it == responses.end())
00299         return;
00300     Response* response = it->second;
00301     if (response->hasWaiter) {
00302         response->status = Response::CANCELED;
00303         response->ready.notify_all();
00304     } else {
00305         delete response;
00306         responses.erase(it);
00307     }
00308 
00309     --numActiveRPCs;
00310     // Even if numActiveRPCs == 0, it's simpler here to just let the timer wake
00311     // up an extra time and clean up. Otherwise, we'd need to grab an
00312     // Event::Loop::Lock prior to the mutex to call deschedule() without
00313     // inducing deadlock.

00320     // temporary reference makes sure this object isn't destroyed until after
00321     // we return from this method. It must be the first line in this method.

00323

00326     if (it == responses.end()) {
00327         // RPC was cancelled, fields set already

00329         return;
00330     }
00331     Response* response = it->second;
00332     if (response->status == Response::HAS_REPLY) {
00333         rpc.reply = std::move(response->reply);
00334         rpc.status = OpaqueClientRPC::Status::OK;
00335     } else if (!errorMessage.empty()) {
00336         rpc.errorMessage = errorMessage;
00337         rpc.status = OpaqueClientRPC::Status::ERROR;
00338     } else {
00339         // If the RPC was canceled, then it'd be marked ready and update()
00340         // wouldn't be called again.
00341         assert(response->status != Response::CANCELED);
00342         return; // not ready
00343     }
00344     rpc.session.reset();
00345 
00346     delete response;

00350 void
00351 ClientSession::wait(const OpaqueClientRPC& rpc)
00352 {
00353     // The RPC may be holding the last reference to this session. This
00354     // temporary reference makes sure this object isn't destroyed until after
00355     // we return from this method. It must be the first line in this method.
00356     std::shared_ptr<ClientSession> selfGuard(self.lock());
00357 
00358     std::unique_lock<std::mutex> mutexGuard(mutex);
00359     while (true) {
00360         auto it = responses.find(rpc.responseToken);
00361         if (it == responses.end())
00362             return; // RPC was cancelled or already updated
00363         Response* response = it->second;
00364         if (response->status == Response::HAS_REPLY) {
00365             return; // RPC has completed
00366         } else if (response->status == Response::CANCELED) {
00367             // RPC was cancelled, finish cleaning up
00368             delete response;
00369             responses.erase(it);
00370             return;
00371         } else if (!errorMessage.empty()) {
00372             return; // session has error
00373         }
00374         response->hasWaiter = true;
00375         response->ready.wait(mutexGuard);
00376         response->hasWaiter = false;
00377     }

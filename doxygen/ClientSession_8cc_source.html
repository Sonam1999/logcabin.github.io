00080 void
00081 ClientSession::MessageSocketHandler::handleReceivedMessage(
00082         MessageId messageId,
00083         Core::Buffer message)
00084 {
00085     std::unique_lock<std::mutex> mutexGuard(session.mutex);
00086 
00087     if (messageId == Protocol::Common::PING_MESSAGE_ID) {
00088         if (session.numActiveRPCs > 0 && session.activePing) {
00089             // The server has shown that it is alive for now.
00090             // Let's get suspicious again in another PING_TIMEOUT_MS.
00091             session.activePing = false;
00092             session.timer.schedule(session.PING_TIMEOUT_MS * 1000 * 1000);
00093         } else {
00094             VERBOSE("Received an unexpected ping response. This can happen "
00095                     "for a number of reasons and is no cause for alarm. For "
00096                     "example, this happens if a ping request was sent out, "
00097                     "then all RPCs completed before the ping response "
00098                     "arrived.");
00099         }
00100         return;
00101     }
00102 
00103     auto it = session.responses.find(messageId);
00104     if (it == session.responses.end()) {
00105         VERBOSE("Received an unexpected response with message ID %lu. "
00106                 "This can happen for a number of reasons and is no cause "
00107                 "for alarm. For example, this happens if the RPC was "
00108                 "cancelled before its response arrived.",
00109                 messageId);
00110         return;
00111     }
00112     Response& response = *it->second;
00113     if (response.status == Response::HAS_REPLY) {
00114         WARNING("Received a second response from the server for "
00115                 "message ID %lu. This indicates that either the client or "
00116                 "server is assigning message IDs incorrectly, or "
00117                 "the server is misbehaving. Dropped this response.",
00118                 messageId);
00119         return;
00120     }
00121 
00122     // Book-keeping for timeouts
00123     --session.numActiveRPCs;
00124     if (session.numActiveRPCs == 0)
00125         session.timer.deschedule();
00126     else
00127         session.timer.schedule(session.PING_TIMEOUT_MS * 1000 * 1000);
00128 
00129     // Fill in the response
00130     response.status = Response::HAS_REPLY;
00131     response.reply = std::move(message);
00132     response.ready.notify_all();
00133 }

00238                   "PING_MESSAGE_ID changed?");

00240     // Be careful not to pass a sockaddr of length 0 to conect(). Although it
00241     // should return -1 EINVAL, on some systems (e.g., RHEL6) it instead
00242     // returns OK but leaves the socket unconnected! See
00243     // https://github.com/logcabin/logcabin/issues/66 for more details.
00244     if (!address.isValid()) {
00245         errorMessage = "Failed to resolve " + address.toString();
00246         return;
00247     }
00248 
00249     // Some TCP connection timeouts appear to be ridiculously long in the wild.
00250     // Limit this to 10 seconds, after which you'd most likely want to retry.
00251     timeout = std::min(timeout, Clock::now() + std::chrono::seconds(10));
00252 
00253     // Setting NONBLOCK here makes connect return right away with EINPROGRESS.
00254     // Then we can monitor the fd until it's writable to know when it's done,
00255     // along with a timeout. See man page for connect under EINPROGRESS.
00256     int fd = socket(AF_INET, SOCK_STREAM|SOCK_NONBLOCK, 0);
00257     if (fd < 0) {
00258         errorMessage = "Failed to create socket";
00259         return;
00260     }
00261 
00262     // According to the spec, connect() could return OK done here, but in
00263     // practice it'll return EINPROGRESS.
00264     bool waiting = false;
00265     int r = connectFn(fd,
00266                       address.getSockAddr(),
00267                       address.getSockAddrLen());
00268     if (r != 0) {
00269         switch (errno) {
00270             case EINPROGRESS:
00271                 waiting = true;
00272                 break;
00273             default:
00274                 errorMessage = Core::StringUtil::format(
00275                     "Failed to connect socket to %s: %s",
00276                     address.toString().c_str(),
00277                     strerror(errno));
00278                 close(fd);
00279                 return;
00280         }
00281     }
00282 
00283     if (waiting) {
00284         // This is a pretty heavy-weight method of watching a file descriptor
00285         // for a given period of time. On the other hand, it's only a few lines
00286         // of code with the LogCabin::Event classes, so it's easier for now.
00287         Event::Loop loop;
00288         FileNotifier fileNotifier(loop, fd, Event::File::CALLER_CLOSES_FD);
00289         TimerNotifier timerNotifier(loop);
00290         Event::File::Monitor fileMonitor(loop, fileNotifier, EPOLLOUT);
00291         Event::Timer::Monitor timerMonitor(loop, timerNotifier);
00292         timerNotifier.scheduleAbsolute(timeout);
00293         while (true) {
00294             loop.runForever();
00295             if (fileNotifier.count > 0) {
00296                 int error = 0;
00297                 socklen_t errorlen = sizeof(error);
00298                 r = getsockopt(fd, SOL_SOCKET, SO_ERROR, &error, &errorlen);
00299                 if (r != 0)
00300                     PANIC("getsockopt failed: %s", strerror(errno));
00301                 if (error != 0) {
00302                     errorMessage = Core::StringUtil::format(
00303                         "Failed to connect socket to %s: %s",
00304                         address.toString().c_str(),
00305                         strerror(error));
00306                 }
00307                 break;
00308             }
00309             if (Clock::now() > timeout) {
00310                 errorMessage = Core::StringUtil::format(
00311                     "Failed to connect socket to %s: timeout expired",
00312                     address.toString().c_str());
00313                 break;
00314             }
00315             WARNING("spurious exit from event loop?");
00316         }
00317     }
00318     if (!errorMessage.empty()) {
00319         close(fd);
00320         return;
00321     }

00324         messageSocketHandler, eventLoop, fd, maxMessageLength));

00399 void
00400 ClientSession::cancel(OpaqueClientRPC& rpc)
00401 {
00402     // The RPC may be holding the last reference to this session. This
00403     // temporary reference makes sure this object isn't destroyed until after
00404     // we return from this method. It must be the first line in this method.
00405     std::shared_ptr<ClientSession> selfGuard(self.lock());
00406 
00407     // There are two ways to cancel an RPC:
00408     // 1. If there's some thread currently blocked in wait(), this method marks
00409     //    the Response's status as CANCELED, and wait() will delete it later.
00410     // 2. If there's no thread currently blocked in wait(), the Response is
00411     //    deleted entirely.
00412     std::unique_lock<std::mutex> mutexGuard(mutex);
00413     auto it = responses.find(rpc.responseToken);
00414     if (it == responses.end())
00415         return;
00416     Response* response = it->second;
00417     if (response->hasWaiter) {
00418         response->status = Response::CANCELED;
00419         response->ready.notify_all();
00420     } else {
00421         delete response;
00422         responses.erase(it);
00423     }
00424 
00425     --numActiveRPCs;
00426     // Even if numActiveRPCs == 0, it's simpler here to just let the timer wake
00427     // up an extra time and clean up. Otherwise, we'd need to grab an
00428     // Event::Loop::Lock prior to the mutex to call deschedule() without
00429     // inducing deadlock.
00430 }

00452         rpc.errorMessage = errorMessage;

00456         // wouldn't be called again.

00466 void
00467 ClientSession::wait(const OpaqueClientRPC& rpc, TimePoint timeout)
00468 {
00469     // The RPC may be holding the last reference to this session. This
00470     // temporary reference makes sure this object isn't destroyed until after
00471     // we return from this method. It must be the first line in this method.
00472     std::shared_ptr<ClientSession> selfGuard(self.lock());
00473 
00474     std::unique_lock<std::mutex> mutexGuard(mutex);
00475     while (true) {
00476         auto it = responses.find(rpc.responseToken);
00477         if (it == responses.end())
00478             return; // RPC was cancelled or already updated
00479         Response* response = it->second;
00480         if (response->status == Response::HAS_REPLY) {
00481             return; // RPC has completed
00482         } else if (response->status == Response::CANCELED) {
00483             // RPC was cancelled, finish cleaning up
00484             delete response;
00485             responses.erase(it);
00486             return;
00487         } else if (!errorMessage.empty()) {
00488             return; // session has error
00489         } else if (timeout < Clock::now()) {
00490             return; // timeout
00491         }
00492         response->hasWaiter = true;
00493         response->ready.wait_until(mutexGuard, timeout);
00494         response->hasWaiter = false;
00495     }
00496 }

00239     parents.push_back("root");
00240 
00241     // Split the path into a list of parent components and a target.
00242     std::string word;
00243     for (auto it = symbolic.begin(); it != symbolic.end(); ++it) {
00244         if (*it == '/') {
00245             if (!word.empty()) {
00246                 parents.push_back(word);
00247                 word.clear();
00248             }
00249         } else {
00250             word += *it;
00251         }
00252     }
00253     if (!word.empty())
00254         parents.push_back(word);
00255     target = parents.back();
00256     parents.pop_back();

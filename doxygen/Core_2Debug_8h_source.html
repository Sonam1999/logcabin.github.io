00031 /**
00032  * The levels of verbosity for log messages. Higher values are noisier.
00033  */
00034 enum class LogLevel {
00035     // Keep this in sync with logLevelToString in Core/Debug.cc.
00036     /**
00037      * This log level is just used for disabling all log messages, which is
00038      * really only useful in unit tests.
00039      */
00040     SILENT = 0,
00041     /**
00042      * Bad stuff that shouldn't happen. The system broke its contract to users
00043      * in some way or some major assumption was violated.
00044      */
00045     ERROR = 1,
00046     /**
00047      * Messages at the WARNING level indicate that, although something went
00048      * wrong or something unexpected happened, it was transient and
00049      * recoverable.
00050      */
00051     WARNING = 2,
00052     /**
00053      * A system message that might be useful for administrators and developers.
00054      */
00055     NOTICE = 3,
00056     /**
00057      * Messages at the VERBOSE level don't necessarily indicate that anything
00058      * went wrong, but they could be useful in diagnosing problems.
00059      */
00060     VERBOSE = 4,
00061 };

00117 /**
00118  * Unconditionally log the given message to stderr.
00119  * This is normally called by ERROR(), WARNING(), NOTICE(), or VERBOSE().
00120  * \param level
00121  *      The level of importance of the message.
00122  * \param format
00123  *      A printf-style format string for the message. It should not include a
00124  *      line break at the end, as LOG will add one.
00125  * \param ...
00126  *      The arguments to the format string, as in printf.
00127  */
00128 #define LOG(level, format, ...) do { \
00129     if (::LogCabin::Core::Debug::isLogging(level, __FILE__)) { \
00130         ::LogCabin::Core::Debug::log(level, \
00131             __FILE__, __LINE__, __FUNCTION__, \
00132             format "\n", ##__VA_ARGS__); \
00133     } \
00134 } while (0)

00065             result.status = Status::CONDITION_NOT_MET;

00322 void
00323 ClientImpl::ExactlyOnceRPCHelper::keepAliveThreadMain()
00324 {
00325     std::unique_lock<Core::Mutex> lockGuard(mutex);
00326     while (!exiting) {
00327         TimePoint nextKeepAlive;
00328         if (keepAliveIntervalMs > 0) {
00329             nextKeepAlive = (lastKeepAliveStart +
00330                              std::chrono::milliseconds(keepAliveIntervalMs));
00331         } else {
00332             nextKeepAlive = TimePoint::max();
00333         }
00334         if (Clock::now() > nextKeepAlive) {
00335             Protocol::Client::ReadWriteTree::Request request;
00336             *request.mutable_exactly_once() = getRPCInfo(
00337                 Core::HoldingMutex(lockGuard),
00338                 TimePoint::max());
00339             setCondition(request,
00340                  {"keepalive",
00341                  "this is just a no-op to keep the client's session active; "
00342                  "the condition is expected to fail"});
00343             request.mutable_write()->set_path("keepalive");
00344             request.mutable_write()->set_contents("you shouldn't see this!");
00345             Protocol::Client::ReadWriteTree::Response response;
00346             keepAliveCall = client->leaderRPC->makeCall();
00347             keepAliveCall->start(OpCode::READ_WRITE_TREE, request,
00348                                  TimePoint::max());
00349             LeaderRPCBase::Call::Status callStatus;
00350             {
00351                 // release lock to allow concurrent cancellation
00352                 Core::MutexUnlock<Core::Mutex> unlockGuard(lockGuard);
00353                 callStatus = keepAliveCall->wait(response, TimePoint::max());
00354             }
00355             keepAliveCall.reset();
00356             switch (callStatus) {
00357                 case LeaderRPCBase::Call::Status::OK:
00358                     break;
00359                 case LeaderRPCBase::Call::Status::RETRY:
00360                     continue; // retry outer loop
00361                 case LeaderRPCBase::Call::Status::TIMEOUT:
00362                     PANIC("Unexpected timeout for keep-alive");
00363                 case LeaderRPCBase::Call::Status::INVALID_REQUEST:
00364                     PANIC("The server rejected our keep-alive request (Tree "
00365                           "write with unmet condition) as invalid");
00366             }
00367             doneWithRPC(request.exactly_once(),
00368                         Core::HoldingMutex(lockGuard));
00369             if (response.status() !=
00370                 Protocol::Client::Status::CONDITION_NOT_MET) {
00371                 WARNING("Keep-alive write should have failed its condition. "
00372                         "Unexpected status was %d: %s",
00373                         response.status(),
00374                         response.error().c_str());
00375             }
00376             continue;
00377         }
00378         keepAliveCV.wait_until(lockGuard, nextKeepAlive);
00379     }
00380 }

00621                          const std::string& workingDirectory,
00622                          std::string& canonical)
00623 {
00624     canonical = "";

00626     if (!path.empty() && *path.begin() != '/') {
00627         if (workingDirectory.empty() || *workingDirectory.begin() != '/') {
00628             Result result;
00629             result.status = Status::INVALID_ARGUMENT;
00630             result.error = Core::StringUtil::format(
00631                         "Can't use relative path '%s' from working directory "
00632                         "'%s' (working directory should be an absolute path)",
00633                         path.c_str(),
00634                         workingDirectory.c_str());
00635             return result;
00636 
00637         }
00638         split(workingDirectory, components);
00639     }
00640     split(path, components);
00641     // Iron out any ".", ".."
00642     size_t i = 0;
00643     while (i < components.size()) {
00644         if (components.at(i) == "..") {
00645             if (i > 0) {
00646                 // erase previous and ".." components
00647                 components.erase(components.begin() + i - 1,
00648                                  components.begin() + i + 1);
00649                 --i;
00650             } else {
00651                 Result result;
00652                 result.status = Status::INVALID_ARGUMENT;

00046 Result
00047 treeError(const Message& response)
00048 {
00049     Result result;
00050     result.error = response.error();
00051     switch (response.status()) {
00052         case Protocol::Client::Status::OK:
00053             result.status = Status::OK;
00054             break;
00055         case Protocol::Client::Status::INVALID_ARGUMENT:
00056             result.status = Status::INVALID_ARGUMENT;
00057             break;
00058         case Protocol::Client::Status::LOOKUP_ERROR:
00059             result.status = Status::LOOKUP_ERROR;
00060             break;
00061         case Protocol::Client::Status::TYPE_ERROR:
00062             result.status = Status::TYPE_ERROR;
00063             break;
00064         case Protocol::Client::Status::CONDITION_NOT_MET:
00065             result.status = Status::CONDITION_NOT_MET;
00066             break;
00067         case Protocol::Client::Status::TIMEOUT:
00068             result.status = Status::TIMEOUT;
00069             break;
00070         case Protocol::Client::Status::SESSION_EXPIRED:
00071             PANIC("The client's session to the cluster expired. This is a "
00072                   "fatal error, since without a session the servers can't "
00073                   "tell if retried requests were already applied or not.");
00074             break;
00075         default:
00076             result.status = Status::INVALID_ARGUMENT;
00077             result.error = Core::StringUtil::format(
00078                 "Did not understand status code in response (%u). "
00079                 "Original error was: %s",
00080                 response.status(),
00081                 response.error().c_str());
00082             break;
00083     }
00084     return result;
00085 }

00110 {
00111     std::string word;
00112     for (auto it = path.begin(); it != path.end(); ++it) {
00113         if (*it == '/') {
00114             if (!word.empty()) {
00115                 components.push_back(word);

00331 void
00332 ClientImpl::ExactlyOnceRPCHelper::keepAliveThreadMain()
00333 {
00334     std::unique_lock<Core::Mutex> lockGuard(mutex);
00335     while (!exiting) {
00336         TimePoint nextKeepAlive;
00337         if (keepAliveIntervalMs > 0) {
00338             nextKeepAlive = (lastKeepAliveStart +
00339                              std::chrono::milliseconds(keepAliveIntervalMs));
00340         } else {
00341             nextKeepAlive = TimePoint::max();
00342         }
00343         if (Clock::now() > nextKeepAlive) {
00344             Protocol::Client::StateMachineCommand::Request request;
00345             Protocol::Client::ReadWriteTree::Request& trequest =
00346                 *request.mutable_tree();
00347             *trequest.mutable_exactly_once() = getRPCInfo(
00348                 Core::HoldingMutex(lockGuard),
00349                 TimePoint::max());
00350             setCondition(trequest,
00351                  {"keepalive",
00352                  "this is just a no-op to keep the client's session active; "
00353                  "the condition is expected to fail"});
00354             trequest.mutable_write()->set_path("keepalive");
00355             trequest.mutable_write()->set_contents("you shouldn't see this!");
00356             Protocol::Client::StateMachineCommand::Response response;
00357             keepAliveCall = client->leaderRPC->makeCall();
00358             keepAliveCall->start(OpCode::STATE_MACHINE_COMMAND, request,
00359                                  TimePoint::max());
00360             LeaderRPCBase::Call::Status callStatus;
00361             {
00362                 // release lock to allow concurrent cancellation
00363                 Core::MutexUnlock<Core::Mutex> unlockGuard(lockGuard);
00364                 callStatus = keepAliveCall->wait(response, TimePoint::max());
00365             }
00366             keepAliveCall.reset();
00367             switch (callStatus) {
00368                 case LeaderRPCBase::Call::Status::OK:
00369                     break;
00370                 case LeaderRPCBase::Call::Status::RETRY:
00371                     continue; // retry outer loop
00372                 case LeaderRPCBase::Call::Status::TIMEOUT:
00373                     PANIC("Unexpected timeout for keep-alive");
00374                 case LeaderRPCBase::Call::Status::INVALID_REQUEST:
00375                     PANIC("The server rejected our keep-alive request (Tree "
00376                           "write with unmet condition) as invalid");
00377             }
00378             doneWithRPC(trequest.exactly_once(),
00379                         Core::HoldingMutex(lockGuard));
00380             const Protocol::Client::ReadWriteTree::Response& tresponse =
00381                 response.tree();
00382             if (tresponse.status() !=
00383                 Protocol::Client::Status::CONDITION_NOT_MET) {
00384                 WARNING("Keep-alive write should have failed its condition. "
00385                         "Unexpected status was %d: %s",
00386                         tresponse.status(),
00387                         tresponse.error().c_str());
00388             }
00389             continue;
00390         }
00391         keepAliveCV.wait_until(lockGuard, nextKeepAlive);
00392     }
00393 }

00067 /**
00068  * Split a path into its components. Helper for ClientImpl::canonicalize.
00069  * \param[in] path
00070  *      Forward slash-delimited path (relative or absolute).
00071  * \param[out] components
00072  *      The components of path are appended to this vector.
00073  */
00074 void
00075 split(const std::string& path, std::vector<std::string>& components)
00076 {
00077     std::string word;
00078     for (auto it = path.begin(); it != path.end(); ++it) {
00079         if (*it == '/') {
00080             if (!word.empty()) {
00081                 components.push_back(word);
00082                 word.clear();
00083             }
00084         } else {
00085             word += *it;
00086         }
00087     }
00088     if (!word.empty())
00089         components.push_back(word);
00090 }

00189 void
00190 ClientImpl::ExactlyOnceRPCHelper::keepAliveThreadMain()
00191 {
00192     std::unique_lock<Core::Mutex> lockGuard(mutex);
00193     while (!exiting) {
00194         TimePoint nextKeepAlive;
00195         if (keepAliveIntervalMs > 0) {
00196             nextKeepAlive = (lastKeepAliveStart +
00197                              std::chrono::milliseconds(keepAliveIntervalMs));
00198         } else {
00199             nextKeepAlive = TimePoint::max();
00200         }
00201         if (Clock::now() > nextKeepAlive) {
00202             Protocol::Client::ReadWriteTree::Request request;
00203             *request.mutable_exactly_once() = getRPCInfo(lockGuard);
00204             setCondition(request,
00205                  {"keepalive",
00206                  "this is just a no-op to keep the client's session active; "
00207                  "the condition is expected to fail"});
00208             request.mutable_write()->set_path("keepalive");
00209             request.mutable_write()->set_contents("you shouldn't see this!");
00210             Protocol::Client::ReadWriteTree::Response response;
00211             keepAliveCall = client->leaderRPC->makeCall();
00212             keepAliveCall->start(OpCode::READ_WRITE_TREE, request);
00213             bool ok;
00214             {
00215                 // release lock to allow concurrent cancellation
00216                 Core::MutexUnlock<Core::Mutex> unlockGuard(lockGuard);
00217                 ok = keepAliveCall->wait(response);
00218             }
00219             keepAliveCall.reset();
00220             if (!ok)
00221                 continue;
00222             doneWithRPC(request.exactly_once(), lockGuard);
00223             if (response.status() !=
00224                 Protocol::Client::Status::CONDITION_NOT_MET) {
00225                 WARNING("Keep-alive write should have failed its condition. "
00226                         "Unexpected status was %d: %s",
00227                         response.status(),
00228                         response.error().c_str());
00229             }
00230             continue;
00231         }
00232         keepAliveCV.wait_until(lockGuard, nextKeepAlive);
00233     }
00234 }

00319 Result
00320 ClientImpl::canonicalize(const std::string& path,
00321                          const std::string& workingDirectory,
00322                          std::string& canonical)
00323 {
00324     canonical = "";
00325     std::vector<std::string> components;
00326     if (!path.empty() && *path.begin() != '/') {
00327         if (workingDirectory.empty() || *workingDirectory.begin() != '/') {
00328             Result result;
00329             result.status = Status::INVALID_ARGUMENT;
00330             result.error = Core::StringUtil::format(
00331                         "Can't use relative path '%s' from working directory "
00332                         "'%s' (working directory should be an absolute path)",
00333                         path.c_str(),
00334                         workingDirectory.c_str());
00335             return result;
00336 
00337         }
00338         split(workingDirectory, components);
00339     }
00340     split(path, components);
00341     // Iron out any ".", ".."
00342     size_t i = 0;
00343     while (i < components.size()) {
00344         if (components.at(i) == "..") {
00345             if (i > 0) {
00346                 // erase previous and ".." components
00347                 components.erase(components.begin() + i - 1,
00348                                  components.begin() + i + 1);
00349                 --i;
00350             } else {
00351                 Result result;
00352                 result.status = Status::INVALID_ARGUMENT;
00353                 result.error = Core::StringUtil::format(
00354                             "Path '%s' from working directory '%s' attempts "
00355                             "to look up directory above root ('/')",
00356                             path.c_str(),
00357                             workingDirectory.c_str());
00358                 return result;
00359             }
00360         } else if (components.at(i) == ".") {
00361             components.erase(components.begin() + i);
00362         } else {
00363             ++i;
00364         }
00365     }
00366     if (components.empty()) {
00367         canonical = "/";
00368     } else {
00369         for (auto it = components.begin(); it != components.end(); ++it)
00370             canonical += "/" + *it;
00371     }
00372     return Result();
00373 }

00507 uint32_t
00508 ClientImpl::negotiateRPCVersion()
00509 {
00510     Protocol::Client::GetSupportedRPCVersions::Request request;
00511     Protocol::Client::GetSupportedRPCVersions::Response response;
00512     leaderRPC->call(OpCode::GET_SUPPORTED_RPC_VERSIONS,
00513                     request, response);
00514     uint32_t serverMin = response.min_version();
00515     uint32_t serverMax = response.max_version();
00516     if (MAX_RPC_PROTOCOL_VERSION < serverMin) {
00517         PANIC("This client is too old to talk to your LogCabin cluster. "
00518               "You'll need to update your LogCabin client library. The "
00519               "server supports down to version %u, but this library only "
00520               "supports up to version %u.",
00521               serverMin, MAX_RPC_PROTOCOL_VERSION);
00522 
00523     } else if (MIN_RPC_PROTOCOL_VERSION > serverMax) {
00524         PANIC("This client is too new to talk to your LogCabin cluster. "
00525               "You'll need to upgrade your LogCabin cluster or downgrade "
00526               "your LogCabin client library. The server supports up to "
00527               "version %u, but this library only supports down to version %u.",
00528               serverMax, MIN_RPC_PROTOCOL_VERSION);
00529     } else {
00530         // There exists a protocol version both the client and server speak.
00531         // The preferred one is the maximum one they both support.
00532         return std::min(MAX_RPC_PROTOCOL_VERSION, serverMax);
00533     }

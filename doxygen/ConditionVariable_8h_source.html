00032 /**
00033  * A wrapper around std::condition_variable that is useful for testing
00034  * purposes and works around a bug in libstdc++.
00035  *
00036  * You can set a callback to be called when the condition variable is
00037  * waited on; instead of waiting, this callback will be called. This callback
00038  * can, for example, change some shared state so that the calling thread's
00039  * condition is satisfied. It also counts how many times the condition variable
00040  * has been notified.
00041  *
00042  * The interface to this class is the same as std::condition_variable.
00043  * This class also goes through some trouble so that it can be used with
00044  * std::unique_lock<Mutex>, since normal condition variables may only be used
00045  * with std::unique_lock<std::mutex>.
00046  */

00081     wait(std::unique_lock<Core::Mutex>& lockGuard) {
00082         Core::Mutex& mutex(*lockGuard.mutex());
00083         if (mutex.callback)
00084             mutex.callback();
00085         assert(lockGuard);
00086         std::unique_lock<std::mutex> stdLockGuard(mutex.m,
00087                                                   std::adopt_lock_t());
00088         lockGuard.release();
00089         wait(stdLockGuard);
00090         assert(stdLockGuard);
00091         lockGuard = std::unique_lock<Core::Mutex>(mutex, std::adopt_lock_t());
00092         stdLockGuard.release();
00093         if (mutex.callback)
00094             mutex.callback();
00095     }

00101     // Returns void instead of cv_status since it's almost always clearer to
00102     // check whether timeout has elapsed explicitly. Also, gcc 4.4 doesn't have
00103     // cv_status and uses bool instead.
00104     template<typename Clock, typename Duration>
00105     void
00106     wait_until(std::unique_lock<std::mutex>& lockGuard,
00107                const std::chrono::time_point<Clock, Duration>& abs_time) {
00108         lastWaitUntilTimeSinceEpoch =
00109             std::chrono::duration_cast<std::chrono::milliseconds>(
00110                                             abs_time.time_since_epoch());
00111         if (callback) {
00112             lockGuard.unlock();
00113             callback();
00114             lockGuard.lock();
00115         } else {
00116             // Work-around for bug in libstdc++: wait for no more than an hour
00117             // to avoid overflow. See
00118             // http://gcc.gnu.org/bugzilla/show_bug.cgi?id=58931
00119             std::chrono::time_point<Clock, Duration> now = Clock::now();
00120             if (abs_time < now + std::chrono::hours(1))
00121                 cv.wait_until(lockGuard, abs_time);
00122             else
00123                 cv.wait_until(lockGuard, now + std::chrono::hours(1));
00124         }
00125     }

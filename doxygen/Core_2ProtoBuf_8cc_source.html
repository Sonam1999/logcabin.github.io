00086 std::string
00087 dumpString(const google::protobuf::Message& protoBuf,
00088            bool forCopyingIntoTest)
00089 {
00090     std::string output;
00091     google::protobuf::TextFormat::Printer printer;
00092     if (forCopyingIntoTest) {
00093         // Most lines that use these strings will look like this:
00094         // ^    EXPECT_EQ(...,$
00095         // ^              "..."
00096         // ^              "...");
00097         //  12345678901234
00098         // Therefore, we want 14 leading spaces. Tell gtest we want 16, though,
00099         // so that when we add in the surrounding quotes later, lines won't
00100         // wrap.
00101         printer.SetInitialIndentLevel(8);
00102     }
00103     printer.SetUseShortRepeatedPrimitives(true);
00104     printer.PrintToString(protoBuf, &output);
00105     if (forCopyingIntoTest) {
00106         // TextFormat::Printer escapes ' already.
00107         replaceAll(output, "\"", "'");
00108         replaceAll(output, "                ", "              \"");
00109         replaceAll(output, "\n", "\"\n");
00110     }
00111     if (!protoBuf.IsInitialized()) {
00112         std::vector<std::string> errors;
00113         protoBuf.FindInitializationErrors(&errors);
00114         std::ostringstream outputBuf;
00115         outputBuf << output;
00116         for (auto it = errors.begin(); it != errors.end(); ++it) {
00117             if (forCopyingIntoTest) {
00118                  outputBuf << "              \"" << *it << ": UNDEFINED\"\n";
00119             } else {
00120                  outputBuf << *it << ": UNDEFINED\n";
00121             }
00122         }
00123         return output + outputBuf.str();
00124     }
00125     return output;
00126 }

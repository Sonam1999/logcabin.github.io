00050 class Algorithm {
00051   public:
00052     explicit Algorithm(std::unique_ptr<CryptoPP::HashTransformation> hashFn)
00053         : name(hashFn->AlgorithmName())
00054         , digestSize(hashFn->DigestSize())
00055         , outputSize(downCast<uint32_t>(
00056             name.length() + 1 +
00057             digestSize * 2 + 1))
00058         , hashFnMutex()
00059         , hashFn(std::move(hashFn))
00060     {
00061         assert(outputSize <= MAX_LENGTH);

00064     /**
00065      * Calculate the checksum for some data.
00066      * \param data
00067      *      An list of (pointer, length) pairs describing what to checksum.
00068      * \param[out] result
00069      *      The result of the hash function will be placed here.
00070      *      This will be a null-terminated, printable C-string.
00071      * \return
00072      *      The number of valid characters in 'output', including the null
00073      *      terminator. This is guaranteed to be greater than 1.
00074      */
00075     uint32_t
00076     writeChecksum(std::initializer_list<std::pair<const void*, uint64_t>> data,
00077                   char result[MAX_LENGTH]) {
00078         // copy name and : to result
00079         memcpy(result, name.c_str(), name.length());
00080         result = result + name.length();
00081         *result = ':';
00082         ++result;
00083 
00084         uint8_t binary[digestSize];
00085         { // calculate binary digest
00086             std::unique_lock<std::mutex> lockGuard(hashFnMutex);
00087             for (auto it = data.begin(); it != data.end(); ++it) {
00088                 hashFn->Update(static_cast<const uint8_t*>(it->first),
00089                                it->second);
00090             }
00091             hashFn->Final(binary);
00092         }
00093 
00094         // add hex digest to result
00095         const char* hexArray = "0123456789abcdef";
00096         for (uint32_t i = 0; i < digestSize; ++i) {
00097             *result = hexArray[binary[i] >> 4];
00098             ++result;
00099             *result = hexArray[binary[i] & 15];
00100             ++result;
00101         }
00102 
00103         // add null terminator to result and return total length
00104         *result = '\0';
00105         return outputSize;
00106     }
00107 
00108     /// Name of hash function.
00109     const std::string name;
00110     /// Size in bytes of binary hash function output.
00111     const uint32_t digestSize;
00112     /// Size in bytes of name:hexdigest string.
00113     const uint32_t outputSize;
00114   private:
00115     /// Protects hashFn (which is stateful) from concurrent modification.
00116     std::mutex hashFnMutex;
00117     /// Stateful hash function.
00118     std::unique_ptr<CryptoPP::HashTransformation> hashFn;
00119 };

00241 std::string
00242 verify(const char* checksum,
00243        std::initializer_list<std::pair<const void*, uint64_t>> data)
00244 {
00245     if (!Core::StringUtil::isPrintable(checksum))
00246         return "The given checksum value is corrupt and not printable.";
00247 
00248     Algorithm* algo;
00249     { // find algo
00250         char algorithmName[MAX_LENGTH];
00251         char* colon = strchr(const_cast<char*>(checksum), ':');
00252         if (colon == NULL)
00253             return format("Missing colon in checksum: %s", checksum);
00254 
00255         // nameLength does not include the null terminator
00256         uint32_t nameLength = downCast<uint32_t>(colon - checksum);
00257         memcpy(algorithmName, checksum, nameLength);
00258         algorithmName[nameLength] = '\0';
00259 
00260          algo = algorithms.find(algorithmName);
00261          if (algo == NULL)
00262              return format("No such checksum algorithm: %s",
00263                                  algorithmName);
00264     }
00265 
00266     // compare calculated checksum with the one given
00267     char calculated[MAX_LENGTH];
00268     algo->writeChecksum(data, calculated);
00269     if (strcmp(calculated, checksum) != 0) {
00270         return format("Checksum doesn't match: expected %s "
00271                             "but calculated %s", checksum, calculated);
00272     }
00273 
00274     return std::string();
00275 }

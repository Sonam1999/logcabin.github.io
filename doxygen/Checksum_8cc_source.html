00061      * Calculate the checksum for some data.
00062      * \param data
00063      *      An list of (pointer, length) pairs describing what to checksum.
00064      * \param[out] result
00065      *      The result of the hash function will be placed here.
00066      *      This will be a null-terminated, printable C-string.
00067      * \return
00068      *      The number of valid characters in 'output', including the null
00069      *      terminator. This is guaranteed to be greater than 1.
00070      */
00071     uint32_t
00072     writeChecksum(std::initializer_list<std::pair<const void*, uint64_t>> data,
00073                   char result[MAX_LENGTH]) {
00074         // copy name and : to result
00075         memcpy(result, name.c_str(), name.length());
00076         result = result + name.length();
00077         *result = ':';
00078         ++result;
00079 
00080         // calculate binary digest
00081         uint32_t binarySize = downCast<uint32_t>(hashFn->DigestSize());
00082         uint8_t binary[binarySize];
00083 
00084         for (auto it = data.begin(); it != data.end(); ++it) {
00085             hashFn->Update(static_cast<const uint8_t*>(it->first),
00086                            it->second);
00087         }
00088         hashFn->Final(binary);
00089 
00090         // add hex digest to result
00091         const char* hexArray = "0123456789abcdef";
00092         for (uint32_t i = 0; i < binarySize; ++i) {
00093             *result = hexArray[binary[i] >> 4];
00094             ++result;
00095             *result = hexArray[binary[i] & 15];
00096             ++result;
00097         }
00098 
00099         // add null terminator to result and return total length
00100         *result = '\0';
00101         return outputSize;
00102     }

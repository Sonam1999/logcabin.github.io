00033     const FilesystemUtil::File& parentDir = storageLayout.logDir;
00034     std::string module =

00036     std::unique_ptr<Log> log;
00037     if (module == "Memory") {
00038         log.reset(new MemoryLog());

00040         log.reset(new SimpleFileLog(parentDir));
00041     } else if (module == "Segmented" ||
00042                module == "Segmented-Binary") {
00043         log.reset(new SegmentedLog(parentDir,
00044                                    SegmentedLog::Encoding::BINARY,

00047         log.reset(new SegmentedLog(parentDir,
00048                                    SegmentedLog::Encoding::TEXT,
00049                                    config));
00050     } else {
00051         EXIT("Unknown storage module from config file: %s", module.c_str());
00052     }
00053     return log;

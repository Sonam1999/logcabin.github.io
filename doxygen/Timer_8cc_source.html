00058 void
00059 Timer::schedule(uint64_t nanoseconds)
00060 {
00061     // avoid accidental de-schedules: epoll's semantics are that a timer for 0
00062     // seconds and 0 nanoseconds will never fire.
00063     if (nanoseconds == 0)
00064         nanoseconds = 1;
00065 
00066     const uint64_t nanosPerSecond = 1000 * 1000 * 1000;
00067     struct itimerspec newValue;
00068     memset(&newValue, 0, sizeof(newValue));
00069     newValue.it_value.tv_sec  = nanoseconds / nanosPerSecond;
00070     newValue.it_value.tv_nsec = nanoseconds % nanosPerSecond;
00071     int r = timerfd_settime(fd, 0, &newValue, NULL);
00072     if (r != 0) {
00073         PANIC("Could not set timer to +%luns: %s",
00074               nanoseconds,
00075               strerror(errno));
00076     }
00077 }

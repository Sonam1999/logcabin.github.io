00039 /**
00040  * Keeps state needed by the random number generator, protected by a mutex.
00041  */
00042 class RandomState {
00043   public:
00044     RandomState()
00045         : mutex()
00046         , init(false)
00047         , statebuf()
00048         , randbuf()
00049     {
00050         int fd = open("/dev/urandom", O_RDONLY);
00051         if (fd < 0) {
00052             // too early to call PANIC in here
00053             fprintf(stderr, "Couldn't open /dev/urandom: %s\n",
00054                     strerror(errno));
00055             abort();
00056         }
00057         unsigned int seed;
00058         ssize_t bytesRead = read(fd, &seed, sizeof(seed));
00059         close(fd);
00060         if (bytesRead != sizeof(seed)) {
00061             // too early to call PANIC in here
00062             fprintf(stderr, "Couldn't read full seed from /dev/urandom\n");
00063             abort();
00064         }
00065         initstate_r(seed, statebuf, STATE_BYTES, &randbuf);
00066         static_assert(RAND_MAX >= (1 << 31), "RAND_MAX too small");
00067         init = true;
00068     }

00070     uint64_t random64() {
00071         std::unique_lock<std::mutex> lockGuard(mutex);
00072         if (!init) {
00073             // probably too early to call PANIC in here
00074             fprintf(stderr, "Looks like you hit the so-called static "
00075                     "initialization order fiasco in Core::Random\n");
00076             abort();
00077         }
00078 
00079         // Each call to random returns 31 bits of randomness,
00080         // so we need three to get 64 bits of randomness.
00081         int32_t lo, mid, hi;
00082         random_r(&randbuf, &lo);
00083         random_r(&randbuf, &mid);
00084         random_r(&randbuf, &hi);
00085         uint64_t r = (((uint64_t(hi) & 0x7FFFFFFF) << 33) | // NOLINT
00086                       ((uint64_t(mid) & 0x7FFFFFFF) << 2)  | // NOLINT
00087                       (uint64_t(lo) & 0x00000003)); // NOLINT
00088         return r;
00089     }

00121 /**
00122  * Fill a variable of type T with some random bytes.
00123  */
00124 template<typename T>
00125 T
00126 getRandomBytes()
00127 {
00128     T buf;
00129     size_t offset = 0;
00130     while (offset < sizeof(buf)) {
00131         uint64_t r = randomState.random64();
00132         size_t copy = std::min(sizeof(r), sizeof(buf) - offset);
00133         memcpy(reinterpret_cast<char*>(&buf) + offset, &r, copy);
00134         offset += copy;
00135     }
00136     return buf;
00137 }

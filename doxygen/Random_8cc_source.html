00037 // Internal scratch state used by random_r 128 is the same size as
00038 // initstate() uses for regular random(), see manpages for details.
00039 // statebuf is malloc'ed and this memory is leaked, it could be a __thread
00040 // buffer, but after running into linker issues with large thread local
00041 // storage buffers, we thought better.
00042 enum { STATE_BYTES = 128 };
00043 __thread char* statebuf;
00044 // random_r's state, must be handed to each call, and seems to refer to
00045 // statebuf in some undocumented way.
00046 __thread random_data randbuf;

00048 /**
00049  * Initialize thread-local state for random number generator.
00050  * Must be called before any random numbers are generated.
00051  */
00052 void
00053 initRandom()
00054 {
00055     if (statebuf == NULL) {
00056         int fd = open("/dev/urandom", O_RDONLY);
00057         if (fd < 0)
00058             PANIC("Couldn't open /dev/urandom: %s", strerror(errno));
00059         unsigned int seed;
00060         ssize_t bytesRead = read(fd, &seed, sizeof(seed));
00061         close(fd);
00062         if (bytesRead != sizeof(seed))
00063             PANIC("Couldn't read full seed from /dev/urandom");
00064         statebuf = static_cast<char*>(malloc(STATE_BYTES));
00065         initstate_r(seed, statebuf, STATE_BYTES, &randbuf);
00066     }
00067 
00068     static_assert(RAND_MAX >= (1 << 31), "RAND_MAX too small");
00069 }

00071 /**
00072  * Fill a variable of type T with some random bytes.
00073  */
00074 template<typename T>
00075 T
00076 getRandomBytes()
00077 {
00078     T buf;
00079     size_t offset = 0;
00080     while (offset < sizeof(buf)) {
00081         uint64_t r = random64();
00082         size_t copy = std::min(sizeof(r), sizeof(buf) - offset);
00083         memcpy(reinterpret_cast<char*>(&buf) + offset, &r, copy);
00084         offset += copy;
00085     }
00086     return buf;
00087 }

00117 uint64_t
00118 random64()
00119 {
00120     // Each call to random returns 31 bits of randomness,
00121     // so we need three to get 64 bits of randomness.
00122     initRandom();
00123     int32_t lo, mid, hi;
00124     random_r(&randbuf, &lo);
00125     random_r(&randbuf, &mid);
00126     random_r(&randbuf, &hi);
00127     uint64_t r = (((uint64_t(hi) & 0x7FFFFFFF) << 33) | // NOLINT
00128                   ((uint64_t(mid) & 0x7FFFFFFF) << 2)  | // NOLINT
00129                   (uint64_t(lo) & 0x00000003)); // NOLINT
00130     return r;
00131 }

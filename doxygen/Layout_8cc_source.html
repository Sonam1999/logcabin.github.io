00075 void
00076 Layout::init(const std::string& storagePath, uint64_t serverId)
00077 {
00078     if (removeAllFiles) {
00079         Storage::FilesystemUtil::remove(topDir.path);
00080         removeAllFiles = false;
00081     }
00082     topDir = FS::openDir(storagePath);
00083     serverDir = FS::openDir(
00084         topDir,
00085         Core::StringUtil::format("server%lu", serverId));
00086     // We used to lock serverDir, but that doesn't work across NFS clients, at
00087     // least on RHEL6. Locking a file within the directory does seem to work.
00088     lockFile = FS::openFile(
00089         serverDir,
00090         "lock",
00091         O_CREAT);
00092     // lock file so that Storage/Tool doesn't use serverDir while the daemon is
00093     // running
00094     std::string error = FS::tryFlock(lockFile, LOCK_EX|LOCK_NB);
00095     if (!error.empty()) {
00096         PANIC("Could not lock storage directory. Is LogCabin already running? "
00097               "Error was: %s", error.c_str());
00098     }
00099 }

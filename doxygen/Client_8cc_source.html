00224 {
00225     // This method sets the working directory regardless of whether it
00226     // succeeds -- that way if it doesn't, future relative paths on this Tree
00227     // will result in errors instead of operating on the prior working
00228     // directory.
00229 
00230     ClientImpl::TimePoint timeout =
00231         ClientImpl::absTimeout(treeDetails->timeoutNanos);
00232 
00233     std::lock_guard<std::mutex> lockGuard(mutex);
00234     std::string realPath;
00235     Result result = treeDetails->clientImpl->canonicalize(
00236                                 newWorkingDirectory,
00237                                 treeDetails->workingDirectory,
00238                                 realPath);
00239     std::shared_ptr<TreeDetails> newTreeDetails(new TreeDetails(*treeDetails));
00240     if (result.status != Status::OK) {
00241         newTreeDetails->workingDirectory = Core::StringUtil::format(
00242                     "invalid from prior call to setWorkingDirectory('%s') "
00243                     "relative to '%s'",
00244                     newWorkingDirectory.c_str(),
00245                     treeDetails->workingDirectory.c_str());
00246         treeDetails = newTreeDetails;
00247         return result;
00248     }
00249     newTreeDetails->workingDirectory = realPath;
00250     treeDetails = newTreeDetails;

00029  * Return the absolute time when the calling operation should timeout.
00030  */
00031 ClientImpl::TimePoint
00032 absTimeout(uint64_t relTimeoutNanos)
00033 {
00034     if (relTimeoutNanos == 0)
00035         return ClientImpl::TimePoint::max();
00036     ClientImpl::TimePoint now = ClientImpl::Clock::now();
00037     ClientImpl::TimePoint then =
00038         now + std::chrono::nanoseconds(relTimeoutNanos);
00039     if (then < now) // overflow
00040         return ClientImpl::TimePoint::max();
00041     else
00042         return then;
00043 }

00241 Result
00242 Tree::setWorkingDirectory(const std::string& newWorkingDirectory)
00243 {
00244     // This method sets the working directory regardless of whether it
00245     // succeeds -- that way if it doesn't, future relative paths on this Tree
00246     // will result in errors instead of operating on the prior working
00247     // directory.
00248 
00249     ClientImpl::TimePoint timeout = absTimeout(treeDetails->timeoutNanos);
00250 
00251     std::unique_lock<std::mutex> lockGuard(mutex);
00252     std::string realPath;
00253     Result result = treeDetails->clientImpl->canonicalize(
00254                                 newWorkingDirectory,
00255                                 treeDetails->workingDirectory,
00256                                 realPath);
00257     std::shared_ptr<TreeDetails> newTreeDetails(new TreeDetails(*treeDetails));
00258     if (result.status != Status::OK) {
00259         newTreeDetails->workingDirectory = Core::StringUtil::format(
00260                     "invalid from prior call to setWorkingDirectory('%s') "
00261                     "relative to '%s'",
00262                     newWorkingDirectory.c_str(),
00263                     treeDetails->workingDirectory.c_str());
00264         treeDetails = newTreeDetails;
00265         return result;
00266     }
00267     newTreeDetails->workingDirectory = realPath;
00268     treeDetails = newTreeDetails;
00269     return treeDetails->clientImpl->makeDirectory(realPath, "",
00270                                                   treeDetails->condition,

00038 void
00039 ControlService::handleRPC(RPC::ServerRPC rpc)
00040 {
00041     using Protocol::ServerControl::OpCode;
00042 
00043     // Call the appropriate RPC handler based on the request's opCode.
00044     switch (rpc.getOpCode()) {
00045         case OpCode::DEBUG_FILENAME_GET:
00046             debugFilenameGet(std::move(rpc));
00047             break;
00048         case OpCode::DEBUG_FILENAME_SET:
00049             debugFilenameSet(std::move(rpc));
00050             break;
00051         case OpCode::DEBUG_ROTATE:
00052             debugRotate(std::move(rpc));
00053             break;
00054         case OpCode::SERVER_INFO_GET:
00055             serverInfoGet(std::move(rpc));
00056             break;
00057         case OpCode::SERVER_STATS_DUMP:
00058             serverStatsDump(std::move(rpc));
00059             break;
00060         case OpCode::SERVER_STATS_GET:
00061             serverStatsGet(std::move(rpc));
00062             break;
00063         default:
00064             WARNING("Client sent request with bad op code (%u) to "
00065                     "ControlService", rpc.getOpCode());
00066             rpc.rejectInvalidRequest();
00067     }
00068 }

00097 void
00098 ControlService::debugFilenameSet(RPC::ServerRPC rpc)
00099 {
00100     PRELUDE(DebugFilenameSet);
00101     std::string prev = Core::Debug::getLogFilename();
00102     NOTICE("Switching to log file %s",
00103            request.filename().c_str());
00104     std::string error = Core::Debug::setLogFilename(request.filename());
00105     if (error.empty()) {
00106         NOTICE("Switched from log file %s",
00107                prev.c_str());
00108     } else {
00109         ERROR("Failed to switch to log file %s: %s",
00110               request.filename().c_str(),
00111               error.c_str());
00112         response.set_error(error);
00113     }
00114     rpc.reply(response);

00037 void
00038 ControlService::handleRPC(RPC::ServerRPC rpc)
00039 {
00040     using Protocol::ServerControl::OpCode;
00041 
00042     // Call the appropriate RPC handler based on the request's opCode.
00043     switch (rpc.getOpCode()) {
00044         case OpCode::DEBUG_FILENAME_GET:
00045             debugFilenameGet(std::move(rpc));
00046             break;
00047         case OpCode::DEBUG_FILENAME_SET:
00048             debugFilenameSet(std::move(rpc));
00049             break;
00050         case OpCode::DEBUG_POLICY_GET:
00051             debugPolicyGet(std::move(rpc));
00052             break;
00053         case OpCode::DEBUG_POLICY_SET:
00054             debugPolicySet(std::move(rpc));
00055             break;
00056         case OpCode::DEBUG_ROTATE:
00057             debugRotate(std::move(rpc));
00058             break;
00059         case OpCode::SERVER_INFO_GET:
00060             serverInfoGet(std::move(rpc));
00061             break;
00062         case OpCode::SERVER_STATS_DUMP:
00063             serverStatsDump(std::move(rpc));
00064             break;
00065         case OpCode::SERVER_STATS_GET:
00066             serverStatsGet(std::move(rpc));
00067             break;
00068         default:
00069             WARNING("Client sent request with bad op code (%u) to "
00070                     "ControlService", rpc.getOpCode());
00071             rpc.rejectInvalidRequest();
00072     }
00073 }

00103 ControlService::debugFilenameSet(RPC::ServerRPC rpc)
00104 {
00105     PRELUDE(DebugFilenameSet);
00106     std::string prev = Core::Debug::getLogFilename();
00107     NOTICE("Switching to log file %s",
00108            request.filename().c_str());
00109     std::string error = Core::Debug::setLogFilename(request.filename());
00110     if (error.empty()) {
00111         NOTICE("Switched from log file %s",

00114         ERROR("Failed to switch to log file %s: %s",
00115               request.filename().c_str(),

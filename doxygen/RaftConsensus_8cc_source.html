00242               std::unique_lock<Mutex>& lockGuard)

00246                          Protocol::Common::ServiceId::RAFT_SERVICE,
00247                          /* serviceSpecificErrorVersion = */ 0,
00248                          opCode,

00250     // release lock for concurrency
00251     Core::MutexUnlock<Mutex> unlockGuard(lockGuard);
00252     switch (rpc.waitForReply(&response, NULL, TimePoint::max())) {

00256                         rpcFailuresSinceLastWarning);

00260         case RPCStatus::SERVICE_SPECIFIC_ERROR:
00261             PANIC("unexpected service-specific error");
00262         case RPCStatus::TIMEOUT:
00263             PANIC("unexpected RPC timeout");
00264         case RPCStatus::RPC_FAILED:
00265             ++rpcFailuresSinceLastWarning;
00266             if (rpcFailuresSinceLastWarning == 1) {
00267                 WARNING("RPC to server failed: %s",
00268                         rpc.getErrorMessage().c_str());
00269             } else if (rpcFailuresSinceLastWarning % 100 == 0) {
00270                 WARNING("Last %lu RPCs to server failed. This failure: %s",
00271                         rpcFailuresSinceLastWarning,
00272                         rpc.getErrorMessage().c_str());
00273             }
00274             return false;
00275         case RPCStatus::RPC_CANCELED:

00510 void
00511 Configuration::setConfiguration(
00512         uint64_t newId,
00513         const Protocol::Raft::Configuration& newDescription)
00514 {
00515     NOTICE("Activating configuration %lu:\n%s", newId,
00516            Core::ProtoBuf::dumpString(newDescription).c_str());
00517 
00518     if (newDescription.next_configuration().servers().size() == 0)
00519         state = State::STABLE;
00520     else
00521         state = State::TRANSITIONAL;
00522     id = newId;
00523     description = newDescription;
00524     oldServers.servers.clear();
00525     newServers.servers.clear();
00526 
00527     // Build up the list of old servers
00528     for (auto confIt = description.prev_configuration().servers().begin();
00529          confIt != description.prev_configuration().servers().end();
00530          ++confIt) {
00531         std::shared_ptr<Server> server = getServer(confIt->server_id());
00532         server->address = confIt->address();
00533         oldServers.servers.push_back(server);
00534     }
00535 
00536     // Build up the list of new servers
00537     for (auto confIt = description.next_configuration().servers().begin();
00538          confIt != description.next_configuration().servers().end();
00539          ++confIt) {
00540         std::shared_ptr<Server> server = getServer(confIt->server_id());
00541         server->address = confIt->address();
00542         newServers.servers.push_back(server);
00543     }
00544 
00545     // Servers not in the current configuration need to be told to exit
00546     setGCFlag(*localServer);
00547     oldServers.forEach(setGCFlag);
00548     newServers.forEach(setGCFlag);
00549     auto it = knownServers.begin();
00550     while (it != knownServers.end()) {
00551         std::shared_ptr<Server> server = it->second;
00552         if (!server->gcFlag) {
00553             server->exit();
00554             it = knownServers.erase(it);
00555         } else {
00556             server->gcFlag = false; // clear flag for next time
00557             ++it;
00558         }
00559     }

00927 Consensus::Entry
00928 RaftConsensus::getNextEntry(uint64_t lastEntryId) const
00929 {
00930     std::unique_lock<Mutex> lockGuard(mutex);
00931     uint64_t nextEntryId = lastEntryId + 1;
00932     while (true) {
00933         if (exiting)
00934             throw ThreadInterruptedException();

00936             Consensus::Entry entry;
00937 
00938             // Make the state machine load a snapshot if we don't have the next
00939             // entry it needs in the log.
00940             if (log->getLogStartIndex() > nextEntryId) {
00941                 entry.type = Consensus::Entry::SNAPSHOT;
00942                 // For well-behaved state machines, we expect 'snapshotReader'
00943                 // to contain a SnapshotFile::Reader that we can return
00944                 // directly to the state machine. In the case that a State

00946                 // SnapshotFile::Reader again.
00947                 entry.snapshotReader = std::move(snapshotReader);

00949                     WARNING("State machine asked for same snapshot twice; "
00950                             "this shouldn't happen in normal operation. "
00951                             "Having to re-read it from disk.");
00952                     // readSnapshot() shouldn't have any side effects since the
00953                     // snapshot should have already been read, so const_cast
00954                     // should be ok (though ugly).
00955                     const_cast<RaftConsensus*>(this)->readSnapshot();
00956                     entry.snapshotReader = std::move(snapshotReader);
00957                 }
00958                 entry.entryId = lastSnapshotIndex;
00959             } else {

00965                     entry.data = logEntry.data();
00966                 } else {

00968                 }
00969             }

00973     }

00991 void
00992 RaftConsensus::handleAppendEntries(
00993                     const Protocol::Raft::AppendEntries::Request& request,
00994                     Protocol::Raft::AppendEntries::Response& response)
00995 {
00996     std::unique_lock<Mutex> lockGuard(mutex);
00997     assert(!exiting);
00998 
00999     // Set response to a rejection. We'll overwrite these later if we end up
01000     // accepting the request.
01001     response.set_term(currentTerm);
01002     response.set_success(false);
01003 
01004     // If the caller's term is stale, just return our term to it.
01005     if (request.term() < currentTerm) {
01006         VERBOSE("Caller(%lu) is stale. Our term is %lu, theirs is %lu",
01007                  request.server_id(), currentTerm, request.term());
01008         return; // response was set to a rejection above
01009     }
01010     if (request.term() > currentTerm) {
01011         VERBOSE("Caller(%lu) has newer term, updating. "
01012                 "Ours was %lu, theirs is %lu",
01013                 request.server_id(), currentTerm, request.term());
01014         // We're about to bump our term in the stepDown below: update
01015         // 'response' accordingly.
01016         response.set_term(request.term());
01017     }
01018     // This request is a sign of life from the current leader. Update our term
01019     // and convert to follower if necessary; reset the election timer.
01020     stepDown(request.term());
01021     setElectionTimer();
01022     withholdVotesUntil = Clock::now() +
01023                          std::chrono::milliseconds(ELECTION_TIMEOUT_MS);
01024 
01025     // Record the leader ID as a hint for clients.
01026     if (leaderId == 0) {
01027         leaderId = request.server_id();
01028         NOTICE("All hail leader %lu for term %lu", leaderId, currentTerm);
01029     } else {
01030         assert(leaderId == request.server_id());
01031     }
01032 
01033     // For an entry to fit into our log, it must not leave a gap.
01034     if (request.prev_log_index() > log->getLastLogIndex()) {
01035         VERBOSE("Rejecting AppendEntries RPC: would leave gap");
01036         return; // response was set to a rejection above
01037     }
01038     // It must also agree with the previous entry in the log (and, inductively
01039     // all prior entries).
01040     // Always match on index 0, and always match on any discarded indexes:
01041     // since we know those were committed, the leader must agree with them.
01042     // We could truncate the log here, but there's no real advantage to doing
01043     // that.
01044     if (request.prev_log_index() >= log->getLogStartIndex() &&
01045         log->getEntry(request.prev_log_index()).term() !=
01046             request.prev_log_term()) {
01047         VERBOSE("Rejecting AppendEntries RPC: terms don't agree");
01048         return; // response was set to a rejection above
01049     }
01050 
01051     // If we got this far, we're accepting the request.
01052     response.set_success(true);
01053 
01054     // This needs to be able to handle duplicated RPC requests. We compare the
01055     // entries' terms to know if we need to do the operation; otherwise,
01056     // reapplying requests can result in data loss.
01057     //
01058     // The first problem this solves is that an old AppendEntries request may be
01059     // duplicated and received after a newer request, which could cause
01060     // undesirable data loss. For example, suppose the leader appends entry 4
01061     // and then entry 5, but the follower receives 4, then 5, then 4 again.
01062     // Without this extra guard, the follower would truncate 5 out of its
01063     // log.
01064     //
01065     // The second problem is more subtle: if the same request is duplicated but
01066     // the leader processes an earlier response, it will assume the
01067     // acknowledged data is safe. However, there is a window of vulnerability
01068     // on the follower's disk between the truncate and append operations (which
01069     // are not done atomically) when the follower processes the later request.
01070     uint64_t entryId = request.prev_log_index();
01071     for (auto it = request.entries().begin();
01072          it != request.entries().end();
01073          ++it) {
01074         ++entryId;
01075         const Protocol::Raft::Entry& entry = *it;
01076         if (entryId < log->getLogStartIndex()) {
01077             // We already snapshotted and discarded this index, so presumably
01078             // we've received a committed entry we once already had.
01079             continue;
01080         }
01081         if (log->getLastLogIndex() >= entryId) {
01082             if (log->getEntry(entryId).term() == entry.term())
01083                 continue;
01084             // should never truncate committed entries:
01085             assert(commitIndex < entryId);
01086             NOTICE("Truncating %lu entries after %lu from the log",
01087                    log->getLastLogIndex() - entryId + 1,
01088                    entryId - 1);
01089             log->truncateSuffix(entryId - 1);
01090             configurationManager->truncateSuffix(entryId - 1);
01091         }
01092 
01093         // Append this and all following entries.
01094         std::vector<const Protocol::Raft::Entry*> entries;
01095         while (it != request.entries().end()) {
01096             entries.push_back(&*it);
01097             ++it;
01098         }
01099         append(entries);
01100         break;
01101     }
01102 
01103     // Set our committed ID from the request's. In rare cases, this would make
01104     // our committed ID decrease. For example, this could happen with a new
01105     // leader who has not yet replicated one of its own entries. While that'd
01106     // be perfectly safe, guarding against it with an if statement lets us
01107     // make stronger assertions.
01108     if (commitIndex < request.commit_index()) {
01109         commitIndex = request.commit_index();
01110         assert(commitIndex <= log->getLastLogIndex());
01111         stateChanged.notify_all();
01112         VERBOSE("New commitIndex: %lu", commitIndex);
01113     }
01114 }

01261 RaftConsensus::ClientResult
01262 RaftConsensus::setConfiguration(
01263         uint64_t oldId,
01264         const Protocol::Raft::SimpleConfiguration& nextConfiguration)
01265 {
01266     std::unique_lock<Mutex> lockGuard(mutex);
01267 
01268     if (state != State::LEADER)
01269         return ClientResult::NOT_LEADER;
01270     if (configuration->id != oldId ||
01271         configuration->state != Configuration::State::STABLE) {
01272         // configurations has changed in the meantime
01273         return ClientResult::FAIL;
01274     }
01275 
01276     uint64_t term = currentTerm;
01277     configuration->setStagingServers(nextConfiguration);
01278     stateChanged.notify_all();
01279 
01280     // Wait for new servers to be caught up. This will abort if not every
01281     // server makes progress in a ELECTION_TIMEOUT_MS period.
01282     ++currentEpoch;
01283     uint64_t epoch = currentEpoch;
01284     TimePoint checkProgressAt =
01285         Clock::now() + std::chrono::milliseconds(ELECTION_TIMEOUT_MS);
01286     while (true) {
01287         if (exiting || term != currentTerm)
01288             return ClientResult::NOT_LEADER;
01289         if (configuration->stagingAll(&Server::isCaughtUp))
01290             break;
01291         if (Clock::now() >= checkProgressAt) {
01292             if (configuration->stagingMin(&Server::getLastAckEpoch) < epoch) {
01293                 configuration->resetStagingServers();
01294                 stateChanged.notify_all();
01295                 // TODO(ongaro): probably need to return a different type of
01296                 // message: confuses oldId mismatch from new server down
01297                 return ClientResult::FAIL;
01298             } else {
01299                 ++currentEpoch;
01300                 epoch = currentEpoch;
01301                 checkProgressAt =
01302                     (Clock::now() +
01303                      std::chrono::milliseconds(ELECTION_TIMEOUT_MS));
01304             }
01305         }
01306         stateChanged.wait_until(lockGuard, checkProgressAt);
01307     }
01308 
01309     // Write and commit transitional configuration
01310     Protocol::Raft::Configuration newConfiguration;
01311     *newConfiguration.mutable_prev_configuration() =
01312         configuration->description.prev_configuration();
01313     *newConfiguration.mutable_next_configuration() = nextConfiguration;
01314     Log::Entry entry;
01315     entry.set_type(Protocol::Raft::EntryType::CONFIGURATION);
01316     *entry.mutable_configuration() = newConfiguration;
01317     std::pair<ClientResult, uint64_t> result =
01318         replicateEntry(entry, lockGuard);
01319     if (result.first != ClientResult::SUCCESS)
01320         return result.first;
01321     uint64_t transitionalId = result.second;
01322 
01323     // Wait until the configuration that removes the old servers has been
01324     // committed. This is the first configuration with ID greater than
01325     // transitionalId.
01326     while (true) {
01327         // Check this first: if the new configuration excludes us so we've
01328         // stepped down upon committing it, we still want to return success.
01329         if (configuration->id > transitionalId &&
01330             commitIndex >= configuration->id) {
01331             return ClientResult::SUCCESS;
01332         }
01333         if (exiting || term != currentTerm)
01334             return ClientResult::NOT_LEADER;
01335         stateChanged.wait(lockGuard);
01336     }
01337 }

01339 std::unique_ptr<Storage::SnapshotFile::Writer>
01340 RaftConsensus::beginSnapshot(uint64_t lastIncludedIndex)
01341 {
01342     std::unique_lock<Mutex> lockGuard(mutex);
01343 
01344     NOTICE("Creating new snapshot through log index %lu (inclusive)",
01345            lastIncludedIndex);
01346     std::unique_ptr<Storage::SnapshotFile::Writer> writer(
01347                 new Storage::SnapshotFile::Writer(storageDirectory));
01348 
01349     // Only committed entries may be snapshotted.
01350     // (This check relies on commitIndex monotonically increasing.)
01351     if (lastIncludedIndex > commitIndex) {
01352         PANIC("Attempted to snapshot uncommitted entries (%lu requested but "
01353               "%lu is last committed entry)", lastIncludedIndex, commitIndex);
01354     }
01355 
01356     // set header fields
01357     SnapshotMetadata::Header header;
01358     header.set_last_included_index(lastIncludedIndex);
01359     // Set last_included_term:
01360     if (lastIncludedIndex >= log->getLogStartIndex() &&
01361         lastIncludedIndex <= log->getLastLogIndex()) {
01362         header.set_last_included_term(log->getEntry(lastIncludedIndex).term());
01363     } else if (lastIncludedIndex == 0) {
01364         WARNING("Taking a snapshot covering no log entries");
01365         header.set_last_included_term(0);
01366     } else if (lastIncludedIndex == lastSnapshotIndex) {
01367         WARNING("Taking a snapshot where we already have one, covering "
01368                 "entries 1 through %lu (inclusive)", lastIncludedIndex);
01369         header.set_last_included_term(lastSnapshotTerm);
01370     } else {
01371         WARNING("We've already discarded the entries that the state machine "
01372                 "wants to snapshot. This can happen in rare cases if the "
01373                 "leader already sent us a newer snapshot. We'll go ahead and "
01374                 "compute the snapshot, but it'll be discarded later in "
01375                 "snapshotDone(). Setting the last included term in the "
01376                 "snapshot header to 0 (a bogus value).");
01377         // If this turns out to be common, we should return NULL instead and
01378         // change the state machines to deal with that.
01379         header.set_last_included_term(0);
01380     }
01381     // Copy the configuration as of lastIncludedIndex to the header.
01382     std::pair<uint64_t, Protocol::Raft::Configuration> c =
01383         configurationManager->getLatestConfigurationAsOf(lastIncludedIndex);
01384     if (c.first == 0) {
01385         WARNING("Taking snapshot with no configuration. "
01386                 "This should have been the first thing in the log.");
01387     } else {
01388         header.set_configuration_index(c.first);
01389         *header.mutable_configuration() = c.second;
01390     }
01391 
01392     // write header to file
01393     google::protobuf::io::CodedOutputStream& stream = writer->getStream();
01394     int size = header.ByteSize();
01395     stream.WriteLittleEndian32(size);
01396     header.SerializeWithCachedSizes(&stream);
01397 
01398     return writer;
01399 }

01401 void
01402 RaftConsensus::snapshotDone(
01403         uint64_t lastIncludedIndex,
01404         std::unique_ptr<Storage::SnapshotFile::Writer> writer)
01405 {
01406     std::unique_lock<Mutex> lockGuard(mutex);
01407     if (lastIncludedIndex <= lastSnapshotIndex) {
01408         NOTICE("Discarding snapshot through %lu since we already have one "
01409                "(presumably from another server) through %lu",
01410                lastIncludedIndex, lastSnapshotIndex);
01411         writer->discard();
01412         return;
01413     }
01414 
01415     // log->getEntry(lastIncludedIndex) is safe:
01416     // If the log prefix for this snapshot was truncated, that means we have a
01417     // newer snapshot (handled above).
01418     assert(lastIncludedIndex >= log->getLogStartIndex());
01419     // We never truncate committed entries from the end of our log, and
01420     // beginSnapshot() made sure that lastIncludedIndex covers only committed
01421     // entries.
01422     assert(lastIncludedIndex <= log->getLastLogIndex());
01423 
01424     lastSnapshotBytes = writer->save();
01425     lastSnapshotIndex = lastIncludedIndex;
01426     lastSnapshotTerm = log->getEntry(lastIncludedIndex).term();
01427 
01428     // It's easier to grab this configuration out of the manager again than to
01429     // carry it around after writing the header.
01430     std::pair<uint64_t, Protocol::Raft::Configuration> c =
01431         configurationManager->getLatestConfigurationAsOf(lastIncludedIndex);
01432     if (c.first == 0) {
01433         WARNING("Could not find the latest configuration as of index %lu "
01434                 "(inclusive). This shouldn't happen if the snapshot was "
01435                 "created with a configuration, as they should be.",
01436                 lastIncludedIndex);
01437     } else {
01438         configurationManager->setSnapshot(c.first, c.second);
01439     }
01440 
01441     NOTICE("Completed snapshot through log index %lu (inclusive)",
01442            lastSnapshotIndex);
01443 
01444     // It may be beneficial to defer discarding entries if some followers are
01445     // a little bit slow, to avoid having to send them a snapshot when a few
01446     // entries would do the trick. Best to avoid premature optimization though.
01447     discardUnneededEntries();
01448 }

01530 {
01531     std::unique_lock<Mutex> lockGuard(mutex);
01532     Core::ThreadId::setName(
01533         Core::StringUtil::format("Peer(%lu)", peer->serverId));
01534 
01535     // Each iteration of this loop issues a new RPC or sleeps on the condition
01536     // variable.
01537     while (!peer->exiting) {

01540 
01541         if (peer->backoffUntil > now) {
01542             waitUntil = peer->backoffUntil;
01543         } else {

01545                 // Followers don't issue RPCs.
01546                 case State::FOLLOWER:
01547                     waitUntil = TimePoint::max();

01549

01551                 case State::CANDIDATE:

01553                         requestVote(lockGuard, *peer);
01554                     else
01555                         waitUntil = TimePoint::max();
01556                     break;

01558                 // Leaders replicate entries and periodically send heartbeats.
01559                 case State::LEADER:

01565                     } else {

01580 void
01581 RaftConsensus::stepDownThreadMain()
01582 {
01583     std::unique_lock<Mutex> lockGuard(mutex);
01584     Core::ThreadId::setName("stepDown");
01585     while (true) {
01586         // Wait until this server is the leader and is not the only server in
01587         // the cluster.
01588         while (true) {
01589             if (exiting)
01590                 return;
01591             if (state == State::LEADER) {
01592                 // If this local server forms a quorum (it is the only server
01593                 // in the configuration), we need to sleep. Without this guard,
01594                 // this method would not relinquish the CPU.
01595                 ++currentEpoch;
01596                 if (configuration->quorumMin(&Server::getLastAckEpoch) <
01597                     currentEpoch) {
01598                     break;
01599                 }
01600             }
01601             stateChanged.wait(lockGuard);
01602         }
01603         // Now, if an election timeout goes by without confirming leadership,
01604         // step down. The election timeout is a reasonable amount of time,
01605         // since it's about when other servers will start elections and bump
01606         // the term.
01607         TimePoint stepDownAt =
01608             Clock::now() + std::chrono::milliseconds(ELECTION_TIMEOUT_MS);
01609         uint64_t term = currentTerm;
01610         uint64_t epoch = currentEpoch; // currentEpoch was incremented above
01611         while (true) {
01612             if (exiting)
01613                 return;
01614             if (currentTerm > term)
01615                 break;
01616             if (configuration->quorumMin(&Server::getLastAckEpoch) >= epoch)
01617                 break;
01618             if (Clock::now() >= stepDownAt) {
01619                 NOTICE("No broadcast for a timeout, stepping down");
01620                 stepDown(currentTerm + 1);
01621                 break;
01622             }
01623             stateChanged.wait_until(lockGuard, stepDownAt);
01624         }
01625     }
01626 }

01630 void
01631 RaftConsensus::advanceCommittedId()
01632 {
01633     if (state != State::LEADER) {
01634         // getLastAgreeIndex is undefined unless we're leader
01635         WARNING("advanceCommittedId called as %s",
01636                 Core::StringUtil::toString(state).c_str());
01637         return;
01638     }
01639 
01640     // calculate the largest entry ID stored on a quorum of servers
01641     uint64_t newCommittedId =
01642         configuration->quorumMin(&Server::getLastAgreeIndex);
01643     if (commitIndex >= newCommittedId)
01644         return;
01645     // If we have discarded the entry, it's because we already knew it was
01646     // committed.
01647     assert(newCommittedId >= log->getLogStartIndex());
01648     // At least one of these entries must also be from the current term to
01649     // guarantee that no server without them can be elected.
01650     if (log->getEntry(newCommittedId).term() != currentTerm)
01651         return;
01652     commitIndex = newCommittedId;
01653     VERBOSE("New commitIndex: %lu", commitIndex);
01654     assert(commitIndex <= log->getLastLogIndex());
01655     stateChanged.notify_all();
01656 
01657     if (state == State::LEADER && commitIndex >= configuration->id) {
01658         // Upon committing a configuration that excludes itself, the leader
01659         // steps down.
01660         if (!configuration->hasVote(configuration->localServer)) {
01661             stepDown(currentTerm + 1);
01662             return;
01663         }
01664 
01665         // Upon committing a reconfiguration (Cold,new) entry, the leader
01666         // creates the next configuration (Cnew) entry.
01667         if (configuration->state == Configuration::State::TRANSITIONAL) {
01668             Log::Entry entry;
01669             entry.set_term(currentTerm);
01670             entry.set_type(Protocol::Raft::EntryType::CONFIGURATION);
01671             *entry.mutable_configuration()->mutable_prev_configuration() =
01672                 configuration->description.next_configuration();
01673             append({&entry});
01674             return;
01675         }
01676     }
01677 }

01702 void
01703 RaftConsensus::appendEntries(std::unique_lock<Mutex>& lockGuard,
01704                              Peer& peer)
01705 {
01706     uint64_t lastLogIndex = log->getLastLogIndex();
01707     uint64_t prevLogIndex = peer.nextIndex - 1;
01708     assert(prevLogIndex <= lastLogIndex);
01709 
01710     // Don't have needed entry: send a snapshot instead.
01711     if (peer.nextIndex < log->getLogStartIndex()) {
01712         appendSnapshotChunk(lockGuard, peer);
01713         return;
01714     }
01715 
01716     // Find prevLogTerm or fall back to sending a snapshot.
01717     uint64_t prevLogTerm;
01718     if (prevLogIndex >= log->getLogStartIndex()) {
01719         prevLogTerm = log->getEntry(prevLogIndex).term();
01720     } else if (prevLogIndex == 0) {
01721         prevLogTerm = 0;
01722     } else if (prevLogIndex == lastSnapshotIndex) {
01723         prevLogTerm = lastSnapshotTerm;
01724     } else {
01725         // Don't have needed entry for prevLogTerm: send snapshot instead.
01726         appendSnapshotChunk(lockGuard, peer);
01727         return;
01728     }
01729 
01730     // Build up request
01731     Protocol::Raft::AppendEntries::Request request;
01732     request.set_server_id(serverId);
01733     request.set_recipient_id(peer.serverId);
01734     request.set_term(currentTerm);
01735     request.set_prev_log_term(prevLogTerm);
01736     request.set_prev_log_index(prevLogIndex);
01737 
01738     // Add as many as entries as will fit comfortably in the request. It's
01739     // easiest to add one entry at a time until the RPC gets too big, then back
01740     // the last one out.
01741     uint64_t numEntries = 0;
01742     if (!peer.forceHeartbeat) {
01743         for (uint64_t entryId = peer.nextIndex;
01744              entryId <= lastLogIndex;
01745              ++entryId) {
01746             const Log::Entry& entry = log->getEntry(entryId);
01747             *request.add_entries() = entry;
01748             uint64_t requestSize =
01749                 Core::Util::downCast<uint64_t>(request.ByteSize());
01750             if (requestSize < SOFT_RPC_SIZE_LIMIT || numEntries == 0) {
01751                 // this entry fits, send it
01752                 VERBOSE("sending entry <id=%lu,term=%lu>",
01753                         entryId, entry.term());
01754                 ++numEntries;
01755             } else {
01756                 // this entry doesn't fit, discard it
01757                 request.mutable_entries()->RemoveLast();
01758             }
01759         }
01760     }
01761     request.set_commit_index(std::min(commitIndex, prevLogIndex + numEntries));
01762 
01763     // Execute RPC
01764     Protocol::Raft::AppendEntries::Response response;
01765     TimePoint start = Clock::now();
01766     uint64_t epoch = currentEpoch;
01767     bool ok = peer.callRPC(Protocol::Raft::OpCode::APPEND_ENTRIES,
01768                            request, response,
01769                            lockGuard);
01770     if (!ok) {
01771         peer.backoffUntil = start +
01772             std::chrono::milliseconds(RPC_FAILURE_BACKOFF_MS);
01773         return;
01774     }
01775 
01776     // Process response
01777 
01778     if (currentTerm != request.term() || peer.exiting) {
01779         // we don't care about result of RPC
01780         return;
01781     }
01782     // Since we were leader in this term before, we must still be leader in
01783     // this term.
01784     assert(state == State::LEADER);
01785     if (response.term() > currentTerm) {
01786         stepDown(response.term());
01787     } else {
01788         assert(response.term() == currentTerm);
01789         peer.lastAckEpoch = epoch;
01790         stateChanged.notify_all();
01791         peer.nextHeartbeatTime = start +
01792             std::chrono::milliseconds(HEARTBEAT_PERIOD_MS);
01793         if (response.success()) {
01794             if (peer.lastAgreeIndex > prevLogIndex + numEntries) {
01795                 // Revisit this warning if we pipeline AppendEntries RPCs for
01796                 // performance.
01797                 WARNING("lastAgreeIndex should monotonically increase within a "
01798                         "term, since servers don't forget entries. But it "
01799                         "didn't.");
01800             } else {
01801                 peer.lastAgreeIndex = prevLogIndex + numEntries;
01802                 advanceCommittedId();
01803             }
01804             peer.nextIndex = peer.lastAgreeIndex + 1;
01805             peer.forceHeartbeat = false;
01806 
01807             if (!peer.isCaughtUp_ &&
01808                 peer.thisCatchUpIterationGoalId <= peer.lastAgreeIndex) {
01809                 Clock::duration duration =
01810                     Clock::now() - peer.thisCatchUpIterationStart;
01811                 uint64_t thisCatchUpIterationMs =
01812                     uint64_t(std::chrono::duration_cast<
01813                                  std::chrono::milliseconds>(duration).count());
01814                 if (uint64_t(labs(int64_t(peer.lastCatchUpIterationMs -
01815                                           thisCatchUpIterationMs))) <
01816                     ELECTION_TIMEOUT_MS) {
01817                     peer.isCaughtUp_ = true;
01818                     stateChanged.notify_all();
01819                 } else {
01820                     peer.lastCatchUpIterationMs = thisCatchUpIterationMs;
01821                     peer.thisCatchUpIterationStart = Clock::now();
01822                     peer.thisCatchUpIterationGoalId = log->getLastLogIndex();
01823                 }
01824             }
01825         } else {
01826             if (peer.nextIndex > 1)
01827                 --peer.nextIndex;
01828         }
01829     }
01830 }

01832 void
01833 RaftConsensus::appendSnapshotChunk(std::unique_lock<Mutex>& lockGuard,
01834                                    Peer& peer)
01835 {
01836     // Build up request
01837     Protocol::Raft::AppendSnapshotChunk::Request request;
01838     request.set_server_id(serverId);
01839     request.set_recipient_id(peer.serverId);
01840     request.set_term(currentTerm);
01841 
01842     // Open the latest snapshot if we haven't already. Stash a copy of the
01843     // lastSnapshotIndex that goes along with the file, since it's possible
01844     // that this will change while we're transferring chunks).
01845     if (!peer.snapshotFile) {
01846         namespace FilesystemUtil = Storage::FilesystemUtil;
01847         peer.snapshotFile.reset(new FilesystemUtil::FileContents(
01848             FilesystemUtil::openFile(storageDirectory, "snapshot", O_RDONLY)));
01849         peer.snapshotFileOffset = 0;
01850         peer.lastSnapshotIndex = lastSnapshotIndex;
01851     }
01852     request.set_last_snapshot_index(peer.lastSnapshotIndex);
01853     request.set_byte_offset(peer.snapshotFileOffset);
01854     // The amount of data we can send is bounded by the remaining bytes in the
01855     // file and the maximum length for RPCs.
01856     uint64_t numDataBytes = std::min(
01857         peer.snapshotFile->getFileLength() - peer.snapshotFileOffset,
01858         SOFT_RPC_SIZE_LIMIT);
01859     request.set_data(peer.snapshotFile->get<char>(peer.snapshotFileOffset,
01860                                                   numDataBytes),
01861                      numDataBytes);
01862     request.set_done(peer.snapshotFileOffset + numDataBytes ==
01863                      peer.snapshotFile->getFileLength());
01864 
01865     // Execute RPC
01866     Protocol::Raft::AppendSnapshotChunk::Response response;
01867     TimePoint start = Clock::now();
01868     uint64_t epoch = currentEpoch;
01869     bool ok = peer.callRPC(Protocol::Raft::OpCode::APPEND_SNAPSHOT_CHUNK,
01870                            request, response,
01871                            lockGuard);
01872     if (!ok) {
01873         peer.backoffUntil = start +
01874             std::chrono::milliseconds(RPC_FAILURE_BACKOFF_MS);
01875         return;
01876     }
01877 
01878     // Process response
01879 
01880     if (currentTerm != request.term() || peer.exiting) {
01881         // we don't care about result of RPC
01882         return;
01883     }
01884     // Since we were leader in this term before, we must still be leader in
01885     // this term.
01886     assert(state == State::LEADER);
01887     if (response.term() > currentTerm) {
01888         stepDown(response.term());
01889     } else {
01890         assert(response.term() == currentTerm);
01891         peer.lastAckEpoch = epoch;
01892         stateChanged.notify_all();
01893         peer.nextHeartbeatTime = start +
01894             std::chrono::milliseconds(HEARTBEAT_PERIOD_MS);
01895         peer.snapshotFileOffset += numDataBytes;
01896         if (request.done()) {
01897             peer.lastAgreeIndex = peer.lastSnapshotIndex;
01898             peer.nextIndex = peer.lastSnapshotIndex + 1;
01899             // These entries are already committed if they're in a snapshot, so
01900             // the commitIndex shouldn't advance, but let's just follow the
01901             // simple rule that bumping lastAgreeIndex should always be
01902             // followed by a call to advanceCommittedId():
01903             advanceCommittedId();
01904             peer.snapshotFile.reset();
01905             peer.snapshotFileOffset = 0;
01906             peer.lastSnapshotIndex = 0;
01907         }
01908     }
01909 }

01911 void
01912 RaftConsensus::becomeLeader()
01913 {
01914     assert(state == State::CANDIDATE);
01915     NOTICE("Now leader for term %lu", currentTerm);
01916     state = State::LEADER;
01917     leaderId = serverId;
01918     startElectionAt = TimePoint::max();
01919     withholdVotesUntil = TimePoint::max();
01920 
01921     // The ordering is pretty important here: First set nextIndex and
01922     // lastAgreeIndex for ourselves and each follower, then append the no op.
01923     // Otherwise we'll set our localServer's last agree index too high.
01924     configuration->forEach(&Server::beginLeadership);
01925 
01926     // Append a new entry so that commitment is not delayed indefinitely.
01927     // Otherwise, if the leader never gets anything to append, it will never
01928     // return to read-only operations (it can't prove that its committed index
01929     // is up-to-date).
01930     Log::Entry entry;
01931     entry.set_term(currentTerm);
01932     entry.set_type(Protocol::Raft::EntryType::NOOP);
01933     append({&entry});
01934 
01935     // Outstanding RequestVote RPCs are no longer needed.
01936     interruptAll();
01937 }

01972 void
01973 RaftConsensus::readSnapshot()
01974 {
01975     std::unique_ptr<Storage::SnapshotFile::Reader> reader;
01976     if (storageDirectory.fd != -1) {
01977         try {
01978             reader.reset(new Storage::SnapshotFile::Reader(storageDirectory));
01979         } catch (const std::runtime_error& e) { // file not found
01980             NOTICE("%s", e.what());
01981         }
01982     }
01983     if (reader) {
01984         google::protobuf::io::CodedInputStream& stream = reader->getStream();
01985 
01986         // read header protobuf from stream
01987         bool ok = true;
01988         uint32_t numBytes = 0;
01989         ok = stream.ReadLittleEndian32(&numBytes);
01990         if (!ok)
01991             PANIC("couldn't read snapshot");
01992         SnapshotMetadata::Header header;
01993         auto limit = stream.PushLimit(numBytes);
01994         ok = header.MergePartialFromCodedStream(&stream);
01995         stream.PopLimit(limit);
01996         if (!ok)
01997             PANIC("couldn't read snapshot");
01998 
01999         // load header contents
02000         if (header.last_included_index() < lastSnapshotIndex) {
02001             PANIC("Trying to load a snapshot that is more stale than one this "
02002                   "server loaded earlier. The earlier snapshot covers through "
02003                   "log index %lu (inclusive); this one covers through log "
02004                   "index %lu (inclusive)",
02005                   lastSnapshotIndex,
02006                   header.last_included_index());
02007 
02008         }
02009         lastSnapshotIndex = header.last_included_index();

02226 RaftConsensus::upToDateLeader(std::unique_lock<Mutex>& lockGuard) const
02227 {
02228     ++currentEpoch;

02230     // schedule a heartbeat now so that this returns quickly
02231     configuration->forEach(&Server::scheduleHeartbeat);
02232     while (true) {
02233         if (exiting || state != State::LEADER)
02234             return false;

02236             // So we know we're the current leader, but do we have an
02237             // up-to-date commitIndex yet? What we'd like to check is whether
02238             // the entry's term at commitIndex matches our currentTerm, but
02239             // snapshots mean that we may not have the entry in our log. Since
02240             // commitIndex >= lastSnapshotIndex, we split into two cases:
02241             uint64_t commitTerm;
02242             if (commitIndex == lastSnapshotIndex) {
02243                 commitTerm = lastSnapshotTerm;
02244             } else {
02245                 assert(commitIndex > lastSnapshotIndex);
02246                 assert(commitIndex >= log->getLogStartIndex());
02247                 assert(commitIndex <= log->getLastLogIndex());
02248                 commitTerm = log->getEntry(commitIndex).term();
02249             }
02250             if (commitTerm == currentTerm)

02252         }

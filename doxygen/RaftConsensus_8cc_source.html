00529     }
00530 
00531     // Build up the list of new servers
00532     for (auto confIt = description.next_configuration().servers().begin();
00533          confIt != description.next_configuration().servers().end();
00534          ++confIt) {
00535         std::shared_ptr<Server> server = getServer(confIt->server_id());
00536         server->address = confIt->address();
00537         newServers.servers.push_back(server);
00538     }

00541     setGCFlag(*localServer);

00546         std::shared_ptr<Server> server = it->second;

00922 Consensus::Entry
00923 RaftConsensus::getNextEntry(uint64_t lastEntryId) const
00924 {
00925     std::unique_lock<Mutex> lockGuard(mutex);
00926     uint64_t nextEntryId = lastEntryId + 1;
00927     while (true) {
00928         if (exiting)
00929             throw ThreadInterruptedException();
00930         if (commitIndex >= nextEntryId) {
00931             Consensus::Entry entry;
00932 
00933             // Make the state machine load a snapshot if we don't have the next
00934             // entry it needs in the log.
00935             if (log->getLogStartIndex() > nextEntryId) {
00936                 entry.type = Consensus::Entry::SNAPSHOT;
00937                 // For well-behaved state machines, we expect 'snapshotReader'
00938                 // to contain a SnapshotFile::Reader that we can return
00939                 // directly to the state machine. In the case that a State
00940                 // Machine asks for the snapshot again, we have to build a new
00941                 // SnapshotFile::Reader again.
00942                 entry.snapshotReader = std::move(snapshotReader);
00943                 if (!entry.snapshotReader) {
00944                     WARNING("State machine asked for same snapshot twice; "
00945                             "this shouldn't happen in normal operation. "
00946                             "Having to re-read it from disk.");
00947                     // readSnapshot() shouldn't have any side effects since the
00948                     // snapshot should have already been read, so const_cast
00949                     // should be ok (though ugly).
00950                     const_cast<RaftConsensus*>(this)->readSnapshot();
00951                     entry.snapshotReader = std::move(snapshotReader);
00952                 }
00953                 entry.entryId = lastSnapshotIndex;
00954             } else {
00955                 // not a snapshot
00956                 const Log::Entry& logEntry = log->getEntry(nextEntryId);
00957                 entry.entryId = nextEntryId;
00958                 if (logEntry.type() == Protocol::Raft::EntryType::DATA) {
00959                     entry.type = Consensus::Entry::DATA;
00960                     entry.data = logEntry.data();
00961                 } else {
00962                     entry.type = Consensus::Entry::SKIP;
00963                 }
00964             }
00965             return entry;
00966         }
00967         stateChanged.wait(lockGuard);
00968     }

00986 void
00987 RaftConsensus::handleAppendEntries(
00988                     const Protocol::Raft::AppendEntries::Request& request,
00989                     Protocol::Raft::AppendEntries::Response& response)
00990 {
00991     std::unique_lock<Mutex> lockGuard(mutex);
00992     assert(!exiting);
00993 
00994     // Set response to a rejection. We'll overwrite these later if we end up
00995     // accepting the request.
00996     response.set_term(currentTerm);
00997     response.set_success(false);
00998 
00999     // If the caller's term is stale, just return our term to it.
01000     if (request.term() < currentTerm) {
01001         VERBOSE("Caller(%lu) is stale. Our term is %lu, theirs is %lu",
01002                  request.server_id(), currentTerm, request.term());
01003         return; // response was set to a rejection above
01004     }
01005     if (request.term() > currentTerm) {
01006         VERBOSE("Caller(%lu) has newer term, updating. "
01007                 "Ours was %lu, theirs is %lu",
01008                 request.server_id(), currentTerm, request.term());
01009         // We're about to bump our term in the stepDown below: update
01010         // 'response' accordingly.
01011         response.set_term(request.term());
01012     }
01013     // This request is a sign of life from the current leader. Update our term
01014     // and convert to follower if necessary; reset the election timer.
01015     stepDown(request.term());
01016     setElectionTimer();
01017     withholdVotesUntil = Clock::now() +
01018                          std::chrono::milliseconds(ELECTION_TIMEOUT_MS);
01019 
01020     // Record the leader ID as a hint for clients.
01021     if (leaderId == 0) {
01022         leaderId = request.server_id();
01023         NOTICE("All hail leader %lu for term %lu", leaderId, currentTerm);
01024     } else {
01025         assert(leaderId == request.server_id());
01026     }
01027 
01028     // For an entry to fit into our log, it must not leave a gap.
01029     if (request.prev_log_index() > log->getLastLogIndex()) {
01030         VERBOSE("Rejecting AppendEntries RPC: would leave gap");
01031         return; // response was set to a rejection above
01032     }
01033     // It must also agree with the previous entry in the log (and, inductively
01034     // all prior entries).
01035     // Always match on index 0, and always match on any discarded indexes:
01036     // since we know those were committed, the leader must agree with them.
01037     // We could truncate the log here, but there's no real advantage to doing
01038     // that.
01039     if (request.prev_log_index() >= log->getLogStartIndex() &&
01040         log->getEntry(request.prev_log_index()).term() !=
01041             request.prev_log_term()) {
01042         VERBOSE("Rejecting AppendEntries RPC: terms don't agree");
01043         return; // response was set to a rejection above
01044     }
01045 
01046     // If we got this far, we're accepting the request.
01047     response.set_success(true);
01048 
01049     // This needs to be able to handle duplicated RPC requests. We compare the
01050     // entries' terms to know if we need to do the operation; otherwise,
01051     // reapplying requests can result in data loss.
01052     //
01053     // The first problem this solves is that an old AppendEntries request may be
01054     // duplicated and received after a newer request, which could cause
01055     // undesirable data loss. For example, suppose the leader appends entry 4
01056     // and then entry 5, but the follower receives 4, then 5, then 4 again.
01057     // Without this extra guard, the follower would truncate 5 out of its
01058     // log.
01059     //
01060     // The second problem is more subtle: if the same request is duplicated but
01061     // the leader processes an earlier response, it will assume the
01062     // acknowledged data is safe. However, there is a window of vulnerability
01063     // on the follower's disk between the truncate and append operations (which
01064     // are not done atomically) when the follower processes the later request.
01065     uint64_t entryId = request.prev_log_index();
01066     for (auto it = request.entries().begin();
01067          it != request.entries().end();
01068          ++it) {
01069         ++entryId;
01070         const Protocol::Raft::Entry& entry = *it;
01071         if (entryId < log->getLogStartIndex()) {
01072             // We already snapshotted and discarded this index, so presumably
01073             // we've received a committed entry we once already had.
01074             continue;
01075         }
01076         if (log->getLastLogIndex() >= entryId) {
01077             if (log->getEntry(entryId).term() == entry.term())
01078                 continue;
01079             // should never truncate committed entries:
01080             assert(commitIndex < entryId);
01081             NOTICE("Truncating %lu entries after %lu from the log",
01082                    log->getLastLogIndex() - entryId + 1,
01083                    entryId - 1);
01084             log->truncateSuffix(entryId - 1);
01085             configurationManager->truncateSuffix(entryId - 1);
01086         }
01087 
01088         // Append this and all following entries.
01089         std::vector<const Protocol::Raft::Entry*> entries;
01090         while (it != request.entries().end()) {
01091             entries.push_back(&*it);
01092             ++it;
01093         }
01094         append(entries);
01095         break;
01096     }
01097 
01098     // Set our committed ID from the request's. In rare cases, this would make
01099     // our committed ID decrease. For example, this could happen with a new
01100     // leader who has not yet replicated one of its own entries. While that'd
01101     // be perfectly safe, guarding against it with an if statement lets us
01102     // make stronger assertions.
01103     if (commitIndex < request.commit_index()) {
01104         commitIndex = request.commit_index();
01105         assert(commitIndex <= log->getLastLogIndex());
01106         stateChanged.notify_all();
01107         VERBOSE("New commitIndex: %lu", commitIndex);
01108     }
01109 }

01192 RaftConsensus::handleRequestVote(
01193                     const Protocol::Raft::RequestVote::Request& request,
01194                     Protocol::Raft::RequestVote::Response& response)
01195 {
01196     std::unique_lock<Mutex> lockGuard(mutex);
01197     assert(!exiting);
01198 
01199     if (withholdVotesUntil > Clock::now()) {
01200         NOTICE("Rejecting RequestVote for term %lu from server %lu, since "

01204                request.server_id());

01206         response.set_granted(false);
01207         return;
01208     }
01209 
01210     if (request.term() > currentTerm) {

01218     // However, this is just an optimization that does not affect correctness
01219     // or really even efficiency, so it's not worth the trouble.
01220

01222     uint64_t lastLogIndex = log->getLastLogIndex();

01225                     (request.last_log_term() == lastLogTerm &&
01226                      request.last_log_index() >= lastLogIndex));
01227

01230         VERBOSE("Voting for %lu in term %lu",

01241     response.set_granted(request.term() == currentTerm &&

01256 RaftConsensus::ClientResult
01257 RaftConsensus::setConfiguration(
01258         uint64_t oldId,
01259         const Protocol::Raft::SimpleConfiguration& nextConfiguration)
01260 {
01261     std::unique_lock<Mutex> lockGuard(mutex);
01262 
01263     if (state != State::LEADER)
01264         return ClientResult::NOT_LEADER;
01265     if (configuration->id != oldId ||
01266         configuration->state != Configuration::State::STABLE) {
01267         // configurations has changed in the meantime
01268         return ClientResult::FAIL;
01269     }
01270 
01271     uint64_t term = currentTerm;
01272     configuration->setStagingServers(nextConfiguration);
01273     stateChanged.notify_all();
01274 
01275     // Wait for new servers to be caught up. This will abort if not every
01276     // server makes progress in a ELECTION_TIMEOUT_MS period.
01277     ++currentEpoch;
01278     uint64_t epoch = currentEpoch;
01279     TimePoint checkProgressAt =
01280         Clock::now() + std::chrono::milliseconds(ELECTION_TIMEOUT_MS);
01281     while (true) {
01282         if (exiting || term != currentTerm)
01283             return ClientResult::NOT_LEADER;
01284         if (configuration->stagingAll(&Server::isCaughtUp))
01285             break;
01286         if (Clock::now() >= checkProgressAt) {
01287             if (configuration->stagingMin(&Server::getLastAckEpoch) < epoch) {
01288                 configuration->resetStagingServers();
01289                 stateChanged.notify_all();
01290                 // TODO(ongaro): probably need to return a different type of
01291                 // message: confuses oldId mismatch from new server down
01292                 return ClientResult::FAIL;
01293             } else {
01294                 ++currentEpoch;
01295                 epoch = currentEpoch;
01296                 checkProgressAt =
01297                     (Clock::now() +
01298                      std::chrono::milliseconds(ELECTION_TIMEOUT_MS));
01299             }
01300         }
01301         stateChanged.wait_until(lockGuard, checkProgressAt);
01302     }
01303 
01304     // Write and commit transitional configuration
01305     Protocol::Raft::Configuration newConfiguration;
01306     *newConfiguration.mutable_prev_configuration() =
01307         configuration->description.prev_configuration();
01308     *newConfiguration.mutable_next_configuration() = nextConfiguration;
01309     Log::Entry entry;
01310     entry.set_type(Protocol::Raft::EntryType::CONFIGURATION);
01311     *entry.mutable_configuration() = newConfiguration;
01312     std::pair<ClientResult, uint64_t> result =
01313         replicateEntry(entry, lockGuard);
01314     if (result.first != ClientResult::SUCCESS)
01315         return result.first;
01316     uint64_t transitionalId = result.second;
01317 
01318     // Wait until the configuration that removes the old servers has been
01319     // committed. This is the first configuration with ID greater than
01320     // transitionalId.
01321     while (true) {
01322         // Check this first: if the new configuration excludes us so we've
01323         // stepped down upon committing it, we still want to return success.
01324         if (configuration->id > transitionalId &&
01325             commitIndex >= configuration->id) {
01326             return ClientResult::SUCCESS;
01327         }
01328         if (exiting || term != currentTerm)
01329             return ClientResult::NOT_LEADER;
01330         stateChanged.wait(lockGuard);
01331     }
01332 }

01334 std::unique_ptr<Storage::SnapshotFile::Writer>
01335 RaftConsensus::beginSnapshot(uint64_t lastIncludedIndex)
01336 {
01337     std::unique_lock<Mutex> lockGuard(mutex);
01338 
01339     NOTICE("Creating new snapshot through log index %lu (inclusive)",
01340            lastIncludedIndex);
01341     std::unique_ptr<Storage::SnapshotFile::Writer> writer(
01342                 new Storage::SnapshotFile::Writer(storageDirectory));
01343 
01344     // Only committed entries may be snapshotted.
01345     // (This check relies on commitIndex monotonically increasing.)
01346     if (lastIncludedIndex > commitIndex) {
01347         PANIC("Attempted to snapshot uncommitted entries (%lu requested but "
01348               "%lu is last committed entry)", lastIncludedIndex, commitIndex);
01349     }
01350 
01351     // set header fields
01352     SnapshotMetadata::Header header;
01353     header.set_last_included_index(lastIncludedIndex);
01354     // Set last_included_term:
01355     if (lastIncludedIndex >= log->getLogStartIndex() &&
01356         lastIncludedIndex <= log->getLastLogIndex()) {
01357         header.set_last_included_term(log->getEntry(lastIncludedIndex).term());
01358     } else if (lastIncludedIndex == 0) {
01359         WARNING("Taking a snapshot covering no log entries");
01360         header.set_last_included_term(0);
01361     } else if (lastIncludedIndex == lastSnapshotIndex) {
01362         WARNING("Taking a snapshot where we already have one, covering "
01363                 "entries 1 through %lu (inclusive)", lastIncludedIndex);
01364         header.set_last_included_term(lastSnapshotTerm);
01365     } else {
01366         WARNING("We've already discarded the entries that the state machine "
01367                 "wants to snapshot. This can happen in rare cases if the "
01368                 "leader already sent us a newer snapshot. We'll go ahead and "
01369                 "compute the snapshot, but it'll be discarded later in "
01370                 "snapshotDone(). Setting the last included term in the "
01371                 "snapshot header to 0 (a bogus value).");
01372         // If this turns out to be common, we should return NULL instead and
01373         // change the state machines to deal with that.
01374         header.set_last_included_term(0);
01375     }
01376     // Copy the configuration as of lastIncludedIndex to the header.
01377     std::pair<uint64_t, Protocol::Raft::Configuration> c =
01378         configurationManager->getLatestConfigurationAsOf(lastIncludedIndex);
01379     if (c.first == 0) {
01380         WARNING("Taking snapshot with no configuration. "
01381                 "This should have been the first thing in the log.");
01382     } else {
01383         header.set_configuration_index(c.first);
01384         *header.mutable_configuration() = c.second;
01385     }
01386 
01387     // write header to file
01388     google::protobuf::io::CodedOutputStream& stream = writer->getStream();
01389     int size = header.ByteSize();
01390     stream.WriteLittleEndian32(size);
01391     header.SerializeWithCachedSizes(&stream);
01392 
01393     return writer;

01396 void
01397 RaftConsensus::snapshotDone(
01398         uint64_t lastIncludedIndex,
01399         std::unique_ptr<Storage::SnapshotFile::Writer> writer)
01400 {
01401     std::unique_lock<Mutex> lockGuard(mutex);
01402     if (lastIncludedIndex <= lastSnapshotIndex) {
01403         NOTICE("Discarding snapshot through %lu since we already have one "
01404                "(presumably from another server) through %lu",
01405                lastIncludedIndex, lastSnapshotIndex);
01406         writer->discard();
01407         return;
01408     }
01409 
01410     // log->getEntry(lastIncludedIndex) is safe:
01411     // If the log prefix for this snapshot was truncated, that means we have a
01412     // newer snapshot (handled above).
01413     assert(lastIncludedIndex >= log->getLogStartIndex());
01414     // We never truncate committed entries from the end of our log, and
01415     // beginSnapshot() made sure that lastIncludedIndex covers only committed
01416     // entries.
01417     assert(lastIncludedIndex <= log->getLastLogIndex());
01418 
01419     lastSnapshotBytes = writer->save();
01420     lastSnapshotIndex = lastIncludedIndex;
01421     lastSnapshotTerm = log->getEntry(lastIncludedIndex).term();
01422 
01423     // It's easier to grab this configuration out of the manager again than to
01424     // carry it around after writing the header.
01425     std::pair<uint64_t, Protocol::Raft::Configuration> c =
01426         configurationManager->getLatestConfigurationAsOf(lastIncludedIndex);
01427     if (c.first == 0) {
01428         WARNING("Could not find the latest configuration as of index %lu "
01429                 "(inclusive). This shouldn't happen if the snapshot was "
01430                 "created with a configuration, as they should be.",
01431                 lastIncludedIndex);
01432     } else {
01433         configurationManager->setSnapshot(c.first, c.second);
01434     }
01435 
01436     NOTICE("Completed snapshot through log index %lu (inclusive)",
01437            lastSnapshotIndex);
01438 
01439     // It may be beneficial to defer discarding entries if some followers are
01440     // a little bit slow, to avoid having to send them a snapshot when a few
01441     // entries would do the trick. Best to avoid premature optimization though.
01442     discardUnneededEntries();
01443 }

01525 {
01526     std::unique_lock<Mutex> lockGuard(mutex);
01527     Core::ThreadId::setName(
01528         Core::StringUtil::format("Peer(%lu)", peer->serverId));
01529 
01530     // Each iteration of this loop issues a new RPC or sleeps on the condition
01531     // variable.

01535

01537             waitUntil = peer->backoffUntil;

01546                 case State::CANDIDATE:

01575 void
01576 RaftConsensus::stepDownThreadMain()
01577 {
01578     std::unique_lock<Mutex> lockGuard(mutex);
01579     Core::ThreadId::setName("stepDown");
01580     while (true) {
01581         // Wait until this server is the leader and is not the only server in
01582         // the cluster.
01583         while (true) {
01584             if (exiting)
01585                 return;
01586             if (state == State::LEADER) {
01587                 // If this local server forms a quorum (it is the only server
01588                 // in the configuration), we need to sleep. Without this guard,
01589                 // this method would not relinquish the CPU.
01590                 ++currentEpoch;
01591                 if (configuration->quorumMin(&Server::getLastAckEpoch) <
01592                     currentEpoch) {
01593                     break;
01594                 }
01595             }
01596             stateChanged.wait(lockGuard);
01597         }
01598         // Now, if an election timeout goes by without confirming leadership,
01599         // step down. The election timeout is a reasonable amount of time,
01600         // since it's about when other servers will start elections and bump
01601         // the term.
01602         TimePoint stepDownAt =
01603             Clock::now() + std::chrono::milliseconds(ELECTION_TIMEOUT_MS);
01604         uint64_t term = currentTerm;
01605         uint64_t epoch = currentEpoch; // currentEpoch was incremented above
01606         while (true) {
01607             if (exiting)
01608                 return;
01609             if (currentTerm > term)
01610                 break;
01611             if (configuration->quorumMin(&Server::getLastAckEpoch) >= epoch)
01612                 break;
01613             if (Clock::now() >= stepDownAt) {
01614                 NOTICE("No broadcast for a timeout, stepping down");
01615                 stepDown(currentTerm + 1);
01616                 break;
01617             }
01618             stateChanged.wait_until(lockGuard, stepDownAt);
01619         }
01620     }
01621 }

01625 void
01626 RaftConsensus::advanceCommittedId()
01627 {
01628     if (state != State::LEADER) {
01629         // getLastAgreeIndex is undefined unless we're leader
01630         WARNING("advanceCommittedId called as %s",
01631                 Core::StringUtil::toString(state).c_str());
01632         return;
01633     }
01634 
01635     // calculate the largest entry ID stored on a quorum of servers
01636     uint64_t newCommittedId =
01637         configuration->quorumMin(&Server::getLastAgreeIndex);
01638     if (commitIndex >= newCommittedId)
01639         return;
01640     // If we have discarded the entry, it's because we already knew it was
01641     // committed.
01642     assert(newCommittedId >= log->getLogStartIndex());
01643     // At least one of these entries must also be from the current term to
01644     // guarantee that no server without them can be elected.
01645     if (log->getEntry(newCommittedId).term() != currentTerm)
01646         return;
01647     commitIndex = newCommittedId;
01648     VERBOSE("New commitIndex: %lu", commitIndex);
01649     assert(commitIndex <= log->getLastLogIndex());
01650     stateChanged.notify_all();
01651 
01652     if (state == State::LEADER && commitIndex >= configuration->id) {
01653         // Upon committing a configuration that excludes itself, the leader
01654         // steps down.
01655         if (!configuration->hasVote(configuration->localServer)) {
01656             stepDown(currentTerm + 1);
01657             return;
01658         }
01659 
01660         // Upon committing a reconfiguration (Cold,new) entry, the leader
01661         // creates the next configuration (Cnew) entry.
01662         if (configuration->state == Configuration::State::TRANSITIONAL) {
01663             Log::Entry entry;
01664             entry.set_term(currentTerm);
01665             entry.set_type(Protocol::Raft::EntryType::CONFIGURATION);
01666             *entry.mutable_configuration()->mutable_prev_configuration() =
01667                 configuration->description.next_configuration();
01668             append({&entry});
01669             return;
01670         }
01671     }
01672 }

01697 void
01698 RaftConsensus::appendEntries(std::unique_lock<Mutex>& lockGuard,
01699                              Peer& peer)
01700 {
01701     uint64_t lastLogIndex = log->getLastLogIndex();
01702     uint64_t prevLogIndex = peer.nextIndex - 1;
01703     assert(prevLogIndex <= lastLogIndex);
01704 
01705     // Don't have needed entry: send a snapshot instead.
01706     if (peer.nextIndex < log->getLogStartIndex()) {
01707         appendSnapshotChunk(lockGuard, peer);
01708         return;
01709     }
01710 
01711     // Find prevLogTerm or fall back to sending a snapshot.
01712     uint64_t prevLogTerm;
01713     if (prevLogIndex >= log->getLogStartIndex()) {
01714         prevLogTerm = log->getEntry(prevLogIndex).term();
01715     } else if (prevLogIndex == 0) {
01716         prevLogTerm = 0;
01717     } else if (prevLogIndex == lastSnapshotIndex) {
01718         prevLogTerm = lastSnapshotTerm;
01719     } else {
01720         // Don't have needed entry for prevLogTerm: send snapshot instead.
01721         appendSnapshotChunk(lockGuard, peer);
01722         return;
01723     }
01724 
01725     // Build up request
01726     Protocol::Raft::AppendEntries::Request request;
01727     request.set_server_id(serverId);
01728     request.set_recipient_id(peer.serverId);
01729     request.set_term(currentTerm);
01730     request.set_prev_log_term(prevLogTerm);
01731     request.set_prev_log_index(prevLogIndex);
01732 
01733     // Add as many as entries as will fit comfortably in the request. It's
01734     // easiest to add one entry at a time until the RPC gets too big, then back
01735     // the last one out.
01736     uint64_t numEntries = 0;
01737     if (!peer.forceHeartbeat) {
01738         for (uint64_t entryId = peer.nextIndex;
01739              entryId <= lastLogIndex;
01740              ++entryId) {
01741             const Log::Entry& entry = log->getEntry(entryId);
01742             *request.add_entries() = entry;
01743             uint64_t requestSize =
01744                 Core::Util::downCast<uint64_t>(request.ByteSize());
01745             if (requestSize < SOFT_RPC_SIZE_LIMIT || numEntries == 0) {
01746                 // this entry fits, send it
01747                 VERBOSE("sending entry <id=%lu,term=%lu>",
01748                         entryId, entry.term());
01749                 ++numEntries;
01750             } else {
01751                 // this entry doesn't fit, discard it
01752                 request.mutable_entries()->RemoveLast();
01753             }
01754         }
01755     }
01756     request.set_commit_index(std::min(commitIndex, prevLogIndex + numEntries));
01757 
01758     // Execute RPC
01759     Protocol::Raft::AppendEntries::Response response;
01760     TimePoint start = Clock::now();
01761     uint64_t epoch = currentEpoch;
01762     bool ok = peer.callRPC(Protocol::Raft::OpCode::APPEND_ENTRIES,
01763                            request, response,
01764                            lockGuard);
01765     if (!ok) {
01766         peer.backoffUntil = start +
01767             std::chrono::milliseconds(RPC_FAILURE_BACKOFF_MS);
01768         return;
01769     }
01770 
01771     // Process response
01772 
01773     if (currentTerm != request.term() || peer.exiting) {
01774         // we don't care about result of RPC
01775         return;
01776     }
01777     // Since we were leader in this term before, we must still be leader in
01778     // this term.
01779     assert(state == State::LEADER);
01780     if (response.term() > currentTerm) {
01781         stepDown(response.term());
01782     } else {
01783         assert(response.term() == currentTerm);
01784         peer.lastAckEpoch = epoch;
01785         stateChanged.notify_all();
01786         peer.nextHeartbeatTime = start +
01787             std::chrono::milliseconds(HEARTBEAT_PERIOD_MS);
01788         if (response.success()) {
01789             if (peer.lastAgreeIndex > prevLogIndex + numEntries) {
01790                 // Revisit this warning if we pipeline AppendEntries RPCs for
01791                 // performance.
01792                 WARNING("lastAgreeIndex should monotonically increase within a "
01793                         "term, since servers don't forget entries. But it "
01794                         "didn't.");
01795             } else {
01796                 peer.lastAgreeIndex = prevLogIndex + numEntries;
01797                 advanceCommittedId();
01798             }
01799             peer.nextIndex = peer.lastAgreeIndex + 1;
01800             peer.forceHeartbeat = false;
01801 
01802             if (!peer.isCaughtUp_ &&
01803                 peer.thisCatchUpIterationGoalId <= peer.lastAgreeIndex) {
01804                 Clock::duration duration =
01805                     Clock::now() - peer.thisCatchUpIterationStart;
01806                 uint64_t thisCatchUpIterationMs =
01807                     uint64_t(std::chrono::duration_cast<
01808                                  std::chrono::milliseconds>(duration).count());
01809                 if (uint64_t(labs(int64_t(peer.lastCatchUpIterationMs -
01810                                           thisCatchUpIterationMs))) <
01811                     ELECTION_TIMEOUT_MS) {
01812                     peer.isCaughtUp_ = true;
01813                     stateChanged.notify_all();
01814                 } else {
01815                     peer.lastCatchUpIterationMs = thisCatchUpIterationMs;
01816                     peer.thisCatchUpIterationStart = Clock::now();
01817                     peer.thisCatchUpIterationGoalId = log->getLastLogIndex();
01818                 }
01819             }
01820         } else {
01821             if (peer.nextIndex > 1)
01822                 --peer.nextIndex;
01823         }
01824     }
01825 }

01827 void
01828 RaftConsensus::appendSnapshotChunk(std::unique_lock<Mutex>& lockGuard,
01829                                    Peer& peer)
01830 {
01831     // Build up request
01832     Protocol::Raft::AppendSnapshotChunk::Request request;
01833     request.set_server_id(serverId);
01834     request.set_recipient_id(peer.serverId);
01835     request.set_term(currentTerm);
01836 
01837     // Open the latest snapshot if we haven't already. Stash a copy of the
01838     // lastSnapshotIndex that goes along with the file, since it's possible
01839     // that this will change while we're transferring chunks).
01840     if (!peer.snapshotFile) {
01841         namespace FilesystemUtil = Storage::FilesystemUtil;
01842         peer.snapshotFile.reset(new FilesystemUtil::FileContents(
01843             FilesystemUtil::openFile(storageDirectory, "snapshot", O_RDONLY)));
01844         peer.snapshotFileOffset = 0;
01845         peer.lastSnapshotIndex = lastSnapshotIndex;
01846     }
01847     request.set_last_snapshot_index(peer.lastSnapshotIndex);
01848     request.set_byte_offset(peer.snapshotFileOffset);
01849     // The amount of data we can send is bounded by the remaining bytes in the
01850     // file and the maximum length for RPCs.
01851     uint64_t numDataBytes = std::min(
01852         peer.snapshotFile->getFileLength() - peer.snapshotFileOffset,
01853         SOFT_RPC_SIZE_LIMIT);
01854     request.set_data(peer.snapshotFile->get<char>(peer.snapshotFileOffset,
01855                                                   numDataBytes),
01856                      numDataBytes);
01857     request.set_done(peer.snapshotFileOffset + numDataBytes ==
01858                      peer.snapshotFile->getFileLength());
01859 
01860     // Execute RPC
01861     Protocol::Raft::AppendSnapshotChunk::Response response;
01862     TimePoint start = Clock::now();
01863     uint64_t epoch = currentEpoch;
01864     bool ok = peer.callRPC(Protocol::Raft::OpCode::APPEND_SNAPSHOT_CHUNK,
01865                            request, response,
01866                            lockGuard);
01867     if (!ok) {
01868         peer.backoffUntil = start +
01869             std::chrono::milliseconds(RPC_FAILURE_BACKOFF_MS);
01870         return;
01871     }
01872 
01873     // Process response
01874 
01875     if (currentTerm != request.term() || peer.exiting) {
01876         // we don't care about result of RPC
01877         return;
01878     }
01879     // Since we were leader in this term before, we must still be leader in
01880     // this term.
01881     assert(state == State::LEADER);
01882     if (response.term() > currentTerm) {
01883         stepDown(response.term());
01884     } else {
01885         assert(response.term() == currentTerm);
01886         peer.lastAckEpoch = epoch;
01887         stateChanged.notify_all();
01888         peer.nextHeartbeatTime = start +
01889             std::chrono::milliseconds(HEARTBEAT_PERIOD_MS);
01890         peer.snapshotFileOffset += numDataBytes;
01891         if (request.done()) {
01892             peer.lastAgreeIndex = peer.lastSnapshotIndex;
01893             peer.nextIndex = peer.lastSnapshotIndex + 1;
01894             // These entries are already committed if they're in a snapshot, so
01895             // the commitIndex shouldn't advance, but let's just follow the
01896             // simple rule that bumping lastAgreeIndex should always be
01897             // followed by a call to advanceCommittedId():
01898             advanceCommittedId();
01899             peer.snapshotFile.reset();
01900             peer.snapshotFileOffset = 0;
01901             peer.lastSnapshotIndex = 0;
01902         }
01903     }
01904 }

01906 void
01907 RaftConsensus::becomeLeader()
01908 {
01909     assert(state == State::CANDIDATE);
01910     NOTICE("Now leader for term %lu", currentTerm);
01911     state = State::LEADER;
01912     leaderId = serverId;
01913     startElectionAt = TimePoint::max();
01914     withholdVotesUntil = TimePoint::max();
01915 
01916     // The ordering is pretty important here: First set nextIndex and
01917     // lastAgreeIndex for ourselves and each follower, then append the no op.
01918     // Otherwise we'll set our localServer's last agree index too high.
01919     configuration->forEach(&Server::beginLeadership);
01920 
01921     // Append a new entry so that commitment is not delayed indefinitely.
01922     // Otherwise, if the leader never gets anything to append, it will never
01923     // return to read-only operations (it can't prove that its committed index
01924     // is up-to-date).
01925     Log::Entry entry;
01926     entry.set_term(currentTerm);
01927     entry.set_type(Protocol::Raft::EntryType::NOOP);
01928     append({&entry});
01929 
01930     // Outstanding RequestVote RPCs are no longer needed.
01931     interruptAll();
01932 }

01967 void
01968 RaftConsensus::readSnapshot()
01969 {
01970     std::unique_ptr<Storage::SnapshotFile::Reader> reader;
01971     if (storageDirectory.fd != -1) {
01972         try {
01973             reader.reset(new Storage::SnapshotFile::Reader(storageDirectory));
01974         } catch (const std::runtime_error& e) { // file not found
01975             NOTICE("%s", e.what());
01976         }
01977     }
01978     if (reader) {
01979         google::protobuf::io::CodedInputStream& stream = reader->getStream();
01980 
01981         // read header protobuf from stream
01982         bool ok = true;
01983         uint32_t numBytes = 0;
01984         ok = stream.ReadLittleEndian32(&numBytes);
01985         if (!ok)
01986             PANIC("couldn't read snapshot");
01987         SnapshotMetadata::Header header;
01988         auto limit = stream.PushLimit(numBytes);
01989         ok = header.MergePartialFromCodedStream(&stream);
01990         stream.PopLimit(limit);
01991         if (!ok)
01992             PANIC("couldn't read snapshot");
01993 
01994         // load header contents
01995         if (header.last_included_index() < lastSnapshotIndex) {
01996             PANIC("Trying to load a snapshot that is more stale than one this "
01997                   "server loaded earlier. The earlier snapshot covers through "
01998                   "log index %lu (inclusive); this one covers through log "
01999                   "index %lu (inclusive)",
02000                   lastSnapshotIndex,
02001                   header.last_included_index());
02002 
02003         }
02004         lastSnapshotIndex = header.last_included_index();
02005         lastSnapshotTerm = header.last_included_term();
02006         lastSnapshotBytes = reader->getSizeBytes();

02223     ++currentEpoch;

02234             // snapshots mean that we may not have the entry in our log. Since
02235             // commitIndex >= lastSnapshotIndex, we split into two cases:

02238                 commitTerm = lastSnapshotTerm;

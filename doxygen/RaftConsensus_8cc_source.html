00996 RaftConsensus::Entry
00997 RaftConsensus::getNextEntry(uint64_t lastEntryId) const
00998 {
00999     std::unique_lock<Mutex> lockGuard(mutex);
01000     uint64_t nextEntryId = lastEntryId + 1;
01001     while (true) {
01002         if (exiting)
01003             throw Core::Util::ThreadInterruptedException();
01004         if (commitIndex >= nextEntryId) {
01005             RaftConsensus::Entry entry;
01006 
01007             // Make the state machine load a snapshot if we don't have the next
01008             // entry it needs in the log.
01009             if (log->getLogStartIndex() > nextEntryId) {
01010                 entry.type = Entry::SNAPSHOT;
01011                 // For well-behaved state machines, we expect 'snapshotReader'
01012                 // to contain a SnapshotFile::Reader that we can return
01013                 // directly to the state machine. In the case that a State
01014                 // Machine asks for the snapshot again, we have to build a new
01015                 // SnapshotFile::Reader again.
01016                 entry.snapshotReader = std::move(snapshotReader);
01017                 if (!entry.snapshotReader) {
01018                     WARNING("State machine asked for same snapshot twice; "
01019                             "this shouldn't happen in normal operation. "
01020                             "Having to re-read it from disk.");
01021                     // readSnapshot() shouldn't have any side effects since the
01022                     // snapshot should have already been read, so const_cast
01023                     // should be ok (though ugly).
01024                     const_cast<RaftConsensus*>(this)->readSnapshot();
01025                     entry.snapshotReader = std::move(snapshotReader);
01026                 }
01027                 entry.entryId = lastSnapshotIndex;
01028             } else {
01029                 // not a snapshot
01030                 const Log::Entry& logEntry = log->getEntry(nextEntryId);
01031                 entry.entryId = nextEntryId;
01032                 if (logEntry.type() == Protocol::Raft::EntryType::DATA) {
01033                     entry.type = Entry::DATA;
01034                     entry.data = logEntry.data();
01035                 } else {
01036                     entry.type = Entry::SKIP;
01037                 }
01038             }
01039             return entry;
01040         }
01041         stateChanged.wait(lockGuard);
01042     }
01043 }

01060 void
01061 RaftConsensus::handleAppendEntries(
01062                     const Protocol::Raft::AppendEntries::Request& request,
01063                     Protocol::Raft::AppendEntries::Response& response)
01064 {
01065     std::unique_lock<Mutex> lockGuard(mutex);
01066     assert(!exiting);
01067 
01068     // Set response to a rejection. We'll overwrite these later if we end up
01069     // accepting the request.
01070     response.set_term(currentTerm);
01071     response.set_success(false);
01072 
01073     // If the caller's term is stale, just return our term to it.
01074     if (request.term() < currentTerm) {
01075         VERBOSE("Caller(%lu) is stale. Our term is %lu, theirs is %lu",
01076                  request.server_id(), currentTerm, request.term());
01077         return; // response was set to a rejection above
01078     }
01079     if (request.term() > currentTerm) {
01080         NOTICE("Received AppendEntries request from server %lu in term %lu "
01081                "(this server's term was %lu)",
01082                 request.server_id(), request.term(), currentTerm);
01083         // We're about to bump our term in the stepDown below: update
01084         // 'response' accordingly.
01085         response.set_term(request.term());
01086     }
01087     // This request is a sign of life from the current leader. Update our term
01088     // and convert to follower if necessary; reset the election timer.
01089     stepDown(request.term());
01090     setElectionTimer();
01091     withholdVotesUntil = Clock::now() +
01092                          std::chrono::milliseconds(ELECTION_TIMEOUT_MS);
01093 
01094     // Record the leader ID as a hint for clients.
01095     if (leaderId == 0) {
01096         leaderId = request.server_id();
01097         NOTICE("All hail leader %lu for term %lu", leaderId, currentTerm);
01098     } else {
01099         assert(leaderId == request.server_id());
01100     }
01101 
01102     // For an entry to fit into our log, it must not leave a gap.
01103     if (request.prev_log_index() > log->getLastLogIndex()) {
01104         VERBOSE("Rejecting AppendEntries RPC: would leave gap");
01105         return; // response was set to a rejection above
01106     }
01107     // It must also agree with the previous entry in the log (and, inductively
01108     // all prior entries).
01109     // Always match on index 0, and always match on any discarded indexes:
01110     // since we know those were committed, the leader must agree with them.
01111     // We could truncate the log here, but there's no real advantage to doing
01112     // that.
01113     if (request.prev_log_index() >= log->getLogStartIndex() &&
01114         log->getEntry(request.prev_log_index()).term() !=
01115             request.prev_log_term()) {
01116         VERBOSE("Rejecting AppendEntries RPC: terms don't agree");
01117         return; // response was set to a rejection above
01118     }
01119 
01120     // If we got this far, we're accepting the request.
01121     response.set_success(true);
01122 
01123     // This needs to be able to handle duplicated RPC requests. We compare the
01124     // entries' terms to know if we need to do the operation; otherwise,
01125     // reapplying requests can result in data loss.
01126     //
01127     // The first problem this solves is that an old AppendEntries request may be
01128     // duplicated and received after a newer request, which could cause
01129     // undesirable data loss. For example, suppose the leader appends entry 4
01130     // and then entry 5, but the follower receives 4, then 5, then 4 again.
01131     // Without this extra guard, the follower would truncate 5 out of its
01132     // log.
01133     //
01134     // The second problem is more subtle: if the same request is duplicated but
01135     // the leader processes an earlier response, it will assume the
01136     // acknowledged data is safe. However, there is a window of vulnerability
01137     // on the follower's disk between the truncate and append operations (which
01138     // are not done atomically) when the follower processes the later request.
01139     uint64_t entryId = request.prev_log_index();
01140     for (auto it = request.entries().begin();
01141          it != request.entries().end();
01142          ++it) {
01143         ++entryId;
01144         const Protocol::Raft::Entry& entry = *it;
01145         if (entryId < log->getLogStartIndex()) {
01146             // We already snapshotted and discarded this index, so presumably
01147             // we've received a committed entry we once already had.
01148             continue;
01149         }
01150         if (log->getLastLogIndex() >= entryId) {
01151             if (log->getEntry(entryId).term() == entry.term())
01152                 continue;
01153             // should never truncate committed entries:
01154             assert(commitIndex < entryId);
01155             NOTICE("Truncating %lu entries after %lu from the log",
01156                    log->getLastLogIndex() - entryId + 1,
01157                    entryId - 1);
01158             log->truncateSuffix(entryId - 1);
01159             configurationManager->truncateSuffix(entryId - 1);
01160         }
01161 
01162         // Append this and all following entries.
01163         std::vector<const Protocol::Raft::Entry*> entries;
01164         while (it != request.entries().end()) {
01165             entries.push_back(&*it);
01166             ++it;
01167         }
01168         append(entries);
01169         break;
01170     }
01171 
01172     // Set our committed ID from the request's. In rare cases, this would make
01173     // our committed ID decrease. For example, this could happen with a new
01174     // leader who has not yet replicated one of its own entries. While that'd
01175     // be perfectly safe, guarding against it with an if statement lets us
01176     // make stronger assertions.
01177     if (commitIndex < request.commit_index()) {
01178         commitIndex = request.commit_index();
01179         assert(commitIndex <= log->getLastLogIndex());
01180         stateChanged.notify_all();
01181         VERBOSE("New commitIndex: %lu", commitIndex);
01182     }
01183 }

01266 RaftConsensus::handleRequestVote(
01267                     const Protocol::Raft::RequestVote::Request& request,
01268                     Protocol::Raft::RequestVote::Response& response)
01269 {
01270     std::unique_lock<Mutex> lockGuard(mutex);
01271     assert(!exiting);
01272 
01273     if (withholdVotesUntil > Clock::now()) {
01274         NOTICE("Rejecting RequestVote for term %lu from server %lu, since "
01275                "this server (which is in term %lu) recently heard from a "
01276                "leader (%lu). Should server %lu be shut down?",
01277                request.term(), request.server_id(), currentTerm,
01278                leaderId, request.server_id());
01279         response.set_term(currentTerm);
01280         response.set_granted(false);
01281         return;
01282     }
01283 
01284     if (request.term() > currentTerm) {
01285         NOTICE("Received RequestVote request from server %lu in term %lu "
01286                "(this server's term was %lu)",
01287                 request.server_id(), request.term(), currentTerm);
01288         stepDown(request.term());

01291     // At this point, if leaderId != 0, we could tell the caller to step down.
01292     // However, this is just an optimization that does not affect correctness
01293     // or really even efficiency, so it's not worth the trouble.
01294 
01295     // If the caller has a less complete log, we can't give it our vote.
01296     uint64_t lastLogIndex = log->getLastLogIndex();
01297     uint64_t lastLogTerm = getLastLogTerm();

01299                     (request.last_log_term() == lastLogTerm &&
01300                      request.last_log_index() >= lastLogIndex));
01301 
01302     if (request.term() == currentTerm) {
01303         if (logIsOk && votedFor == 0) {
01304             // Give caller our vote
01305             NOTICE("Voting for %lu in term %lu",
01306                    request.server_id(), currentTerm);
01307             stepDown(currentTerm);
01308             setElectionTimer();
01309             votedFor = request.server_id();
01310             updateLogMetadata();
01311         }
01312     }
01313 
01314     // Fill in response.
01315     response.set_term(currentTerm);
01316     // don't strictly need the first condition
01317     response.set_granted(request.term() == currentTerm &&
01318                          votedFor == request.server_id());

01332 RaftConsensus::ClientResult
01333 RaftConsensus::setConfiguration(
01334         uint64_t oldId,
01335         const Protocol::Raft::SimpleConfiguration& nextConfiguration)
01336 {
01337     std::unique_lock<Mutex> lockGuard(mutex);
01338 
01339     if (state != State::LEADER)
01340         return ClientResult::NOT_LEADER;
01341     if (configuration->id != oldId ||
01342         configuration->state != Configuration::State::STABLE) {
01343         // configurations has changed in the meantime
01344         return ClientResult::FAIL;
01345     }
01346 
01347     uint64_t term = currentTerm;
01348     configuration->setStagingServers(nextConfiguration);
01349     stateChanged.notify_all();
01350 
01351     // Wait for new servers to be caught up. This will abort if not every
01352     // server makes progress in a ELECTION_TIMEOUT_MS period.
01353     ++currentEpoch;
01354     uint64_t epoch = currentEpoch;
01355     TimePoint checkProgressAt =
01356         Clock::now() + std::chrono::milliseconds(ELECTION_TIMEOUT_MS);
01357     while (true) {
01358         if (exiting || term != currentTerm)
01359             return ClientResult::NOT_LEADER;
01360         if (configuration->stagingAll(&Server::isCaughtUp))
01361             break;
01362         if (Clock::now() >= checkProgressAt) {
01363             if (configuration->stagingMin(&Server::getLastAckEpoch) < epoch) {
01364                 configuration->resetStagingServers();
01365                 stateChanged.notify_all();
01366                 // TODO(ongaro): probably need to return a different type of
01367                 // message: confuses oldId mismatch from new server down
01368                 return ClientResult::FAIL;
01369             } else {
01370                 ++currentEpoch;
01371                 epoch = currentEpoch;
01372                 checkProgressAt =
01373                     (Clock::now() +
01374                      std::chrono::milliseconds(ELECTION_TIMEOUT_MS));
01375             }
01376         }
01377         stateChanged.wait_until(lockGuard, checkProgressAt);
01378     }
01379 
01380     // Write and commit transitional configuration
01381     Protocol::Raft::Configuration newConfiguration;
01382     *newConfiguration.mutable_prev_configuration() =
01383         configuration->description.prev_configuration();
01384     *newConfiguration.mutable_next_configuration() = nextConfiguration;
01385     Log::Entry entry;
01386     entry.set_type(Protocol::Raft::EntryType::CONFIGURATION);
01387     *entry.mutable_configuration() = newConfiguration;
01388     std::pair<ClientResult, uint64_t> result =
01389         replicateEntry(entry, lockGuard);
01390     if (result.first != ClientResult::SUCCESS)
01391         return result.first;
01392     uint64_t transitionalId = result.second;
01393 
01394     // Wait until the configuration that removes the old servers has been
01395     // committed. This is the first configuration with ID greater than
01396     // transitionalId.
01397     while (true) {
01398         // Check this first: if the new configuration excludes us so we've
01399         // stepped down upon committing it, we still want to return success.
01400         if (configuration->id > transitionalId &&
01401             commitIndex >= configuration->id) {
01402             return ClientResult::SUCCESS;
01403         }
01404         if (exiting || term != currentTerm)
01405             return ClientResult::NOT_LEADER;
01406         stateChanged.wait(lockGuard);
01407     }
01408 }

01410 std::unique_ptr<Storage::SnapshotFile::Writer>
01411 RaftConsensus::beginSnapshot(uint64_t lastIncludedIndex)
01412 {
01413     std::unique_lock<Mutex> lockGuard(mutex);
01414 
01415     NOTICE("Creating new snapshot through log index %lu (inclusive)",
01416            lastIncludedIndex);
01417     std::unique_ptr<Storage::SnapshotFile::Writer> writer(
01418                 new Storage::SnapshotFile::Writer(storageDirectory));
01419 
01420     // Only committed entries may be snapshotted.
01421     // (This check relies on commitIndex monotonically increasing.)
01422     if (lastIncludedIndex > commitIndex) {
01423         PANIC("Attempted to snapshot uncommitted entries (%lu requested but "
01424               "%lu is last committed entry)", lastIncludedIndex, commitIndex);
01425     }
01426 
01427     // set header fields
01428     SnapshotMetadata::Header header;
01429     header.set_last_included_index(lastIncludedIndex);
01430     // Set last_included_term:
01431     if (lastIncludedIndex >= log->getLogStartIndex() &&
01432         lastIncludedIndex <= log->getLastLogIndex()) {
01433         header.set_last_included_term(log->getEntry(lastIncludedIndex).term());
01434     } else if (lastIncludedIndex == 0) {
01435         WARNING("Taking a snapshot covering no log entries");
01436         header.set_last_included_term(0);
01437     } else if (lastIncludedIndex == lastSnapshotIndex) {
01438         WARNING("Taking a snapshot where we already have one, covering "
01439                 "entries 1 through %lu (inclusive)", lastIncludedIndex);
01440         header.set_last_included_term(lastSnapshotTerm);
01441     } else {
01442         WARNING("We've already discarded the entries that the state machine "
01443                 "wants to snapshot. This can happen in rare cases if the "
01444                 "leader already sent us a newer snapshot. We'll go ahead and "
01445                 "compute the snapshot, but it'll be discarded later in "
01446                 "snapshotDone(). Setting the last included term in the "
01447                 "snapshot header to 0 (a bogus value).");
01448         // If this turns out to be common, we should return NULL instead and
01449         // change the state machines to deal with that.
01450         header.set_last_included_term(0);
01451     }
01452     // Copy the configuration as of lastIncludedIndex to the header.
01453     std::pair<uint64_t, Protocol::Raft::Configuration> c =
01454         configurationManager->getLatestConfigurationAsOf(lastIncludedIndex);
01455     if (c.first == 0) {
01456         WARNING("Taking snapshot with no configuration. "
01457                 "This should have been the first thing in the log.");
01458     } else {
01459         header.set_configuration_index(c.first);
01460         *header.mutable_configuration() = c.second;
01461     }
01462 
01463     // write header to file
01464     google::protobuf::io::CodedOutputStream& stream = writer->getStream();
01465     int size = header.ByteSize();
01466     stream.WriteLittleEndian32(size);
01467     header.SerializeWithCachedSizes(&stream);
01468 
01469     return writer;
01470 }

01472 void
01473 RaftConsensus::snapshotDone(
01474         uint64_t lastIncludedIndex,
01475         std::unique_ptr<Storage::SnapshotFile::Writer> writer)
01476 {
01477     std::unique_lock<Mutex> lockGuard(mutex);
01478     if (lastIncludedIndex <= lastSnapshotIndex) {
01479         NOTICE("Discarding snapshot through %lu since we already have one "
01480                "(presumably from another server) through %lu",
01481                lastIncludedIndex, lastSnapshotIndex);
01482         writer->discard();
01483         return;
01484     }
01485 
01486     // log->getEntry(lastIncludedIndex) is safe:
01487     // If the log prefix for this snapshot was truncated, that means we have a
01488     // newer snapshot (handled above).
01489     assert(lastIncludedIndex >= log->getLogStartIndex());
01490     // We never truncate committed entries from the end of our log, and
01491     // beginSnapshot() made sure that lastIncludedIndex covers only committed
01492     // entries.
01493     assert(lastIncludedIndex <= log->getLastLogIndex());
01494 
01495     lastSnapshotBytes = writer->save();
01496     lastSnapshotIndex = lastIncludedIndex;
01497     lastSnapshotTerm = log->getEntry(lastIncludedIndex).term();
01498 
01499     // It's easier to grab this configuration out of the manager again than to
01500     // carry it around after writing the header.
01501     std::pair<uint64_t, Protocol::Raft::Configuration> c =
01502         configurationManager->getLatestConfigurationAsOf(lastIncludedIndex);
01503     if (c.first == 0) {
01504         WARNING("Could not find the latest configuration as of index %lu "
01505                 "(inclusive). This shouldn't happen if the snapshot was "
01506                 "created with a configuration, as they should be.",
01507                 lastIncludedIndex);
01508     } else {
01509         configurationManager->setSnapshot(c.first, c.second);
01510     }
01511 
01512     NOTICE("Completed snapshot through log index %lu (inclusive)",
01513            lastSnapshotIndex);
01514 
01515     // It may be beneficial to defer discarding entries if some followers are
01516     // a little bit slow, to avoid having to send them a snapshot when a few
01517     // entries would do the trick. Best to avoid premature optimization though.
01518     discardUnneededEntries();
01519 }

01634 void
01635 RaftConsensus::peerThreadMain(std::shared_ptr<Peer> peer)
01636 {
01637     std::unique_lock<Mutex> lockGuard(mutex);
01638     Core::ThreadId::setName(
01639         Core::StringUtil::format("Peer(%lu)", peer->serverId));
01640 
01641     // Each iteration of this loop issues a new RPC or sleeps on the condition
01642     // variable.
01643     while (!peer->exiting) {
01644         TimePoint now = Clock::now();
01645         TimePoint waitUntil = TimePoint::min();
01646 
01647         if (peer->backoffUntil > now) {
01648             waitUntil = peer->backoffUntil;
01649         } else {
01650             switch (state) {
01651                 // Followers don't issue RPCs.
01652                 case State::FOLLOWER:
01653                     waitUntil = TimePoint::max();
01654                     break;
01655 
01656                 // Candidates request votes.
01657                 case State::CANDIDATE:
01658                     if (!peer->requestVoteDone)
01659                         requestVote(lockGuard, *peer);
01660                     else
01661                         waitUntil = TimePoint::max();
01662                     break;
01663 
01664                 // Leaders replicate entries and periodically send heartbeats.
01665                 case State::LEADER:
01666                     if (peer->getLastAgreeIndex() < log->getLastLogIndex() ||
01667                         peer->nextHeartbeatTime < now) {
01668                         // appendEntries delegates to appendSnapshotChunk if we
01669                         // need to send a snapshot instead
01670                         appendEntries(lockGuard, *peer);
01671                     } else {
01672                         waitUntil = peer->nextHeartbeatTime;
01673                     }
01674                     break;
01675             }
01676         }
01677 
01678         stateChanged.wait_until(lockGuard, waitUntil);
01679     }
01680 
01681     // must return immediately after this
01682     --numPeerThreads;
01683     stateChanged.notify_all();
01684 }

01686 void
01687 RaftConsensus::stepDownThreadMain()
01688 {
01689     std::unique_lock<Mutex> lockGuard(mutex);
01690     Core::ThreadId::setName("stepDown");
01691     while (true) {
01692         // Wait until this server is the leader and is not the only server in
01693         // the cluster.
01694         while (true) {
01695             if (exiting)
01696                 return;
01697             if (state == State::LEADER) {
01698                 // If this local server forms a quorum (it is the only server
01699                 // in the configuration), we need to sleep. Without this guard,
01700                 // this method would not relinquish the CPU.
01701                 ++currentEpoch;
01702                 if (configuration->quorumMin(&Server::getLastAckEpoch) <
01703                     currentEpoch) {
01704                     break;
01705                 }
01706             }
01707             stateChanged.wait(lockGuard);
01708         }
01709         // Now, if an election timeout goes by without confirming leadership,
01710         // step down. The election timeout is a reasonable amount of time,
01711         // since it's about when other servers will start elections and bump
01712         // the term.
01713         TimePoint stepDownAt =
01714             Clock::now() + std::chrono::milliseconds(ELECTION_TIMEOUT_MS);
01715         uint64_t term = currentTerm;
01716         uint64_t epoch = currentEpoch; // currentEpoch was incremented above
01717         while (true) {
01718             if (exiting)
01719                 return;
01720             if (currentTerm > term)
01721                 break;
01722             if (configuration->quorumMin(&Server::getLastAckEpoch) >= epoch)
01723                 break;
01724             if (Clock::now() >= stepDownAt) {
01725                 NOTICE("No broadcast for a timeout, stepping down from leader "
01726                        "of term %lu (converting to follower in term %lu)",
01727                        currentTerm, currentTerm + 1);
01728                 stepDown(currentTerm + 1);
01729                 break;
01730             }
01731             stateChanged.wait_until(lockGuard, stepDownAt);
01732         }
01733     }
01734 }

01738 void
01739 RaftConsensus::advanceCommittedId()
01740 {
01741     if (state != State::LEADER) {
01742         // getLastAgreeIndex is undefined unless we're leader
01743         WARNING("advanceCommittedId called as %s",
01744                 Core::StringUtil::toString(state).c_str());
01745         return;
01746     }
01747 
01748     // calculate the largest entry ID stored on a quorum of servers
01749     uint64_t newCommittedId =
01750         configuration->quorumMin(&Server::getLastAgreeIndex);
01751     if (commitIndex >= newCommittedId)
01752         return;
01753     // If we have discarded the entry, it's because we already knew it was
01754     // committed.
01755     assert(newCommittedId >= log->getLogStartIndex());
01756     // At least one of these entries must also be from the current term to
01757     // guarantee that no server without them can be elected.
01758     if (log->getEntry(newCommittedId).term() != currentTerm)
01759         return;
01760     commitIndex = newCommittedId;
01761     VERBOSE("New commitIndex: %lu", commitIndex);
01762     assert(commitIndex <= log->getLastLogIndex());
01763     stateChanged.notify_all();
01764 
01765     if (state == State::LEADER && commitIndex >= configuration->id) {
01766         // Upon committing a configuration that excludes itself, the leader
01767         // steps down.
01768         if (!configuration->hasVote(configuration->localServer)) {
01769             NOTICE("Newly committed configuration does not include self. "
01770                    "Stepping down as leader");
01771             stepDown(currentTerm + 1);
01772             return;
01773         }
01774 
01775         // Upon committing a reconfiguration (Cold,new) entry, the leader
01776         // creates the next configuration (Cnew) entry.
01777         if (configuration->state == Configuration::State::TRANSITIONAL) {
01778             Log::Entry entry;
01779             entry.set_term(currentTerm);
01780             entry.set_type(Protocol::Raft::EntryType::CONFIGURATION);
01781             *entry.mutable_configuration()->mutable_prev_configuration() =
01782                 configuration->description.next_configuration();
01783             append({&entry});
01784             return;
01785         }
01786     }
01787 }

01789 void
01790 RaftConsensus::append(const std::vector<const Log::Entry*>& entries)
01791 {
01792     for (auto it = entries.begin(); it != entries.end(); ++it)
01793         assert((*it)->term() != 0);
01794     std::pair<uint64_t, uint64_t> range = log->append(entries);
01795     if (state == State::LEADER) { // defer log sync
01796         logSyncQueued = true;
01797     } else { // sync log now
01798         std::unique_ptr<Log::Sync> sync = log->takeSync();
01799         sync->wait();
01800         log->syncComplete(std::move(sync));
01801     }
01802     uint64_t index = range.first;
01803     for (auto it = entries.begin(); it != entries.end(); ++it) {
01804         const Log::Entry& entry = **it;
01805         if (entry.type() == Protocol::Raft::EntryType::CONFIGURATION)
01806             configurationManager->add(index, entry.configuration());
01807         ++index;
01808     }
01809     stateChanged.notify_all();
01810 }

01812 void
01813 RaftConsensus::appendEntries(std::unique_lock<Mutex>& lockGuard,
01814                              Peer& peer)
01815 {
01816     uint64_t lastLogIndex = log->getLastLogIndex();
01817     uint64_t prevLogIndex = peer.nextIndex - 1;
01818     assert(prevLogIndex <= lastLogIndex);
01819 
01820     // Don't have needed entry: send a snapshot instead.
01821     if (peer.nextIndex < log->getLogStartIndex()) {
01822         appendSnapshotChunk(lockGuard, peer);
01823         return;
01824     }
01825 
01826     // Find prevLogTerm or fall back to sending a snapshot.
01827     uint64_t prevLogTerm;
01828     if (prevLogIndex >= log->getLogStartIndex()) {
01829         prevLogTerm = log->getEntry(prevLogIndex).term();
01830     } else if (prevLogIndex == 0) {
01831         prevLogTerm = 0;
01832     } else if (prevLogIndex == lastSnapshotIndex) {
01833         prevLogTerm = lastSnapshotTerm;
01834     } else {
01835         // Don't have needed entry for prevLogTerm: send snapshot instead.
01836         appendSnapshotChunk(lockGuard, peer);
01837         return;
01838     }
01839 
01840     // Build up request
01841     Protocol::Raft::AppendEntries::Request request;
01842     request.set_server_id(serverId);
01843     request.set_recipient_id(peer.serverId);
01844     request.set_term(currentTerm);
01845     request.set_prev_log_term(prevLogTerm);
01846     request.set_prev_log_index(prevLogIndex);
01847 
01848     // Add as many as entries as will fit comfortably in the request. It's
01849     // easiest to add one entry at a time until the RPC gets too big, then back
01850     // the last one out.
01851     uint64_t numEntries = 0;
01852     if (!peer.forceHeartbeat) {
01853         for (uint64_t entryId = peer.nextIndex;
01854              entryId <= lastLogIndex;
01855              ++entryId) {
01856             const Log::Entry& entry = log->getEntry(entryId);
01857             *request.add_entries() = entry;
01858             uint64_t requestSize =
01859                 Core::Util::downCast<uint64_t>(request.ByteSize());
01860             if (requestSize < SOFT_RPC_SIZE_LIMIT || numEntries == 0) {
01861                 // this entry fits, send it
01862                 VERBOSE("sending entry <id=%lu,term=%lu>",
01863                         entryId, entry.term());
01864                 ++numEntries;
01865             } else {
01866                 // this entry doesn't fit, discard it
01867                 request.mutable_entries()->RemoveLast();
01868             }
01869         }
01870     }
01871     request.set_commit_index(std::min(commitIndex, prevLogIndex + numEntries));
01872 
01873     // Execute RPC
01874     Protocol::Raft::AppendEntries::Response response;
01875     TimePoint start = Clock::now();
01876     uint64_t epoch = currentEpoch;
01877     bool ok = peer.callRPC(Protocol::Raft::OpCode::APPEND_ENTRIES,
01878                            request, response,
01879                            lockGuard);
01880     if (!ok) {
01881         peer.backoffUntil = start +
01882             std::chrono::milliseconds(RPC_FAILURE_BACKOFF_MS);
01883         return;
01884     }
01885 
01886     // Process response
01887 
01888     if (currentTerm != request.term() || peer.exiting) {
01889         // we don't care about result of RPC
01890         return;
01891     }
01892     // Since we were leader in this term before, we must still be leader in
01893     // this term.
01894     assert(state == State::LEADER);
01895     if (response.term() > currentTerm) {
01896         NOTICE("Received AppendEntries response from server %lu in term %lu "
01897                "(this server's term was %lu)",
01898                 peer.serverId, response.term(), currentTerm);
01899         stepDown(response.term());
01900     } else {
01901         assert(response.term() == currentTerm);
01902         peer.lastAckEpoch = epoch;
01903         stateChanged.notify_all();
01904         peer.nextHeartbeatTime = start +
01905             std::chrono::milliseconds(HEARTBEAT_PERIOD_MS);
01906         if (response.success()) {
01907             if (peer.lastAgreeIndex > prevLogIndex + numEntries) {
01908                 // Revisit this warning if we pipeline AppendEntries RPCs for
01909                 // performance.
01910                 WARNING("lastAgreeIndex should monotonically increase within a "
01911                         "term, since servers don't forget entries. But it "
01912                         "didn't.");
01913             } else {
01914                 peer.lastAgreeIndex = prevLogIndex + numEntries;
01915                 advanceCommittedId();
01916             }
01917             peer.nextIndex = peer.lastAgreeIndex + 1;
01918             peer.forceHeartbeat = false;
01919 
01920             if (!peer.isCaughtUp_ &&
01921                 peer.thisCatchUpIterationGoalId <= peer.lastAgreeIndex) {
01922                 Clock::duration duration =
01923                     Clock::now() - peer.thisCatchUpIterationStart;
01924                 uint64_t thisCatchUpIterationMs =
01925                     uint64_t(std::chrono::duration_cast<
01926                                  std::chrono::milliseconds>(duration).count());
01927                 if (uint64_t(labs(int64_t(peer.lastCatchUpIterationMs -
01928                                           thisCatchUpIterationMs))) <
01929                     ELECTION_TIMEOUT_MS) {
01930                     peer.isCaughtUp_ = true;
01931                     stateChanged.notify_all();
01932                 } else {
01933                     peer.lastCatchUpIterationMs = thisCatchUpIterationMs;
01934                     peer.thisCatchUpIterationStart = Clock::now();
01935                     peer.thisCatchUpIterationGoalId = log->getLastLogIndex();
01936                 }
01937             }
01938         } else {
01939             if (peer.nextIndex > 1)
01940                 --peer.nextIndex;
01941         }
01942     }
01943 }

01945 void
01946 RaftConsensus::appendSnapshotChunk(std::unique_lock<Mutex>& lockGuard,
01947                                    Peer& peer)
01948 {
01949     // Build up request
01950     Protocol::Raft::AppendSnapshotChunk::Request request;
01951     request.set_server_id(serverId);
01952     request.set_recipient_id(peer.serverId);
01953     request.set_term(currentTerm);
01954 
01955     // Open the latest snapshot if we haven't already. Stash a copy of the
01956     // lastSnapshotIndex that goes along with the file, since it's possible
01957     // that this will change while we're transferring chunks).
01958     if (!peer.snapshotFile) {
01959         namespace FilesystemUtil = Storage::FilesystemUtil;
01960         peer.snapshotFile.reset(new FilesystemUtil::FileContents(
01961             FilesystemUtil::openFile(storageDirectory, "snapshot", O_RDONLY)));
01962         peer.snapshotFileOffset = 0;
01963         peer.lastSnapshotIndex = lastSnapshotIndex;
01964     }
01965     request.set_last_snapshot_index(peer.lastSnapshotIndex);
01966     request.set_byte_offset(peer.snapshotFileOffset);
01967     // The amount of data we can send is bounded by the remaining bytes in the
01968     // file and the maximum length for RPCs.
01969     uint64_t numDataBytes = std::min(
01970         peer.snapshotFile->getFileLength() - peer.snapshotFileOffset,
01971         SOFT_RPC_SIZE_LIMIT);
01972     request.set_data(peer.snapshotFile->get<char>(peer.snapshotFileOffset,
01973                                                   numDataBytes),
01974                      numDataBytes);
01975     request.set_done(peer.snapshotFileOffset + numDataBytes ==
01976                      peer.snapshotFile->getFileLength());
01977 
01978     // Execute RPC
01979     Protocol::Raft::AppendSnapshotChunk::Response response;
01980     TimePoint start = Clock::now();
01981     uint64_t epoch = currentEpoch;
01982     bool ok = peer.callRPC(Protocol::Raft::OpCode::APPEND_SNAPSHOT_CHUNK,
01983                            request, response,
01984                            lockGuard);
01985     if (!ok) {
01986         peer.backoffUntil = start +
01987             std::chrono::milliseconds(RPC_FAILURE_BACKOFF_MS);
01988         return;
01989     }
01990 
01991     // Process response
01992 
01993     if (currentTerm != request.term() || peer.exiting) {
01994         // we don't care about result of RPC
01995         return;
01996     }

02108

02112         ok = stream.ReadLittleEndian32(&numBytes);
02113         if (!ok)
02114             PANIC("couldn't read snapshot");

02116         auto limit = stream.PushLimit(numBytes);
02117         ok = header.MergePartialFromCodedStream(&stream);
02118         stream.PopLimit(limit);
02119         if (!ok)

02121 
02122         // load header contents
02123         if (header.last_included_index() < lastSnapshotIndex) {
02124             PANIC("Trying to load a snapshot that is more stale than one this "
02125                   "server loaded earlier. The earlier snapshot covers through "
02126                   "log index %lu (inclusive); this one covers through log "
02127                   "index %lu (inclusive)",
02128                   lastSnapshotIndex,
02129                   header.last_included_index());
02130 
02131         }
02132         lastSnapshotIndex = header.last_included_index();
02133         lastSnapshotTerm = header.last_included_term();

02136 
02137         NOTICE("Reading snapshot which covers log entries 1 through %lu "
02138                "(inclusive)", lastSnapshotIndex);
02139 
02140         // We should keep log entries if they might be needed for a quorum. So:
02141         // 1. Discard log if it is shorter than the snapshot.
02142         // 2. Discard log if its lastSnapshotIndex entry disagrees with the
02143         //    lastSnapshotTerm.
02144         if (log->getLastLogIndex() < lastSnapshotIndex ||

02146              log->getEntry(lastSnapshotIndex).term() != lastSnapshotTerm)) {

02148             // don't print it in that case. We still want to shift the log

02152                        "consistent with the snapshot that is being read");
02153             }

02155             // right place.
02156             log->truncatePrefix(lastSnapshotIndex + 1);
02157             log->truncateSuffix(lastSnapshotIndex);

02160             // Clean up resources.
02161             if (state == State::LEADER) { // defer log sync
02162                 logSyncQueued = true;
02163             } else { // sync log now
02164                 std::unique_ptr<Log::Sync> sync = log->takeSync();
02165                 sync->wait();

02167             }
02168         }
02169 
02170         discardUnneededEntries();
02171 
02172         if (header.has_configuration_index() && header.has_configuration()) {
02173             configurationManager->setSnapshot(header.configuration_index(),

02175         } else {

02177                     "should contain a configuration (they're the first thing "
02178                     "found in any log).");
02179         }
02180

02358 bool
02359 RaftConsensus::upToDateLeader(std::unique_lock<Mutex>& lockGuard) const
02360 {
02361     ++currentEpoch;
02362     uint64_t epoch = currentEpoch;
02363     // schedule a heartbeat now so that this returns quickly
02364     configuration->forEach(&Server::scheduleHeartbeat);
02365     stateChanged.notify_all();
02366     while (true) {
02367         if (exiting || state != State::LEADER)
02368             return false;
02369         if (configuration->quorumMin(&Server::getLastAckEpoch) >= epoch) {
02370             // So we know we're the current leader, but do we have an
02371             // up-to-date commitIndex yet? What we'd like to check is whether
02372             // the entry's term at commitIndex matches our currentTerm, but
02373             // snapshots mean that we may not have the entry in our log. Since
02374             // commitIndex >= lastSnapshotIndex, we split into two cases:
02375             uint64_t commitTerm;
02376             if (commitIndex == lastSnapshotIndex) {
02377                 commitTerm = lastSnapshotTerm;
02378             } else {
02379                 assert(commitIndex > lastSnapshotIndex);
02380                 assert(commitIndex >= log->getLogStartIndex());
02381                 assert(commitIndex <= log->getLastLogIndex());
02382                 commitTerm = log->getEntry(commitIndex).term();
02383             }
02384             if (commitTerm == currentTerm)
02385                 return true;
02386         }
02387         stateChanged.wait(lockGuard);
02388     }
02389 }

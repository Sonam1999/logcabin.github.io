00060     Status call(OpCode opCode,
00061               const google::protobuf::Message& request,
00062               google::protobuf::Message& response,
00063               TimePoint timeout) {
00064         std::lock_guard<std::recursive_mutex> lockGuard(mutex);
00065         if (opCode == OpCode::STATE_MACHINE_QUERY) {
00066             PC::StateMachineQuery::Request qrequest;
00067             qrequest.CopyFrom(request);
00068             auto& qresponse =
00069                 static_cast<PC::StateMachineQuery::Response&>(response);
00070             auto localCallbacks = std::make_shared<TestingCallbacks>();
00071             // set to noop for recursive calls, then restore
00072             CallbackSwap s(callbacks, localCallbacks);
00073             if (localCallbacks->stateMachineQuery(qrequest, qresponse))
00074                 return Status::OK;
00075             qresponse.Clear();
00076             if (qrequest.has_tree()) {
00077                 LogCabin::Tree::ProtoBuf::readOnlyTreeRPC(
00078                     tree, qrequest.tree(), *qresponse.mutable_tree());
00079                 return Status::OK;
00080             }
00081         } else if (opCode == OpCode::STATE_MACHINE_COMMAND) {
00082             PC::StateMachineCommand::Request crequest;
00083             crequest.CopyFrom(request);
00084             auto& cresponse =
00085                 static_cast<PC::StateMachineCommand::Response&>(response);
00086             auto localCallbacks = std::make_shared<TestingCallbacks>();
00087             // set to noop for recursive calls, then restore
00088             CallbackSwap s(callbacks, localCallbacks);
00089             if (localCallbacks->stateMachineCommand(crequest, cresponse))
00090                 return Status::OK;
00091             cresponse.Clear();
00092             if (crequest.has_tree()) {
00093                 LogCabin::Tree::ProtoBuf::readWriteTreeRPC(
00094                     tree, crequest.tree(), *cresponse.mutable_tree());
00095                 return Status::OK;
00096             } else if (crequest.has_open_session()) {
00097                 cresponse.mutable_open_session()->
00098                     set_client_id(1);
00099                 return Status::OK;
00100             } else if (crequest.has_close_session()) {
00101                 return Status::OK;
00102             }
00103         }
00104         PANIC("Unexpected request: %d %s",
00105               opCode,
00106               Core::ProtoBuf::dumpString(request).c_str());
00107     }

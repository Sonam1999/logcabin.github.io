00413 std::pair<uint64_t, uint64_t>
00414 SegmentedLog::append(const std::vector<const Entry*>& entries)
00415 {
00416     Segment* openSegment = &getOpenSegment();
00417     uint64_t startIndex = openSegment->endIndex + 1;
00418     uint64_t index = startIndex;
00419     for (auto it = entries.begin(); it != entries.end(); ++it) {
00420         Segment::Record record(openSegment->bytes);
00421         // Note that record.offset may change later, if this entry doesn't fit.
00422         record.entry = **it;
00423         record.entry.set_index(index);
00424         Core::Buffer buf = serializeProto(record.entry);
00425 
00426         // See if we need to roll over to a new head segment. If someone is
00427         // writing an entry that is bigger than MAX_SEGMENT_SIZE, just put it
00428         // in its own segment. This duplicates some code from closeSegment(),
00429         // but queues up the operations into 'currentSync'.
00430         if (openSegment->bytes > 0 &&
00431             openSegment->bytes + buf.getLength() > MAX_SEGMENT_SIZE) {
00432             NOTICE("Rolling over to new head segment: trying to append new "
00433                    "entry that is %lu bytes long, but open segment is already "
00434                    "%lu of %lu bytes large",
00435                    buf.getLength(),
00436                    openSegment->bytes,
00437                    MAX_SEGMENT_SIZE);
00438 
00439             // Truncate away any extra 0 bytes at the end from when
00440             // MAX_SEGMENT_SIZE was allocated.
00441             currentSync->ops.emplace_back(openSegmentFile.fd,
00442                                           Sync::Op::TRUNCATE);
00443             currentSync->ops.back().size = openSegment->bytes;
00444             currentSync->ops.emplace_back(openSegmentFile.fd,
00445                                           Sync::Op::FSYNC);
00446             currentSync->ops.emplace_back(openSegmentFile.release(),
00447                                           Sync::Op::CLOSE);
00448 
00449             // Rename the file.
00450             std::string newFilename = openSegment->makeClosedFilename();
00451             NOTICE("Closing full segment (was %s, renaming to %s)",
00452                    openSegment->filename.c_str(),
00453                    newFilename.c_str());
00454             currentSync->ops.emplace_back(dir.fd, Sync::Op::RENAME);
00455             currentSync->ops.back().filename1 = openSegment->filename;
00456             currentSync->ops.back().filename2 = newFilename;
00457             currentSync->ops.emplace_back(dir.fd, Sync::Op::FSYNC);
00458             openSegment->filename = newFilename;
00459 
00460             // Bookkeeping.
00461             openSegment->isOpen = false;
00462             totalClosedSegmentBytes += openSegment->bytes;
00463 
00464             // Open new segment.
00465             openNewSegment();
00466             openSegment = &getOpenSegment();
00467             record.offset = 0;
00468         }
00469 
00470         if (buf.getLength() > MAX_SEGMENT_SIZE) {
00471             WARNING("Trying to append an entry of %lu bytes when the maximum "
00472                     "segment size is %lu bytes. Placing this entry in its own "
00473                     "segment. Consider adjusting 'storageSegmentBytes' in the "
00474                     "config.",
00475                     buf.getLength(),
00476                     MAX_SEGMENT_SIZE);
00477         }
00478 
00479         openSegment->entries.emplace_back(std::move(record));
00480         openSegment->bytes += buf.getLength();
00481         currentSync->ops.emplace_back(openSegmentFile.fd, Sync::Op::WRITE);
00482         currentSync->ops.back().writeData = std::move(buf);
00483         ++openSegment->endIndex;
00484         ++index;
00485     }
00486 
00487     currentSync->ops.emplace_back(openSegmentFile.fd, Sync::Op::FDATASYNC);
00488     currentSync->lastIndex = getLastLogIndex();
00489     checkInvariants();
00490     return {startIndex, getLastLogIndex()};
00491 }

00590         Segment& openSegment = getOpenSegment();
00591         if (newEndIndex >= openSegment.startIndex) {
00592             // Update in-memory segment
00593             uint64_t i = newEndIndex + 1 - openSegment.startIndex;
00594             openSegment.bytes = openSegment.entries.at(i).offset;
00595             openSegment.entries.erase(
00596                 openSegment.entries.begin() + i,
00597                 openSegment.entries.end());

00599             // Truncate and close the open segment, and open a new one.
00600             closeSegment();
00601             openNewSegment();

00604         }
00605     }
00606 
00607     { // Remove the open segment.
00608         Segment& openSegment = getOpenSegment();
00609         openSegment.endIndex = openSegment.startIndex - 1;
00610         openSegment.bytes = 0;
00611         closeSegment();

00615     while (!segmentsByStartIndex.empty()) {
00616         auto it = segmentsByStartIndex.rbegin();

00618         if (segment.endIndex == newEndIndex)
00619             break;

00621             NOTICE("Removing closed segment %s", segment.filename.c_str());

00630             totalClosedSegmentBytes -= (segment.bytes - newBytes);
00631             segment.bytes = newBytes;

00633                 segment.entries.begin() + i,

00804         NOTICE("Removing closed segment whose entries are no longer "

01056 std::string
01057 SegmentedLog::readProtoFromFile(const FS::File& file,
01058                                 FS::FileContents& reader,
01059                                 uint64_t* offset,
01060                                 google::protobuf::Message* out) const
01061 {
01062     uint64_t loffset = *offset;
01063     char checksum[Core::Checksum::MAX_LENGTH];
01064     uint64_t bytesRead = reader.copyPartial(loffset, checksum,
01065                                             sizeof(checksum));
01066     uint32_t checksumBytes = Core::Checksum::length(checksum,
01067                                                     uint32_t(bytesRead));
01068     if (checksumBytes == 0)
01069         return format("Missing checksum in file %s", file.path.c_str());
01070     loffset += checksumBytes;
01071 
01072     uint64_t dataLen;
01073     if (reader.copyPartial(loffset, &dataLen, sizeof(dataLen)) <
01074         sizeof(dataLen)) {
01075         return format("Record length truncated in file %s", file.path.c_str());
01076     }
01077     dataLen = be64toh(dataLen);
01078     if (reader.getFileLength() < loffset + sizeof(dataLen) + dataLen) {
01079         return format("ProtoBuf truncated in file %s", file.path.c_str());
01080     }
01081 
01082     const void* checksumCoverage = reader.get(loffset,
01083                                               sizeof(dataLen) + dataLen);
01084     std::string error = Core::Checksum::verify(checksum, checksumCoverage,
01085                                                sizeof(dataLen) + dataLen);
01086     if (!error.empty()) {
01087         return format("Checksum verification failure on %s: %s",
01088                       file.path.c_str(), error.c_str());
01089     }
01090     loffset += sizeof(dataLen);
01091     const void* data = reader.get(loffset, dataLen);
01092     loffset += dataLen;
01093 
01094     switch (encoding) {
01095         case SegmentedLog::Encoding::BINARY: {
01096             Core::Buffer contents(const_cast<void*>(data),
01097                                   dataLen,
01098                                   NULL);
01099             if (!Core::ProtoBuf::parse(contents, *out)) {
01100                 return format("Failed to parse protobuf in %s",
01101                               file.path.c_str());
01102             }
01103             break;
01104         }
01105         case SegmentedLog::Encoding::TEXT: {
01106             std::string contents(static_cast<const char*>(data), dataLen);
01107             Core::ProtoBuf::Internal::fromString(contents, *out);
01108             break;
01109         }
01110     }
01111     *offset = loffset;
01112     return "";
01113 }

01119     const void* data = NULL;
01120     uint64_t len = 0;
01121     Core::Buffer binaryContents;

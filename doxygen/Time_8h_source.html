00030 /**
00031  * Convert a C++11 time point into a POSIX timespec.
00032  * \param when
00033  *      Time point to convert.
00034  * \return
00035  *      Time in seconds and nanoseconds relative to the Clock's epoch.
00036  */
00037 template<typename Clock, typename Duration>
00038 struct timespec
00039 makeTimeSpec(const std::chrono::time_point<Clock, Duration>& when)
00040 {
00041     std::chrono::nanoseconds::rep nanosSinceEpoch =
00042         std::chrono::duration_cast<std::chrono::nanoseconds>(
00043            when.time_since_epoch()).count();
00044     struct timespec ts;
00045     ts.tv_sec  = nanosSinceEpoch / 1000000000L;
00046     ts.tv_nsec = nanosSinceEpoch % 1000000000L;
00047     // tv_nsec must always be in range [0, 1e9)
00048     if (nanosSinceEpoch < 0) {
00049         ts.tv_sec  -= 1;
00050         ts.tv_nsec += 1000000000L;
00051     }
00052     return ts;
00053 }

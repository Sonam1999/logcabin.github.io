00097 void
00098 dumpTree(const Tree& tree, std::string path)
00099 {
00100     std::cout << path << std::endl;
00101     std::vector<std::string> children = tree.listDirectoryEx(path);
00102     for (auto it = children.begin(); it != children.end(); ++it) {
00103         std::string child = path + *it;
00104         if (*child.rbegin() == '/') { // directory
00105             dumpTree(tree, child);
00106         } else { // file
00107             std::cout << child << ": " << std::endl;
00108             std::cout << "    " << tree.readEx(child) << std::endl;
00109         }
00110     }

00094 Invariants::checkBasic()
00095 {
00096     // Log terms monotonically increase
00097     uint64_t lastTerm = 0;
00098     for (uint64_t entryId = consensus.log->getLogStartIndex();
00099          entryId <= consensus.log->getLastLogIndex();
00100          ++entryId) {
00101         const Storage::Log::Entry& entry = consensus.log->getEntry(entryId);
00102         expect(entry.term() >= lastTerm);
00103         lastTerm = entry.term();
00104     }
00105     // The terms in the log do not exceed currentTerm
00106     expect(lastTerm <= consensus.currentTerm);
00107 
00108     // The current configuration should be the last one found in the log
00109     bool found = false;
00110     for (uint64_t entryId = consensus.log->getLastLogIndex();
00111          entryId >= consensus.log->getLogStartIndex();
00112          --entryId) {
00113         const Storage::Log::Entry& entry = consensus.log->getEntry(entryId);
00114         if (entry.type() == Protocol::Raft::EntryType::CONFIGURATION) {
00115             expect(consensus.configuration->id == entryId);
00116             expect(consensus.configuration->state !=
00117                    Configuration::State::BLANK);
00118             found = true;
00119             break;
00120         }
00121     }
00122     if (!found) {
00123         if (consensus.log->getLogStartIndex() == 1) {
00124             expect(consensus.configuration->id == 0);
00125             expect(consensus.configuration->state ==
00126                    Configuration::State::BLANK);
00127         } else {
00128             expect(consensus.configuration->id <= consensus.lastSnapshotIndex);
00129         }
00130     }
00131 
00132     // Every configuration present in the log should also be present in the
00133     // configurationDescriptions map.
00134     for (uint64_t entryId = consensus.log->getLogStartIndex();
00135          entryId <= consensus.log->getLastLogIndex();
00136          ++entryId) {
00137         const Storage::Log::Entry& entry = consensus.log->getEntry(entryId);
00138         if (entry.type() == Protocol::Raft::EntryType::CONFIGURATION) {
00139             auto it = consensus.configurationManager->
00140                                         descriptions.find(entryId);
00141             expect(it != consensus.configurationManager->descriptions.end());
00142             if (it != consensus.configurationManager->descriptions.end())
00143                 expect(it->second == entry.configuration());
00144         }
00145     }
00146     // The configuration descriptions map shouldn't have anything past the
00147     // snapshot and the log.
00148     expect(consensus.configurationManager->descriptions.upper_bound(
00149                 std::max(consensus.log->getLastLogIndex(),
00150                          consensus.lastSnapshotIndex)) ==
00151            consensus.configurationManager->descriptions.end());
00152 
00153     // Servers with blank configurations should remain passive. Since the first
00154     // entry in every log is a configuration, they should also have empty logs.
00155     if (consensus.configuration->state == Configuration::State::BLANK) {
00156         expect(consensus.state == RaftConsensus::State::FOLLOWER);
00157         expect(consensus.log->getLastLogIndex() == 0);
00158     }
00159 
00160     // The last snapshot covers a committed range.
00161     expect(consensus.commitIndex >= consensus.lastSnapshotIndex);
00162 
00163     // The commitIndex doesn't exceed the length of the log/snapshot.
00164     expect(consensus.commitIndex <= consensus.log->getLastLogIndex());
00165 
00166     // The last log index points at least through the end of the last snapshot.
00167     expect(consensus.log->getLastLogIndex() >= consensus.lastSnapshotIndex);
00168 
00169     // lastLogIndex is either just below the log start (for empty logs) or
00170     // larger (for non-empty logs)
00171     assert(consensus.log->getLastLogIndex() >=
00172            consensus.log->getLogStartIndex() - 1);
00173 
00174     // advanceCommittedId is called everywhere it needs to be.
00175     if (consensus.state == RaftConsensus::State::LEADER) {
00176         uint64_t majorityEntry =
00177             consensus.configuration->quorumMin(&Server::getLastAgreeIndex);
00178         expect(consensus.commitIndex >= majorityEntry ||
00179                majorityEntry < consensus.log->getLogStartIndex() ||
00180                consensus.log->getEntry(majorityEntry).term() !=
00181                     consensus.currentTerm);
00182     }
00183 
00184     // A leader always points its leaderId at itself.
00185     if (consensus.state == RaftConsensus::State::LEADER)
00186         expect(consensus.leaderId == consensus.serverId);
00187 
00188     // A leader always voted for itself. (Candidates can vote for others when
00189     // they abort an election.)
00190     if (consensus.state == RaftConsensus::State::LEADER) {
00191         expect(consensus.votedFor == consensus.serverId);
00192     }
00193 
00194     // A follower and candidate always has a timer set; a leader has it at
00195     // TimePoint::max().
00196     if (consensus.state == RaftConsensus::State::LEADER) {
00197         expect(consensus.startElectionAt == TimePoint::max());
00198     } else {
00199         expect(consensus.startElectionAt > TimePoint::min());
00200         expect(consensus.startElectionAt <=
00201                Clock::now() + std::chrono::milliseconds(
00202                                     RaftConsensus::ELECTION_TIMEOUT_MS * 2));
00203     }
00204

00206     expect(consensus.log->metadata.current_term() == consensus.currentTerm);

00218                                         new ConsensusSnapshot(consensus));
00219     // Within a term, ...
00220     if (previous->currentTerm == current->currentTerm) {
00221         // the leader is set at most once.
00222         if (previous->leaderId != 0)
00223             expect(previous->leaderId == current->leaderId);
00224         // the vote is set at most once.
00225         if (previous->votedFor != 0)
00226             expect(previous->votedFor == current->votedFor);
00227         // a leader stays a leader.
00228         if (previous->state == RaftConsensus::State::LEADER)
00229             expect(current->state == RaftConsensus::State::LEADER);
00230     }
00231 
00232     // Once exiting is set, it doesn't get unset.
00233     if (previous->exiting)
00234         expect(current->exiting);
00235 
00236     // These variables monotonically increase.
00237     expect(previous->currentTerm <= current->currentTerm);
00238     expect(previous->commitIndex <= current->commitIndex);
00239     expect(previous->currentEpoch <= current->currentEpoch);
00240 
00241     // Change requires condition variable notification:
00242     if (previous->stateChangedCount == current->stateChangedCount) {
00243         expect(previous->currentTerm == current->currentTerm);
00244         expect(previous->state == current->state);
00245         expect(previous->lastLogIndex == current->lastLogIndex);

00090 ClientRPC::Status
00091 ClientRPC::waitForReply(google::protobuf::Message* response,
00092                         google::protobuf::Message* serviceSpecificError)
00093 {
00094     opaqueRPC.waitForReply();
00095     switch (opaqueRPC.getStatus()) {
00096         case OpaqueClientRPC::Status::NOT_READY:
00097             PANIC("Waited for RPC but not ready");
00098         case OpaqueClientRPC::Status::OK:
00099             break;
00100         case OpaqueClientRPC::Status::ERROR:
00101             return Status::RPC_FAILED;
00102         case OpaqueClientRPC::Status::CANCELED:
00103             return Status::RPC_CANCELED;
00104     }
00105     const Buffer& responseBuffer = *opaqueRPC.peekReply();
00106 
00107     // Extract the response's status field.
00108     if (responseBuffer.getLength() < sizeof(ResponseHeaderPrefix)) {
00109         PANIC("The response from the server was too short to be valid. "
00110               "This probably indicates network or memory corruption.");
00111     }
00112     ResponseHeaderPrefix responseHeaderPrefix =
00113         *static_cast<const ResponseHeaderPrefix*>(responseBuffer.getData());
00114     responseHeaderPrefix.fromBigEndian();
00115     if (responseHeaderPrefix.status == ProtocolStatus::INVALID_VERSION) {
00116         // The server doesn't understand this version of the header
00117         // protocol. Since this library only runs version 1 of the
00118         // protocol, this shouldn't happen if servers continue supporting
00119         // version 1.
00120         PANIC("This client is too old to talk to the server. "
00121               "You'll need to update your client library.");
00122     }
00123 
00124     if (responseBuffer.getLength() < sizeof(ResponseHeaderVersion1)) {
00125         PANIC("The response from the server was too short to be valid. "
00126               "This probably indicates network or memory corruption.");
00127     }
00128     ResponseHeaderVersion1 responseHeader =
00129         *static_cast<const ResponseHeaderVersion1*>(responseBuffer.getData());
00130     responseHeader.fromBigEndian();
00131 
00132     switch (responseHeader.prefix.status) {
00133 
00134         // The RPC succeeded. Parse the response into a protocol buffer.
00135         case ProtocolStatus::OK:
00136             if (response != NULL &&
00137                 !RPC::ProtoBuf::parse(responseBuffer, *response,
00138                                       sizeof(responseHeader))) {
00139                 PANIC("Could not parse the protocol buffer out of the server "
00140                       "response");
00141             }
00142             return Status::OK;
00143 
00144         // The RPC failed in a service-specific way. Parse the response into a
00145         // protocol buffer.
00146         case ProtocolStatus::SERVICE_SPECIFIC_ERROR:
00147             if (serviceSpecificError != NULL &&
00148                 !RPC::ProtoBuf::parse(responseBuffer, *serviceSpecificError,
00149                                       sizeof(responseHeader))) {
00150                 PANIC("Could not parse the protocol buffer out of the "
00151                       "service-specific error details");
00152             }
00153             return Status::SERVICE_SPECIFIC_ERROR;
00154 
00155         // The server does not have the requested service.
00156         case ProtocolStatus::INVALID_SERVICE:
00157             PANIC("The server is not running the requested service.");
00158 
00159         // The server disliked our request. This shouldn't happen because
00160         // the higher layers of software were supposed to negotiate an RPC
00161         // protocol version.
00162         case ProtocolStatus::INVALID_REQUEST:
00163             PANIC("The server found the request to be invalid. This "
00164                   "indicates a bug in the client or server in negotiating "
00165                   "which RPCs the client may legally send to the server.");
00166 
00167         default:
00168             // The server shouldn't reply back with status codes we don't
00169             // understand. That's why we gave it a version number in the
00170             // request header.
00171             PANIC("Unknown status %u returned from server after sending it "
00172                   "protocol version 1 in the request header. This probably "
00173                   "indicates a bug in the server.",
00174                   uint32_t(responseHeader.prefix.status));
00175     }

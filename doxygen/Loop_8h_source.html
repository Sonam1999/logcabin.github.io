00032 /**
00033  * This class contains an event loop based on Linux's epoll interface.
00034  * It keeps track of interesting events such as timers and socket activity, and
00035  * arranges for callbacks to be invoked when the events happen.
00036  */
00037 class Loop {
00038   public:
00039 
00040     /**
00041      * Lock objects are used to synchronize between the Event::Loop thread and
00042      * other threads.  As long as a Lock object exists the following guarantees
00043      * are in effect: either
00044      * (a) the thread is the event loop thread or
00045      * (b) no other thread has a Lock object and the event loop thread has
00046      *     paused in a safe place (with no event handlers active) waiting for
00047      *     the Lock to be destroyed.
00048      * Locks may be used recursively.
00049      */
00050     class Lock {
00051       public:
00052         /// Constructor. Acquire the lock.
00053         explicit Lock(Event::Loop& eventLoop);
00054         /// Destructor. Release the lock.
00055         ~Lock();
00056         /// Event::Loop to lock.
00057         Event::Loop& eventLoop;
00058       private:
00059         // Lock is not copyable.
00060         Lock(const Lock&) = delete;
00061         Lock& operator=(const Lock&) = delete;
00062     };

00064     /**
00065      * Constructor.
00066      */
00067     Loop();

00069     /**
00070      * Destructor. The caller must ensure that no events still exist and that
00071      * the event loop is not running.
00072      */
00073     ~Loop();

00075     /**
00076      * Run the main event loop until exit() is called.
00077      * It is safe to call this again after it returns.
00078      * The caller must ensure that only one thread is executing runForever() at
00079      * a time.
00080      */

00083     /**
00084      * Exit the main event loop, if one is running. It may return before

00096 LeaderRPC::Call::Status
00097 LeaderRPC::Call::wait(google::protobuf::Message& response,
00098                       TimePoint timeout)
00099 {
00100     typedef RPC::ClientRPC::Status RPCStatus;
00101     Protocol::Client::Error error;
00102     RPCStatus status = rpc.waitForReply(&response, &error, timeout);
00103 
00104     // Decode the response
00105     switch (status) {
00106         case RPCStatus::OK:
00107             return Call::Status::OK;
00108         case RPCStatus::SERVICE_SPECIFIC_ERROR:
00109             switch (error.error_code()) {
00110                 case Protocol::Client::Error::NOT_LEADER:
00111                     // The server we tried is not the current cluster leader.
00112                     if (error.has_leader_hint()) {
00113                         VERBOSE("Will try suggested %s as new leader "
00114                                 "(was using %s)",
00115                                 error.leader_hint().c_str(),
00116                                 cachedSession->toString().c_str());
00117                         leaderRPC.reportRedirect(cachedSession,
00118                                                  error.leader_hint());
00119                     } else {
00120                         VERBOSE("Will try random host as new leader "
00121                                 "(was using %s)",
00122                                 cachedSession->toString().c_str());
00123                         leaderRPC.reportFailure(cachedSession);
00124                     }
00125                     break;
00126                 case Protocol::Client::Error::SESSION_EXPIRED:
00127                     PANIC("Session expired");
00128                 default:
00129                     // Hmm, we don't know what this server is trying to tell
00130                     // us, but something is wrong. The server shouldn't reply
00131                     // back with error codes we don't understand. That's why we
00132                     // gave it a serverSpecificErrorVersion number in the
00133                     // request header.
00134                     PANIC("Unknown error code %u returned in service-specific "
00135                           "error. This probably indicates a bug in the server",
00136                           error.error_code());
00137             }
00138             break;
00139         case RPCStatus::RPC_FAILED:
00140             leaderRPC.reportFailure(cachedSession);
00141             break;
00142         case RPCStatus::RPC_CANCELED:
00143             break;
00144         case RPCStatus::TIMEOUT:
00145             return Call::Status::TIMEOUT;
00146     }
00147     if (timeout < Clock::now())
00148         return Call::Status::TIMEOUT;
00149     else
00150         return Call::Status::RETRY;
00151 }

00200 std::shared_ptr<RPC::ClientSession>
00201 LeaderRPC::getSession(TimePoint timeout)
00202 {
00203     std::unique_lock<std::mutex> lockGuard(mutex);
00204     if (leaderSession)
00205         return leaderSession;
00206 
00207     // sleep if we've tried to connect too much recently
00208     sessionCreationBackoff.delayAndBegin(timeout);
00209 
00210     RPC::Address address;
00211     if (leaderHint.empty()) {
00212         // Hope the next random host is the leader.
00213         // If that turns out to be false, we will soon find out.
00214         hosts.refresh(timeout);
00215         address = hosts;
00216     } else {
00217         address = RPC::Address(leaderHint, Protocol::Common::DEFAULT_PORT);
00218         address.refresh(timeout);
00219         leaderHint.clear();
00220     }
00221     leaderSession = RPC::ClientSession::makeSession(
00222                         eventLoop,
00223                         address,
00224                         Protocol::Common::MAX_MESSAGE_LENGTH,
00225                         timeout);
00226 
00227     return leaderSession;
00228 }

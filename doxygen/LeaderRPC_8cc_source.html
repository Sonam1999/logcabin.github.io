00063 bool
00064 LeaderRPC::Call::wait(google::protobuf::Message& response)
00065 {
00066     typedef RPC::ClientRPC::Status Status;
00067     Protocol::Client::Error serviceSpecificError;
00068     Status status = rpc.waitForReply(&response, &serviceSpecificError);
00069 
00070     // Decode the response
00071     switch (status) {
00072         case Status::OK:
00073             return true;
00074         case Status::SERVICE_SPECIFIC_ERROR:
00075             leaderRPC.handleServiceSpecificError(cachedSession,
00076                                                  serviceSpecificError);
00077             return false;
00078         case Status::RPC_FAILED:
00079             // If the session is broken, get a new one and try again.
00080             leaderRPC.connectRandom(cachedSession);
00081             return false;
00082         case Status::RPC_CANCELED:
00083             return false;
00084     }
00085     PANIC("Unexpected RPC status");
00086 }

00134 void
00135 LeaderRPC::handleServiceSpecificError(
00136         std::shared_ptr<RPC::ClientSession> cachedSession,
00137         const Protocol::Client::Error& error)
00138 {
00139     switch (error.error_code()) {
00140         case Protocol::Client::Error::NOT_LEADER:
00141             // The server we tried is not the current cluster leader.
00142             if (error.has_leader_hint()) {
00143                 // Server returned hint as to who the leader might be.
00144                 VERBOSE("Trying suggested %s as new leader (was using %s)",
00145                         error.leader_hint().c_str(),
00146                         cachedSession->toString().c_str());
00147                 connectHost(error.leader_hint(), cachedSession);
00148             } else {
00149                 // Well, this server isn't the leader. Try someone else.
00150                 VERBOSE("Trying random host as new leader (was using %s)",
00151                         cachedSession->toString().c_str());
00152                 connectRandom(cachedSession);
00153             }
00154             break;
00155         case Protocol::Client::Error::SESSION_EXPIRED:
00156             PANIC("Session expired");
00157         default:
00158             // Hmm, we don't know what this server is trying to tell us, but
00159             // something is wrong. The server shouldn't reply back with error
00160             // codes we don't understand. That's why we gave it a
00161             // serverSpecificErrorVersion number in the request header.
00162             PANIC("Unknown error code %u returned in service-specific error. "
00163                   "This probably indicates a bug in the server.",
00164                   error.error_code());
00165     }
00166 }

00168 void
00169 LeaderRPC::connect(const RPC::Address& address,
00170                    std::unique_lock<std::mutex>& lockGuard)
00171 {
00172     uint64_t nowNanos = Core::Time::getTimeNanos();
00173     if (lastConnectTimes.front() >  nowNanos - windowNanos) {
00174         usleep(unsigned(
00175             std::min(lastConnectTimes.front() + windowNanos - nowNanos,
00176                      windowNanos) / 1000));
00177         nowNanos = Core::Time::getTimeNanos();
00178     }
00179     lastConnectTimes.pop_front();
00180     lastConnectTimes.push_back(nowNanos);
00181     leaderSession = RPC::ClientSession::makeSession(
00182                         eventLoop,
00183                         address,
00184                         Protocol::Common::MAX_MESSAGE_LENGTH);
00185 }

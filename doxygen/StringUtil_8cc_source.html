00046 std::string
00047 format(const char* format, ...)
00048 {
00049     std::string s;
00050     va_list ap;
00051     va_start(ap, format);
00052 
00053     // We're not really sure how big of a buffer will be necessary.
00054     // Try 1K, if not the return value will tell us how much is necessary.
00055     int bufSize = 1024;
00056     while (true) {
00057         char buf[bufSize];
00058         // vsnprintf trashes the va_list, so copy it first
00059         va_list aq;
00060         __va_copy(aq, ap);
00061         int r = vsnprintf(buf, bufSize, format, aq);
00062         assert(r >= 0); // old glibc versions returned -1
00063         if (r < bufSize) {
00064             s = buf;
00065             break;
00066         }
00067         bufSize = r + 1;
00068     }
00069 
00070     va_end(ap);
00071     return s;
00072 }

00093            const std::string& replacement)
00094 {
00095     size_t startPos = 0;
00096     while (true) {
00097         size_t replacePos = haystack.find(needle, startPos);
00098         if (replacePos == haystack.npos)
00099             return;
00100         haystack.replace(replacePos, needle.length(), replacement);
00101         startPos = replacePos + replacement.length();
00102     }

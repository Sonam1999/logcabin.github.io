00052       std::initializer_list<std::pair<int, const char*>> flags)
00053 {
00054     if (value == 0)
00055         return "0";
00056     std::vector<std::string> strings;
00057     for (auto it = flags.begin(); it != flags.end(); ++it) {
00058         if (value & it->first) {
00059             strings.push_back(it->second);
00060             value &= ~it->first;

00062     }
00063     if (value)
00064         strings.push_back(format("0x%x", value));

00088             s = buf;

00090         }
00091         bufSize = r2 + 1;
00092     }

00168 trim(const std::string& original)
00169 {
00170     // The black magic is from https://stackoverflow.com/a/217605
00171     std::string s = original;
00172 
00173     // trim whitespace at end of string

00176                 .base(),
00177             s.end());
00178 
00179     // trim whitespace at beginning of string
00180     s.erase(s.begin(),

00182                          std::not1(std::ptr_fun<int, int>(std::isspace))));

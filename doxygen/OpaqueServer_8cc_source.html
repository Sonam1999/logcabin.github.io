00044 OpaqueServer::MessageSocketHandler::handleReceivedMessage(
00045         MessageId messageId,
00046         Core::Buffer message)
00047 {
00048     if (server == NULL)
00049         return;
00050     if (messageId == Protocol::Common::PING_MESSAGE_ID) { // ping request
00051         std::shared_ptr<SocketWithHandler> socketRef = self.lock();
00052         if (socketRef) { // expect so, since we're receiving messages
00053             VERBOSE("Responding to ping");

00055                                            Core::Buffer());
00056         }
00057     } else { // normal RPC request
00058         VERBOSE("Handling RPC");
00059         OpaqueServerRPC rpc(self, messageId, std::move(message));
00060         server->rpcHandler.handleRPC(std::move(rpc));
00061     }

00064 void
00065 OpaqueServer::MessageSocketHandler::handleDisconnect()
00066 {
00067     VERBOSE("Disconnected from client");
00068     std::shared_ptr<SocketWithHandler> socketRef = self.lock();
00069     if (server != NULL && socketRef) {
00070         // This drops the reference count on the socket. It may cause the
00071         // SocketWithHandler object (which includes this object) to be
00072         // destroyed when 'socketRef' goes out of scope.
00073         server->sockets.erase(socketRef);
00074         server = NULL;
00075     }
00076 }

00158 OpaqueServer::~OpaqueServer()
00159 {
00160     // Stop accepting new connections.
00161     {
00162         std::lock_guard<Core::Mutex> lock(boundListenersMutex);
00163         boundListeners.clear();
00164     }
00165 
00166     // Stop the socket objects from handling new RPCs and accessing the
00167     // 'sockets' set. They may continue to process existing RPCs, though
00168     // idle sockets will be destroyed here.
00169     {
00170         // Block the event loop to operate on 'sockets' safely.
00171         Event::Loop::Lock lockGuard(eventLoop);
00172         for (auto it = sockets.begin(); it != sockets.end(); ++it) {
00173             std::shared_ptr<SocketWithHandler> socket = *it;
00174             socket->handler.server = NULL;

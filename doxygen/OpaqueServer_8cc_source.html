00072 void
00073 OpaqueServer::ServerMessageSocket::onReceiveMessage(
00074         MessageId messageId,
00075         Buffer message)
00076 {
00077     // Reply to ping requests here.
00078     if (messageId == PING_MESSAGE_ID) {
00079         VERBOSE("Responding to ping");
00080         sendMessage(PING_MESSAGE_ID, Buffer());

00082     }
00083     if (server != NULL) {
00084         VERBOSE("Handling RPC");
00085         OpaqueServerRPC rpc(self, messageId, std::move(message));

00097 void
00098 OpaqueServer::ServerMessageSocket::close()
00099 {
00100     Event::Loop::Lock lock(eventLoop);
00101     if (server != NULL) {
00102         auto& sockets = server->sockets;
00103         server = NULL;
00104         std::swap(sockets.at(socketsIndex), sockets.back());
00105         sockets.at(socketsIndex)->socketsIndex = socketsIndex;
00106         sockets.pop_back(); // may destroy this object, so do it last
00107     }
00108 }

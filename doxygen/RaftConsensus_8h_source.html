00771 /**
00772  * An implementation of the Raft consensus algorithm. The algorithm is
00773  * described at https://raftconsensus.github.io
00774  * . In brief, Raft divides time into terms and elects a leader at the
00775  * beginning of each term. This election mechanism guarantees that the emerging
00776  * leader has at least all committed log entries. Once a candidate has received
00777  * votes from a quorum, it replicates its own log entries in order to the
00778  * followers. The leader is the only machine that serves client requests.
00779  */

00784      * This is returned by getNextEntry().

00787         /// Default constructor.
00788         Entry();

00790         Entry(Entry&& other);
00791         /// Destructor.
00792         ~Entry();
00793 
00794         /**
00795          * The entry ID for this entry (or the last one a snapshot covers).
00796          * Pass this as the lastEntryId argument to the next call to
00797          * getNextEntry().
00798          */

00800 
00801         /**
00802          * The type of the entry.
00803          */
00804         enum {
00805             /**
00806              * This is a normal entry containing a client request for the state
00807              * machine. The 'data' field contains that request, and
00808              * 'snapshotReader' is not set.
00809              */
00810             DATA,
00811             /**
00812              * This is a snapshot: the state machine should clear its state and

00815              * from.

00823              * simply call back into getNextEntry() again with the next ID,
00824              * Entries of type 'SKIP' will have neither 'data' nor
00825              * 'snapshotReader' set.
00826              */
00827             SKIP,
00828         } type;
00829 
00830         /**
00831          * The client request for entries of type 'DATA'.

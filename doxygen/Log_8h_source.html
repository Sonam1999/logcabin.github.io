00032 /**
00033  * This interface is used by RaftConsensus to store log entries and metadata.
00034  * Typically, implementations will persist the log entries and metadata to
00035  * stable storage (but MemoryLog keeps it all in volatile memory).
00036  */
00037 class Log {
00038   public:
00039     /**
00040      * An interface for flushing newly appended log entries to stable storage.
00041      * Leaders do this in a separate thread, while followers and candidates do
00042      * this immediately after appending the entries.
00043      *
00044      * Callers should wait() on all Sync objects prior to calling
00045      * truncateSuffix(). This never happens on leaders, so it's not a real
00046      * limitation, but things may go wonky otherwise.
00047      */
00048     class Sync {
00049       public:
00050         explicit Sync(uint64_t lastIndex);
00051         virtual ~Sync();
00052         /**
00053          * Wait for the log entries to be durable.
00054          * This is safe to call while the Log is being accessed and modified
00055          * from a separate thread.
00056          * PANICs on errors.
00057          */
00058         virtual void wait() {}
00059         /**
00060          * The index of the last log entry that is being flushed.
00061          * After the call to wait, every entry in the log up to this one is
00062          * durable.
00063          */
00064         uint64_t lastIndex;
00065         /**
00066          * Used by destructor to make sure that Log::syncComplete was called.
00067          */
00068         bool completed;
00069     };

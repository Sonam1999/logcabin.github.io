00085 /**
00086  * Convert a string to a log level.
00087  * PANICs if the string is not a valid log level (case insensitive).
00088  */
00089 LogLevel
00090 logLevelFromString(const std::string& level)
00091 {
00092     for (uint32_t i = 0; logLevelToString[i] != NULL; ++i) {
00093         if (strcasecmp(logLevelToString[i], level.c_str()) == 0)
00094             return LogLevel(i);
00095     }
00096     log((LogLevel::ERROR), __FILE__, __LINE__, __FUNCTION__,
00097         "'%s' is not a valid log level.\n", level.c_str());
00098     abort();
00099 }

00101 /**
00102  * From the policy, calculate the most verbose log level that should be
00103  * displayed for this file.
00104  * This is slow, so isLogging() caches the results in isLoggingCache.
00105  *
00106  * Must be called with #mutex held.
00107  *
00108  * \param fileName
00109  *      Relative filename.
00110  */
00111 LogLevel
00112 getLogLevel(const char* fileName)
00113 {
00114     for (auto it = policy.begin(); it != policy.end(); ++it) {
00115         const std::string& pattern = it->first;
00116         const std::string& logLevel = it->second;
00117         if (Core::StringUtil::startsWith(fileName, pattern) ||
00118             Core::StringUtil::endsWith(fileName, pattern)) {
00119             return logLevelFromString(logLevel);
00120         }
00121     }
00122     return LogLevel::NOTICE;
00123 }

00142 /**
00143  * Strip out the common prefix of a filename to get a path from the project's
00144  * root directory.
00145  * \param fileName
00146  *      An absolute or relative filename, usually the value of __FILE__.
00147  * \return
00148  *      A nicer way to show display 'fileName' to the user.
00149  *      For example, this file would yield "Core/Debug.cc".
00150  */
00151 const char*
00152 relativeFileName(const char* fileName)
00153 {
00154     // Remove the prefix only if it matches that of __FILE__. This check is
00155     // needed in case someone compiles different files using different paths.
00156     if (strncmp(fileName, __FILE__, lengthFilePrefix) == 0)
00157         return fileName + lengthFilePrefix;
00158     else
00159         return fileName;

00198 bool
00199 isLogging(LogLevel level, const char* fileName)
00200 {
00201     std::unique_lock<std::mutex> lockGuard(mutex);
00202     LogLevel verbosity;
00203     auto it = isLoggingCache.find(fileName);
00204     if (it == isLoggingCache.end()) {
00205         verbosity = getLogLevel(relativeFileName(fileName));
00206         isLoggingCache[fileName] = verbosity;
00207     } else {
00208         verbosity = it->second;
00209     }
00210     return uint32_t(level) <= uint32_t(verbosity);
00211 }

00213 void
00214 log(LogLevel level,
00215     const char* fileName, uint32_t lineNum, const char* functionName,
00216     const char* format, ...)
00217 {
00218     va_list ap;
00219 
00220     // Don't use Core::Time here since it could potentially call PANIC.
00221     struct timespec now;
00222     clock_gettime(CLOCK_REALTIME, &now);
00223 
00224     // Failures are a little annoying here, since we can't exactly log
00225     // errors that come up.
00226     char formattedSeconds[64]; // a human-readable string now.tv_sec
00227     bool ok = false;
00228     { // First, try gmtime and strftime.
00229         struct tm calendarTime;
00230         if (gmtime_r(&now.tv_sec, &calendarTime) != NULL) {
00231             ok = (strftime(formattedSeconds,
00232                            sizeof(formattedSeconds),
00233                            "%F %T",
00234                            &calendarTime) > 0);
00235         }
00236     }
00237     if (!ok) { // If that failed, use the raw number.
00238         snprintf(formattedSeconds,
00239                  sizeof(formattedSeconds),
00240                  "%010lu",
00241                  now.tv_sec);
00242         formattedSeconds[sizeof(formattedSeconds) - 1] = '\0';
00243     }
00244 
00245     // This ensures that output on stderr won't be interspersed with other
00246     // output. This normally happens automatically for a single call to
00247     // fprintf, but must be explicit since we're using two calls here.
00248     flockfile(stream);
00249 
00250     fprintf(stream, "%s.%06lu %s:%d in %s() %s[%s:%s]: ",
00251             formattedSeconds, now.tv_nsec / 1000,
00252             relativeFileName(fileName), lineNum, functionName,
00253             logLevelToString[uint32_t(level)],
00254             processName.c_str(), ThreadId::getName().c_str());
00255 
00256     va_start(ap, format);
00257     vfprintf(stream, format, ap);
00258     va_end(ap);
00259 
00260     funlockfile(stream);
00261 
00262     fflush(stream);
00263 }

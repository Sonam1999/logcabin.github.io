00163 /**
00164  * Convert a string to a log level.
00165  * PANICs if the string is not a valid log level (case insensitive).
00166  */
00167 LogLevel
00168 logLevelFromString(const std::string& level)
00169 {
00170     for (uint32_t i = 0; logLevelToString[i] != NULL; ++i) {
00171         if (strcasecmp(logLevelToString[i], level.c_str()) == 0)
00172             return LogLevel(i);
00173     }
00174     log((LogLevel::ERROR), __FILE__, __LINE__, __FUNCTION__,
00175         "'%s' is not a valid log level.\n", level.c_str());
00176     abort();

00179 /**
00180  * From the policy, calculate the most verbose log level that should be
00181  * displayed for this file.
00182  * This is slow, so isLogging() caches the results in isLoggingCache.
00183  *
00184  * Must be called with #mutex held.
00185  *
00186  * \param fileName
00187  *      Relative filename.
00188  */
00189 LogLevel
00190 getLogLevel(const char* fileName)
00191 {
00192     for (auto it = policy.begin(); it != policy.end(); ++it) {
00193         const std::string& pattern = it->first;
00194         const std::string& logLevel = it->second;
00195         if (Core::StringUtil::startsWith(fileName, pattern) ||
00196             Core::StringUtil::endsWith(fileName, pattern)) {
00197             return logLevelFromString(logLevel);
00198         }
00199     }
00200     return LogLevel::NOTICE;
00201 }

00220 /**
00221  * Strip out the common prefix of a filename to get a path from the project's
00222  * root directory.

00285 bool
00286 isLogging(LogLevel level, const char* fileName)
00287 {
00288     std::unique_lock<std::mutex> lockGuard(mutex);
00289     LogLevel verbosity;
00290     auto it = isLoggingCache.find(fileName);
00291     if (it == isLoggingCache.end()) {
00292         verbosity = getLogLevel(relativeFileName(fileName));
00293         isLoggingCache[fileName] = verbosity;
00294     } else {
00295         verbosity = it->second;
00296     }
00297     return uint32_t(level) <= uint32_t(verbosity);

00046 MemoryLog::append(const std::vector<const Entry*>& newEntries)
00047 {
00048     uint64_t firstIndex = startIndex + entries.size();
00049     uint64_t lastIndex = firstIndex + newEntries.size() - 1;

00094 void
00095 MemoryLog::truncatePrefix(uint64_t firstIndex)
00096 {
00097     if (firstIndex > startIndex) {
00098         // Erase log entries in range [startIndex, firstIndex), so deque
00099         // offsets in range [0, firstIndex - startIndex). Be careful not to
00100         // erase past the end of the deque (STL doesn't check for this).
00101         entries.erase(entries.begin(),
00102                       entries.begin() + std::min(firstIndex - startIndex,
00103                                                  entries.size()));
00104         startIndex = firstIndex;
00105     }
00106 }

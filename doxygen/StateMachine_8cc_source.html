00215 void
00216 StateMachine::apply(const RaftConsensus::Entry& entry)
00217 {
00218     Command::Request command;
00219     if (!Core::ProtoBuf::parse(entry.command, command)) {
00220         PANIC("Failed to parse protobuf for entry %lu",
00221               entry.index);
00222     }
00223     if (command.has_tree()) {
00224         PC::ExactlyOnceRPCInfo rpcInfo = command.tree().exactly_once();
00225         auto it = sessions.find(rpcInfo.client_id());
00226         if (it == sessions.end()) {
00227             // session does not exist
00228         } else {
00229             // session exists
00230             Session& session = it->second;
00231             expireResponses(session, rpcInfo.first_outstanding_rpc());
00232             if (rpcInfo.rpc_number() < session.firstOutstandingRPC) {
00233                 // response already discarded, do not re-apply
00234             } else {
00235                 auto inserted = session.responses.insert(
00236                                                 {rpcInfo.rpc_number(), {}});
00237                 if (inserted.second) {
00238                     // response not found, apply and save it
00239                     Tree::ProtoBuf::readWriteTreeRPC(
00240                         tree,
00241                         command.tree(),
00242                         *inserted.first->second.mutable_tree());
00243                     session.lastModified = entry.clusterTime;
00244                 } else {
00245                     // response exists, do not re-apply
00246                 }
00247             }
00248         }
00249     } else if (command.has_open_session()) {
00250         uint64_t clientId = entry.index;
00251         Session& session = sessions.insert({clientId, {}}).first->second;
00252         session.lastModified = entry.clusterTime;
00253     } else if (command.has_advance_version()) {
00254         uint16_t requested = Core::Util::downCast<uint16_t>(
00255                 command.advance_version(). requested_version());
00256         uint16_t running = getVersion(entry.index - 1);
00257         if (requested < running) {
00258             WARNING("Rejecting downgrade of state machine version "
00259                     "(running version %u but command at log index %lu wants "
00260                     "to switch to version %u)",
00261                     running,
00262                     entry.index,
00263                     requested);
00264             ++numRejectedAdvanceVersionEntries;
00265         } else if (requested > running) {
00266             if (requested > MAX_SUPPORTED_VERSION) {
00267                 PANIC("Cannot upgrade state machine to version %u (from %u) "
00268                       "because this code only supports up to version %u",
00269                       requested,
00270                       running,
00271                       MAX_SUPPORTED_VERSION);
00272             } else {
00273                 // someday, maybe we'll get here
00274                 // versionHistory.insert(...)
00275                 PANIC("state machine version > 1 not supported");
00276             }
00277             ++numSuccessfulAdvanceVersionEntries;
00278         } else { // requested == running
00279             // nothing to do
00280             // If this stat is high, see note in RaftConsensus.cc.
00281             ++numRedundantAdvanceVersionEntries;
00282         }
00283         ++numTotalAdvanceVersionEntries;
00284     } else { // unknown command
00285         // This could be because the state machine hasn't been upgraded yet to
00286         // handle the command. These are (deterministically) ignored by all
00287         // state machines running the current version.
00288         warnUnknownRequest(command);
00289     }
00290 }

00292 void
00293 StateMachine::applyThreadMain()
00294 {
00295     Core::ThreadId::setName("StateMachine");
00296     try {
00297         while (true) {
00298             RaftConsensus::Entry entry = consensus->getNextEntry(lastIndex);
00299             std::lock_guard<std::mutex> lockGuard(mutex);
00300             switch (entry.type) {
00301                 case RaftConsensus::Entry::SKIP:
00302                     break;
00303                 case RaftConsensus::Entry::DATA:
00304                     apply(entry);
00305                     break;
00306                 case RaftConsensus::Entry::SNAPSHOT:
00307                     NOTICE("Loading snapshot through entry %lu into state "
00308                            "machine", entry.index);
00309                     loadSnapshot(*entry.snapshotReader);
00310                     NOTICE("Done loading snapshot");
00311                     break;
00312             }
00313             expireSessions(entry.clusterTime);
00314             lastIndex = entry.index;
00315             entriesApplied.notify_all();
00316             if (shouldTakeSnapshot(lastIndex))
00317                 snapshotSuggested.notify_all();
00318         }
00319     } catch (const Core::Util::ThreadInterruptedException&) {
00320         NOTICE("exiting");
00321         std::lock_guard<std::mutex> lockGuard(mutex);
00322         exiting = true;
00323         entriesApplied.notify_all();
00324         snapshotSuggested.notify_all();
00325         snapshotStarted.notify_all();
00326         killSnapshotProcess(Core::HoldingMutex(lockGuard));
00327     }
00328 }

00493 bool
00494 StateMachine::shouldTakeSnapshot(uint64_t lastIncludedIndex) const
00495 {
00496     SnapshotStats::SnapshotStats stats = consensus->getSnapshotStats();
00497 
00498     // print every 10% but not at 100% because then we'd be printing all the
00499     // time
00500     uint64_t curr = 0;
00501     if (lastIncludedIndex > stats.last_snapshot_index())
00502         curr = lastIncludedIndex - stats.last_snapshot_index();
00503     uint64_t prev = curr - 1;
00504     uint64_t logEntries = stats.last_log_index() - stats.last_snapshot_index();
00505     if (curr != logEntries &&
00506         10 * prev / logEntries != 10 * curr / logEntries) {
00507         NOTICE("Have applied %lu%% of the %lu total log entries",
00508                100 * curr / logEntries,
00509                logEntries);
00510     }
00511 
00512     if (stats.log_bytes() < snapshotMinLogSize)
00513         return false;
00514     if (stats.log_bytes() < stats.last_snapshot_bytes() * snapshotRatio)
00515         return false;
00516     if (lastIncludedIndex < stats.last_snapshot_index())
00517         return false;
00518     if (lastIncludedIndex < stats.last_log_index() * 3 / 4)
00519         return false;
00520     return true;
00521 }

00538 void

00540 {
00541     Core::ThreadId::setName("SnapshotStateMachineWatchdog");
00542     using Core::StringUtil::toString;
00543 
00544     // The snapshot process that this thread is currently tracking, based on
00545     // numSnapshotsAttempted. If set to ~0UL, this thread is not currently
00546     // tracking a snapshot process.
00547     uint64_t tracking = ~0UL;
00548     // The value of writer->sharedBytesWritten at the "start" time.
00549     uint64_t startProgress = 0;
00550     // The time at the "start" time.
00551     TimePoint startTime = TimePoint::min();

00553     const std::chrono::nanoseconds zero = std::chrono::nanoseconds::zero();

00555     while (!exiting) {
00556         std::unique_lock<std::mutex> lockGuard(mutex);
00557         TimePoint waitUntil = TimePoint::max();
00558         TimePoint now = Clock::now();
00559 
00560         if (childPid > 0) { // there is some child process
00561             uint64_t currentProgress = *writer->sharedBytesWritten.value;
00562             if (tracking == numSnapshotsAttempted) { // tracking current child
00563                 if (snapshotWatchdogInterval != zero &&
00564                     now >= startTime + snapshotWatchdogInterval) { // check
00565                     if (currentProgress == startProgress) {
00566                         ERROR("Snapshot process (counter %lu, pid %u) made no "
00567                               "progress for %s. Killing it. If this happens "
00568                               "at all often, you should file a bug to "
00569                               "understand the root cause.",
00570                               numSnapshotsAttempted,
00571                               childPid,
00572                               toString(snapshotWatchdogInterval).c_str());
00573                         killSnapshotProcess(Core::HoldingMutex(lockGuard));
00574                         // Don't kill for another interval,
00575                         // hopefully child will be reaped by then.
00576                     }
00577                     startProgress = currentProgress;
00578                     startTime = now;
00579                 } else {
00580                     // woke up too early, nothing to do
00581                 }
00582             } else { // not yet tracking this child
00583                 VERBOSE("Beginning to track snapshot process "
00584                         "(counter %lu, pid %u)",
00585                         numSnapshotsAttempted,
00586                         childPid);
00587                 tracking = numSnapshotsAttempted;
00588                 startProgress = currentProgress;
00589                 startTime = now;
00590             }
00591             if (snapshotWatchdogInterval != zero)
00592                 waitUntil = startTime + snapshotWatchdogInterval;
00593         } else { // no child process
00594             if (tracking != ~0UL) {
00595                 VERBOSE("Snapshot ended: no longer tracking (counter %lu)",
00596                         tracking);
00597                 tracking = ~0UL;
00598             }
00599         }
00600         snapshotStarted.wait_until(lockGuard, waitUntil);
00601     }

00605 void
00606 StateMachine::takeSnapshot(uint64_t lastIncludedIndex,
00607                            std::unique_lock<std::mutex>& lockGuard)
00608 {
00609     // Open a snapshot file, then fork a child to write a consistent view of
00610     // the state machine to the snapshot file while this process continues
00611     // accepting requests.
00612     writer = consensus->beginSnapshot(lastIncludedIndex);
00613     // Flush the outstanding changes to the snapshot now so that they
00614     // aren't somehow double-flushed later.
00615     writer->flushToOS();
00616 
00617     ++numSnapshotsAttempted;
00618     snapshotStarted.notify_all();
00619 
00620     pid_t pid = fork();
00621     if (pid == -1) { // error
00622         PANIC("Couldn't fork: %s", strerror(errno));
00623     } else if (pid == 0) { // child
00624         Core::Debug::processName += "-child";
00625         usleep(stateMachineChildSleepMs * 1000); // for testing purposes
00626         if (snapshotBlockPercentage > 0) { // for testing purposes
00627             if (Core::Random::randomRange(0, 100) < snapshotBlockPercentage) {
00628                 WARNING("Purposely deadlocking child (probability is %lu%%)",
00629                         snapshotBlockPercentage);
00630                 std::mutex mutex;
00631                 mutex.lock();

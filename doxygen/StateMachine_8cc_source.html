00190                         tree,
00191                         command.tree(),
00192                         *inserted.first->second.mutable_tree());
00193                 } else {
00194                     // response exists, do not re-apply
00195                 }
00196             }
00197         }
00198     } else if (command.has_open_session()) {
00199         uint64_t clientId = entry.index;
00200         session = &sessions.insert({clientId, {}}).first->second;

00219         while (true) {

00233                     NOTICE("Done loading snapshot");
00234                     break;

00237             lastIndex = entry.index;
00238             entriesApplied.notify_all();
00239             if (shouldTakeSnapshot(lastIndex))

00244         std::lock_guard<std::mutex> lockGuard(mutex);
00245         exiting = true;

00348 bool
00349 StateMachine::shouldTakeSnapshot(uint64_t lastIncludedIndex) const
00350 {
00351     SnapshotStats::SnapshotStats stats = consensus->getSnapshotStats();
00352 
00353     // print every 10% but not at 100% because then we'd be printing all the
00354     // time
00355     uint64_t curr = 0;
00356     if (lastIncludedIndex > stats.last_snapshot_index())
00357         curr = lastIncludedIndex - stats.last_snapshot_index();
00358     uint64_t prev = curr - 1;
00359     uint64_t logEntries = stats.last_log_index() - stats.last_snapshot_index();
00360     if (curr != logEntries &&
00361         10 * prev / logEntries != 10 * curr / logEntries) {
00362         NOTICE("Have applied %lu%% of the %lu total log entries",
00363                100 * curr / logEntries,
00364                logEntries);
00365     }
00366 
00367     if (stats.log_bytes() < snapshotMinLogSize)
00368         return false;
00369     if (stats.log_bytes() < stats.last_snapshot_bytes() * snapshotRatio)
00370         return false;
00371     if (lastIncludedIndex < stats.last_snapshot_index())
00372         return false;
00373     if (lastIncludedIndex < stats.last_log_index() * 3 / 4)
00374         return false;
00375     return true;

00421                         ERROR("Snapshot process (counter %lu, pid %u) made no "
00422                               "progress for %s. Killing it. If this happens "
00423                               "at all often, you should file a bug to "
00424                               "understand the root cause.",
00425                               numSnapshotsAttempted,

00429                         // Don't kill for another interval,
00430                         // hopefully child will be reaped by then.
00431                     }
00432                     startProgress = currentProgress;
00433                     startTime = now;
00434                 } else {
00435                     // woke up too early, nothing to do
00436                 }
00437             } else { // not yet tracking this child
00438                 VERBOSE("Beginning to track snapshot process "
00439                         "(counter %lu, pid %u)",
00440                         numSnapshotsAttempted,
00441                         childPid);

00443                 startProgress = currentProgress;
00444                 startTime = now;

00446             if (snapshotWatchdogInterval != zero)

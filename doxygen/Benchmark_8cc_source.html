00176 writeThreadMain(uint64_t id,

00180                 const std::string& value,
00181                 std::atomic<bool>& exit,

00183 {
00184     uint64_t numWrites = options.totalWrites / options.writers;
00185     // assign any odd leftover writes in a balanced way
00186     if (options.totalWrites - numWrites * options.writers > id)
00187         numWrites += 1;
00188     for (uint64_t i = 0; i < numWrites; ++i) {
00189         if (exit)
00190             break;
00191         tree.writeEx(key, value);
00192         writesDone = i + 1;

00207 /**
00208  * Main function for the timer thread, whose job is to wait until a particular
00209  * timeout elapses and then set 'exit' to true.
00210  * \param timeout
00211  *      Seconds to wait before setting exit to true.
00212  * \param[in,out] exit
00213  *      If this is set to true from another thread, the timer thread will exit
00214  *      soonish. Also, if the timeout elapses, the timer thread will set this
00215  *      to true and exit.
00216  */
00217 void
00218 timerThreadMain(uint64_t timeout, std::atomic<bool>& exit)
00219 {
00220     uint64_t start = timeNanos();
00221     while (!exit) {
00222         usleep(50 * 1000);
00223         if ((timeNanos() - start) > timeout) {
00224             exit = true;
00225         }
00226     }
00227 }
